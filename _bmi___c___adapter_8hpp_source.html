517  inline void execModuleRegistration() {
518  if (get_dyn_lib_handle() == nullptr) {
519  dynamic_library_load();
520  }
521  void *symbol;
522  C_Bmi *(*dynamic_register_bmi)(C_Bmi *model);
523 
524  try {
525  // Acquire the BMI struct func pointer registration function
526  symbol = dynamic_load_symbol(get_bmi_registration_function());
527  dynamic_register_bmi = (C_Bmi *(*)(C_Bmi *)) symbol;
528  // Call registration function, which (for C libs) sets up object's pointed-to member BMI struct
529  dynamic_register_bmi(bmi_model.get());
530  }
531  catch (const ::external::ExternalIntegrationException &e) {
532  // "Override" the default message in this case
533  this->init_exception_msg =
534  "Cannot init " + this->model_name + " without valid library registration function: " +
535  this->init_exception_msg;
536  throw ::external::ExternalIntegrationException(this->init_exception_msg);
