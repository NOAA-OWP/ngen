464 inline void execModuleRegistration() {
465 if (get_dyn_lib_handle() == nullptr) {
466 dynamic_library_load();
467 }
468 void *symbol;
469 C_Bmi *(*dynamic_register_bmi)(C_Bmi *model);
470
471 try {
472 // Acquire the BMI struct func pointer registration function
473 symbol = dynamic_load_symbol(get_bmi_registration_function());
474 dynamic_register_bmi = (C_Bmi *(*)(C_Bmi *)) symbol;
475 // Call registration function, which (for C libs) sets up object's pointed-to member BMI struct
476 dynamic_register_bmi(bmi_model.get());
477 }
478 catch (const ::external::ExternalIntegrationException &e) {
479 // "Override" the default message in this case
480 this->init_exception_msg =
481 "Cannot init " + this->model_name + " without valid library registration function: " +
482 this->init_exception_msg;
483 throw ::external::ExternalIntegrationException(this->init_exception_msg);
484 }
485 }

552 inline void construct_and_init_backing_model_for_type() {
553 if (model_initialized)
554 return;
555 bmi_model = std::make_unique<C_Bmi>(C_Bmi());
556 execModuleRegistration();
557 int init_result = bmi_model->initialize(bmi_model.get(), bmi_init_config.c_str());
558 if (init_result != BMI_SUCCESS) {
559 init_exception_msg = "Failure when attempting to initialize " + model_name;
560 throw models::external::State_Exception(init_exception_msg);
