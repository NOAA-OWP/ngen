471 inline void execModuleRegistration() {
472 if (get_dyn_lib_handle() == nullptr) {
473 dynamic_library_load();
474 }
475 void *symbol;
476 C_Bmi *(*dynamic_register_bmi)(C_Bmi *model);
477
478 try {
479 // Acquire the BMI struct func pointer registration function
480 symbol = dynamic_load_symbol(get_bmi_registration_function());
481 dynamic_register_bmi = (C_Bmi *(*)(C_Bmi *)) symbol;
482 // Call registration function, which (for C libs) sets up object's pointed-to member BMI struct
483 dynamic_register_bmi(bmi_model.get());
484 }
485 catch (const ::external::ExternalIntegrationException &e) {
486 // "Override" the default message in this case
487 this->init_exception_msg =
488 "Cannot init " + this->model_name + " without valid library registration function: " +
489 this->init_exception_msg;
490 throw ::external::ExternalIntegrationException(this->init_exception_msg);
491 }
492 }

559 inline void construct_and_init_backing_model_for_type() {

562 bmi_model = std::make_unique<C_Bmi>(C_Bmi());
563 execModuleRegistration();
564 int init_result = bmi_model->initialize(bmi_model.get(), bmi_init_config.c_str());
