475 inline void execModuleRegistration() {
476 if (get_dyn_lib_handle() == nullptr) {
477 dynamic_library_load();
478 }
479 void *symbol;
480 C_Bmi *(*dynamic_register_bmi)(C_Bmi *model);
481
482 try {
483 // Acquire the BMI struct func pointer registration function
484 symbol = dynamic_load_symbol(get_bmi_registration_function());
485 dynamic_register_bmi = (C_Bmi *(*)(C_Bmi *)) symbol;
486 // Call registration function, which (for C libs) sets up object's pointed-to member BMI struct
487 dynamic_register_bmi(bmi_model.get());
488 }
489 catch (const ::external::ExternalIntegrationException &e) {
490 // "Override" the default message in this case
491 this->init_exception_msg =
492 "Cannot init " + this->model_name + " without valid library registration function: " +
493 this->init_exception_msg;
494 throw ::external::ExternalIntegrationException(this->init_exception_msg);
495 }
496 }
