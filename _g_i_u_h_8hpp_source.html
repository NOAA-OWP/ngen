60
61 double ordinate_sum = 0.0;

63 ordinates.push_back(ordinate_sum);
64 ordinate_times.push_back(time_sum);
65 for (unsigned int i = 1; i < ordinates.size(); ++i) {
66 ordinate_sum += incremental_runoffs[i-1];

70 }

72 return giuh_kernel_impl(catchment_id, comid, ordinate_times, ordinates, interpolation_regularity_seconds);
