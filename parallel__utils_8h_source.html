53  bool mpiSyncStatusAnd(bool status, int mpi_rank, int mpi_num_procs, const std::string &taskDesc) {
54  // Expect 0 is good and 1 is no good for goodCode
55  // TODO: assert this in constructor or somewhere, or maybe just in a unit test
56  unsigned short codeBuffer;
57  bool printMessage = !taskDesc.empty();
58  // For the other ranks, start by properly setting the status code value in the buffer and send to rank 0
59  if (mpi_rank != 0) {
60  codeBuffer = status ? MPI_HF_SUB_CODE_GOOD : MPI_HF_SUB_CODE_BAD;
61  MPI_Send(&codeBuffer, 1, MPI_UNSIGNED_SHORT, 0, 0, MPI_COMM_WORLD);
62  }
63  // In rank 0, the first step is to receive and process codes from the other ranks into unified global status
64  else {
65  for (int i = 1; i < mpi_num_procs; ++i) {
66  MPI_Recv(&codeBuffer, 1, MPI_UNSIGNED_SHORT, i, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
67  // If any is ever "not good", overwrite status to be "false"
68  if (codeBuffer != MPI_HF_SUB_CODE_GOOD) {
69  if (printMessage) {
70  std::cout << "Rank " << i << " not successful/ready after " << taskDesc << std::endl;
71  }
72  status = false;
73  }
74  }
75  // Rank 0 must also now prepare the codeBuffer value for broadcasting the global status
76  codeBuffer = status ? MPI_HF_SUB_CODE_GOOD : MPI_HF_SUB_CODE_BAD;
77  }
78  // Execute broadcast of global status rooted at rank 0
79  MPI_Bcast(&codeBuffer, mpi_num_procs - 1, MPI_UNSIGNED_SHORT, 0, MPI_COMM_WORLD);
80  return codeBuffer == MPI_HF_SUB_CODE_GOOD;
81  }

157  void get_hosts_array(int mpi_rank, int mpi_num_procs, int *host_array) {
158  const int ROOT_RANK = 0;
159  // These are the lengths of the (trimmed) C-string representations of the hostname for each rank
160  int actualHostnameCStrLength[mpi_num_procs];
161  // Initialize to -1 to represent unknown
162  for (int i = 0; i < mpi_num_procs; ++i) {
163  actualHostnameCStrLength[i] = -1;
164  }
165 
166  // Get this rank's hostname (things should never be longer than 256)
167  char myhostname[256];
168  gethostname(myhostname, 256);
169 
170  // Set the one for this rank
171  actualHostnameCStrLength[mpi_rank] = std::strlen(myhostname);
172 
173  // First, gather the hostname string lengths
174  MPI_Gather(&actualHostnameCStrLength[mpi_rank], 1, MPI_INT, actualHostnameCStrLength, 1, MPI_INT, ROOT_RANK,
175  MPI_COMM_WORLD);
176  // Per-rank start offsets/displacements in our hostname strings gather buffer.
177  int recvDisplacements[mpi_num_procs];
178  int totalLength = 0;
179  for (int i = 0; i < mpi_num_procs; ++i) {
180  // Displace each rank's string by the sum of the length of the previous rank strings
181  recvDisplacements[i] = totalLength;
182  // Adding the extra to make space for the \0
183  actualHostnameCStrLength[i] += 1;
184  // Then update the total length
185  totalLength += actualHostnameCStrLength[i];
186 
187  }
188  // Now we can create our buffer array and gather the hostname strings into it
189  char hostnames[totalLength];
190  MPI_Gatherv(myhostname, actualHostnameCStrLength[mpi_rank], MPI_CHAR, hostnames, actualHostnameCStrLength,
191  recvDisplacements, MPI_CHAR, ROOT_RANK, MPI_COMM_WORLD);
192 
193  if (mpi_rank == ROOT_RANK) {
194  host_array[0] = 0;
195  int next_host_id = 1;
196 
197  int rank_with_matching_hostname;
198  char *checked_rank_hostname, *known_rank_hostname;
199 
200  for (int rank_being_check = 0; rank_being_check < mpi_num_procs; ++rank_being_check) {
201  // Set this as negative initially for each rank check to indicate no match found (at least yet)
202  rank_with_matching_hostname = -1;
203  // Get a C-string pointer for this rank's hostname, offset by the appropriate displacement
204  checked_rank_hostname = &hostnames[recvDisplacements[rank_being_check]];
205 
206  // Assume that hostnames for any ranks less than the current rank being check are already known
207  for (int known_rank = 0; known_rank < rank_being_check; ++known_rank) {
208  // Get the right C-string pointer for the current known rank's hostname also
209  known_rank_hostname = &hostnames[recvDisplacements[known_rank]];
210  // Compare the hostnames, setting and breaking if a match is found
211  if (std::strcmp(known_rank_hostname, checked_rank_hostname) == 0) {
212  rank_with_matching_hostname = known_rank;
213  break;
214  }
215  }
216  // This indicates this rank had no earlier rank with a matching hostname.
217  if (rank_with_matching_hostname < 0) {
218  // Assign new host id, then increment what the next id will be
219  host_array[rank_being_check] = next_host_id++;
220  }
221  else {
222  host_array[rank_being_check] = host_array[rank_with_matching_hostname];
223  }
224  }
225  }
226  // Now, broadcast the results out
227  MPI_Bcast(host_array, mpi_num_procs, MPI_INT, 0, MPI_COMM_WORLD);
228  }

240  bool mpi_send_text_file(const char *fileName, const int mpi_rank, const int destRank) {
241  int bufSize = 4096;
242  char buf[bufSize];
243  int code;
244  // How much has been transferred so far
245  int totalNumTransferred = 0;
246 
247  FILE *file = fopen(fileName, "r");
248 
249  // Transmit error code instead of expected size and return false if file can't be opened
250  if (file == NULL) {
251  // TODO: output error message
252  code = -1;
253  MPI_Send(&code, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
254  return false;
255  }
256 
257  // Send expected size to start
258  MPI_Send(&bufSize, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
259 
260  // Then get back expected size to infer other side is good to go
261  MPI_Recv(&code, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
262  if (code != bufSize) {
263  // TODO: output error message
264  fclose(file);
265  return false;
266  }
267  int continueCode = 1;
268  // Then while there is more of the file to read and send, read the next batch and ...
269  while (fgets(buf, bufSize, file) != NULL) {
270  // Indicate we are ready to continue sending data
271  MPI_Send(&continueCode, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
272 
273  // Send this batch
274  MPI_Send(buf, bufSize, MPI_CHAR, destRank, NGEN_MPI_DATA_TAG, MPI_COMM_WORLD);
275 
276  // Then get back a code, which will be -1 if bad and need to exit and otherwise good
277  MPI_Recv(&code, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
278  if (code < 0) {
279  // TODO: output error message
280  fclose(file);
281  return false;
282  }
283  }
284  // Once there is no more file to read and send, we should stop continuing
285  continueCode = 0;
286  MPI_Send(&continueCode, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
287  // Expect to get back a code of 0
288  MPI_Recv(&code, 1, MPI_INT, destRank, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
289  fclose(file);
290  return code == 0;
291  }

305  bool mpi_recv_text_file(const char *fileName, const int mpi_rank, const int srcRank) {
306  int bufSize, writeCode;
307  // Receive expected buffer size to start
308  MPI_Recv(&bufSize, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
309 
310  // If the sending side couldn't open the file, then immediately return false
311  if (bufSize == -1) {
312  // TODO: output error
313  return false;
314  }
315 
316  // Try to open recv file ...
317  FILE *file = fopen(fileName, "w");
318  // ... and let sending size know whether this was successful by sending error code if not ...
319  if (file == NULL) {
320  // TODO: output error message
321  bufSize = -1;
322  MPI_Send(&bufSize, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
323  return false;
324  }
325 
326  // Send back the received buffer it if file opened, confirming things are good to go for transfer
327  MPI_Send(&bufSize, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
328 
329  // How much has been transferred so far
330  int totalNumTransferred = 0;
331  char buf[bufSize];
332 
333  int continueCode;
334 
335  while (true) {
336  // Make sure the other side wants to continue sending data
337  MPI_Recv(&continueCode, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
338  if (continueCode <= 0)
339  break;
340 
341  MPI_Recv(buf, bufSize, MPI_CHAR, srcRank, NGEN_MPI_DATA_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
342 
343  writeCode = fputs(buf, file);
344  MPI_Send(&writeCode, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
345 
346  if (writeCode < 0) {
347  fclose(file);
348  return false;
349  }
350  }
351 
352  fclose(file);
353  MPI_Send(&continueCode, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
354  return true;
355  }

407  bool distribute_subdivided_hydrofabric_files(const std::string &baseCatchmentFile, const std::string &baseNexusFile,
408  const int sendingRank, const int mpi_rank, const int mpi_num_procs,
409  const int *hostIdForRank, bool syncReturnStatus, bool blockAll)
410  {
411  // Start with status as good
412  bool isGood = true;
413  // FIXME: For now, just have rank 0 send everything, but optimize with multiple procs or threads later
414  // Only need to process this if sending rank or a receiving ranks (i.e., not on same host as sending rank)
415  if (mpi_rank == sendingRank || hostIdForRank[mpi_rank] != hostIdForRank[sendingRank]) {
416  // Have the sending rank send out all files
417  if (mpi_rank == sendingRank) {
418  // In rank 0, for all the other ranks ...
419  for (int otherRank = 0; otherRank < mpi_num_procs; ++otherRank) {
420  // If another rank is on a different host (note that this covers otherRank == sendingRank case) ...
421  if (hostIdForRank[otherRank] != hostIdForRank[mpi_rank]) {
422  // ... then send that rank its rank-specific catchment and nexus files
423  std::string catFileToSend = baseCatchmentFile + "." + std::to_string(otherRank);
424  std::string nexFileToSend = baseNexusFile + "." + std::to_string(otherRank);
425  // Note that checking previous isGood is necessary here because of loop
426  isGood = isGood && mpi_send_text_file(catFileToSend.c_str(), mpi_rank, otherRank);
427  isGood = isGood && mpi_send_text_file(nexFileToSend.c_str(), mpi_rank, otherRank);
428  }
429  }
430  }
431  else {
432  // For a rank not on the same host as the sending rank, receive the transmitted file
433  std::string catFileToReceive = baseCatchmentFile + "." + std::to_string(mpi_rank);
434  std::string nexFileToReceive = baseNexusFile + "." + std::to_string(mpi_rank);
435  // Note that, unlike a bit earlier, don't need to check prior isGood in 1st receive, because not in loop
436  isGood = mpi_recv_text_file(catFileToReceive.c_str(), mpi_rank, sendingRank);
437  isGood = isGood && mpi_recv_text_file(nexFileToReceive.c_str(), mpi_rank, sendingRank);
438  }
439  }
440 
441  // Wait when appropriate
442  if (blockAll) { MPI_Barrier(MPI_COMM_WORLD); }
443 
444  // Sync status among the ranks also, if appropriate
445  if (syncReturnStatus) {
446  return mpiSyncStatusAnd(isGood, mpi_rank, mpi_num_procs, "distributing subdivided hydrofabric files");
447  }
448  // Otherwise, just return the local status value
449  else {
450  return isGood;
451  }
452  }

469  bool subdivide_hydrofabric(int mpi_rank, int mpi_num_procs, const string &catchmentDataFile,
470  const string &nexusDataFile, const string &partitionConfigFile)
471  {
472  // Track whether things are good, meaning ok to continue and, at the end, whether successful
473  // Start with a value of true
474  bool isGood = true;
475 
476  #ifdef ACTIVATE_PYTHON
477  std::unique_ptr<utils::PyHydrofabricSubsetter> subdivider;
478  // Have rank 0 handle the generation task for all files/partitions
479  if (mpi_rank == 0) {
480  try {
481  subdivider = std::make_unique<utils::PyHydrofabricSubsetter>(catchmentDataFile, nexusDataFile,
482  partitionConfigFile);
483  }
484  catch (const std::exception &e) {
485  std::cerr << e.what() << std::endl;
486  // Set not good if the subdivider object couldn't be instantiated
487  isGood = false;
488  }
489  }
490 
491  // Sync ranks and bail if any aren't ready to proceed for any reason
492  if (!mpiSyncStatusAnd(isGood, mpi_rank, mpi_num_procs, "initializing hydrofabric subdivider")) {
493  return false;
494  }
495 
496  if (mpi_rank == 0) {
497  // Try to perform the subdividing
498  try {
499  isGood = subdivider->execSubdivision();
500  }
501  catch (const std::exception &e) {
502  std::cerr << e.what() << std::endl;
503  // Set not good if the subdivider object couldn't be instantiated
504  isGood = false;
505  }
506  }
507 
508  #else // i.e., ifndef ACTIVATE_PYTHON
509  // Without Python available, there is no way external subdivide package can be used (i.e., not good)
510  isGood = false;
511  std::cerr << "Driver is unable to perform required hydrofabric subdividing when Python integration is not active." << std::endl;
512  #endif // ACTIVATE_PYTHON
513 
514  // Sync ranks again here on whether subdividing was successful, having them all exit at this point if not
515  if (!mpiSyncStatusAnd(isGood, mpi_rank, mpi_num_procs, "executing hydrofabric subdivision")) {
516  return false;
517  }
518  // But if the subdividing went fine ...
519  else {
520  // ... figure out what ranks are on hosts with each other by getting an id for host of each rank
521  int hostIdForRank[mpi_num_procs];
522  get_hosts_array(mpi_rank, mpi_num_procs, hostIdForRank);
523 
524  // ... then (when necessary) transferring files around
525  return distribute_subdivided_hydrofabric_files(catchmentDataFile, nexusDataFile, 0, mpi_rank,
526  mpi_num_procs, hostIdForRank, true, true);
527 
528  }
529  }
