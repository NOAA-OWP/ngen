51 bool mpiSyncStatusAnd(bool status, int mpi_rank, int mpi_num_procs, const std::string &taskDesc) {
52
53 // Expect 0 is good and 1 is no good for goodCode
54 // TODO: assert this in constructor or somewhere, or maybe just in a unit test
55 unsigned short codeBuffer;
56 bool printMessage = !taskDesc.empty();
57 // For the other ranks, start by properly setting the status code value in the buffer and send to rank 0
58 if (mpi_rank != 0) {
59 codeBuffer = status ? MPI_HF_SUB_CODE_GOOD : MPI_HF_SUB_CODE_BAD;
60 MPI_Send(&codeBuffer, 1, MPI_UNSIGNED_SHORT, 0, 0, MPI_COMM_WORLD);
61 }
62 // In rank 0, the first step is to receive and process codes from the other ranks into unified global status
63 else {
64 for (int i = 1; i < mpi_num_procs; ++i) {
65 MPI_Recv(&codeBuffer, 1, MPI_UNSIGNED_SHORT, i, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
66 // If any is ever "not good", overwrite status to be "false"
67 if (codeBuffer != MPI_HF_SUB_CODE_GOOD) {
68 if (printMessage) {
69 std::cout << "Rank " << i << " not successful/ready after " << taskDesc << std::endl;
70 }
71 status = false;
72 }
73 }
74 // Rank 0 must also now prepare the codeBuffer value for broadcasting the global status
75 codeBuffer = status ? MPI_HF_SUB_CODE_GOOD : MPI_HF_SUB_CODE_BAD;
76 }
77
78 // Execute broadcast of global status rooted at rank 0
79 MPI_Bcast(&codeBuffer, 1, MPI_UNSIGNED_SHORT, 0, MPI_COMM_WORLD);
80 return codeBuffer == MPI_HF_SUB_CODE_GOOD;
81 }

158 void get_hosts_array(int mpi_rank, int mpi_num_procs, int *host_array) {
159 const int ROOT_RANK = 0;
160 // These are the lengths of the (trimmed) C-string representations of the hostname for each rank
161 int actualHostnameCStrLength[mpi_num_procs];
162 // Initialize to -1 to represent unknown
163 for (int i = 0; i < mpi_num_procs; ++i) {
164 actualHostnameCStrLength[i] = -1;
165 }
166
167 // Get this rank's hostname (things should never be longer than 256)
168 char myhostname[256];
169 gethostname(myhostname, 256);
170
171 // Set the one for this rank
172 actualHostnameCStrLength[mpi_rank] = std::strlen(myhostname);
173
174 // First, gather the hostname string lengths
175 MPI_Gather(&actualHostnameCStrLength[mpi_rank], 1, MPI_INT, actualHostnameCStrLength, 1, MPI_INT, ROOT_RANK,
176 MPI_COMM_WORLD);
177 // Per-rank start offsets/displacements in our hostname strings gather buffer.
178 int recvDisplacements[mpi_num_procs];
179 int totalLength = 0;
180 for (int i = 0; i < mpi_num_procs; ++i) {
181 // Displace each rank's string by the sum of the length of the previous rank strings
182 recvDisplacements[i] = totalLength;
183 // Adding the extra to make space for the \0
184 actualHostnameCStrLength[i] += 1;
185 // Then update the total length
186 totalLength += actualHostnameCStrLength[i];
187
188 }
189 // Now we can create our buffer array and gather the hostname strings into it
190 char hostnames[totalLength];
191 MPI_Gatherv(myhostname, actualHostnameCStrLength[mpi_rank], MPI_CHAR, hostnames, actualHostnameCStrLength,
192 recvDisplacements, MPI_CHAR, ROOT_RANK, MPI_COMM_WORLD);
193
194 if (mpi_rank == ROOT_RANK) {
195 host_array[0] = 0;
196 int next_host_id = 1;
197
198 int rank_with_matching_hostname;
199 char *checked_rank_hostname, *known_rank_hostname;
200
201 for (int rank_being_check = 0; rank_being_check < mpi_num_procs; ++rank_being_check) {
202 // Set this as negative initially for each rank check to indicate no match found (at least yet)
203 rank_with_matching_hostname = -1;
204 // Get a C-string pointer for this rank's hostname, offset by the appropriate displacement
205 checked_rank_hostname = &hostnames[recvDisplacements[rank_being_check]];
206
207 // Assume that hostnames for any ranks less than the current rank being check are already known
208 for (int known_rank = 0; known_rank < rank_being_check; ++known_rank) {
209 // Get the right C-string pointer for the current known rank's hostname also
210 known_rank_hostname = &hostnames[recvDisplacements[known_rank]];
211 // Compare the hostnames, setting and breaking if a match is found
212 if (std::strcmp(known_rank_hostname, checked_rank_hostname) == 0) {
213 rank_with_matching_hostname = known_rank;
214 break;
215 }
216 }
217 // This indicates this rank had no earlier rank with a matching hostname.
218 if (rank_with_matching_hostname < 0) {
219 // Assign new host id, then increment what the next id will be
220 host_array[rank_being_check] = next_host_id++;
221 }
222 else {
223 host_array[rank_being_check] = host_array[rank_with_matching_hostname];
224 }
225 }
226 }
227 // Now, broadcast the results out
228 MPI_Bcast(host_array, mpi_num_procs, MPI_INT, 0, MPI_COMM_WORLD);
229 }

241 bool mpi_send_text_file(const char *fileName, const int mpi_rank, const int destRank) {
242 int bufSize = 4096;
243 char buf[bufSize];
244 int code;
245 // How much has been transferred so far
246 int totalNumTransferred = 0;
247
248 FILE *file = fopen(fileName, "r");
249
250 // Transmit error code instead of expected size and return false if file can't be opened
251 if (file == NULL) {
252 // TODO: output error message
253 code = -1;
254 MPI_Send(&code, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
255 return false;
256 }
257
258 // Send expected size to start
259 MPI_Send(&bufSize, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
260
261 // Then get back expected size to infer other side is good to go
262 MPI_Recv(&code, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
263 if (code != bufSize) {
264 // TODO: output error message
265 fclose(file);
266 return false;
267 }
268 int continueCode = 1;
269 // Then while there is more of the file to read and send, read the next batch and ...
270 while (fgets(buf, bufSize, file) != NULL) {
271 // Indicate we are ready to continue sending data
272 MPI_Send(&continueCode, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
273
274 // Send this batch
275 MPI_Send(buf, bufSize, MPI_CHAR, destRank, NGEN_MPI_DATA_TAG, MPI_COMM_WORLD);
276
277 // Then get back a code, which will be -1 if bad and need to exit and otherwise good
278 MPI_Recv(&code, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
279 if (code < 0) {
280 // TODO: output error message
281 fclose(file);
282 return false;
283 }
284 }
285 // Once there is no more file to read and send, we should stop continuing
286 continueCode = 0;
287 MPI_Send(&continueCode, 1, MPI_INT, destRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
288 // Expect to get back a code of 0
289 MPI_Recv(&code, 1, MPI_INT, destRank, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
290 fclose(file);
291 return code == 0;
292 }

306 bool mpi_recv_text_file(const char *fileName, const int mpi_rank, const int srcRank) {
307 int bufSize, writeCode;
308 // Receive expected buffer size to start
309 MPI_Recv(&bufSize, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
310
311 // If the sending side couldn't open the file, then immediately return false
312 if (bufSize == -1) {
313 // TODO: output error
314 return false;
315 }
316
317 // Try to open recv file ...
318 FILE *file = fopen(fileName, "w");
319 // ... and let sending size know whether this was successful by sending error code if not ...
320 if (file == NULL) {
321 // TODO: output error message
322 bufSize = -1;
323 MPI_Send(&bufSize, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
324 return false;
325 }
326
327 // Send back the received buffer it if file opened, confirming things are good to go for transfer
328 MPI_Send(&bufSize, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
329
330 // How much has been transferred so far
331 int totalNumTransferred = 0;
332 char buf[bufSize];
333
334 int continueCode;
335
336 while (true) {
337 // Make sure the other side wants to continue sending data
338 MPI_Recv(&continueCode, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
339 if (continueCode <= 0)
340 break;
341
342 MPI_Recv(buf, bufSize, MPI_CHAR, srcRank, NGEN_MPI_DATA_TAG, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
343
344 writeCode = fputs(buf, file);
345 MPI_Send(&writeCode, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
346
347 if (writeCode < 0) {
348 fclose(file);
349 return false;
350 }
351 }
352
353 fclose(file);
354 MPI_Send(&continueCode, 1, MPI_INT, srcRank, NGEN_MPI_PROTOCOL_TAG, MPI_COMM_WORLD);
355 return true;
356 }

408 bool distribute_subdivided_hydrofabric_files(const std::string &baseCatchmentFile, const std::string &baseNexusFile,
409 const int sendingRank, const int mpi_rank, const int mpi_num_procs,
410 const int *hostIdForRank, bool syncReturnStatus, bool blockAll)
411 {
412 // Start with status as good
413 bool isGood = true;
414 // FIXME: For now, just have rank 0 send everything, but optimize with multiple procs or threads later
415 // Only need to process this if sending rank or a receiving ranks (i.e., not on same host as sending rank)
416 if (mpi_rank == sendingRank || hostIdForRank[mpi_rank] != hostIdForRank[sendingRank]) {
417 // Have the sending rank send out all files
418 if (mpi_rank == sendingRank) {
419 // In rank 0, for all the other ranks ...
420 for (int otherRank = 0; otherRank < mpi_num_procs; ++otherRank) {
421 // If another rank is on a different host (note that this covers otherRank == sendingRank case) ...
422 if (hostIdForRank[otherRank] != hostIdForRank[mpi_rank]) {
423 // ... then send that rank its rank-specific catchment and nexus files
424 std::string catFileToSend = baseCatchmentFile + "." + std::to_string(otherRank);
425 std::string nexFileToSend = baseNexusFile + "." + std::to_string(otherRank);
426 // Note that checking previous isGood is necessary here because of loop
427 isGood = isGood && mpi_send_text_file(catFileToSend.c_str(), mpi_rank, otherRank);
428 isGood = isGood && mpi_send_text_file(nexFileToSend.c_str(), mpi_rank, otherRank);
429 }
430 }
431 }
432 else {
433 // For a rank not on the same host as the sending rank, receive the transmitted file
434 std::string catFileToReceive = baseCatchmentFile + "." + std::to_string(mpi_rank);
435 std::string nexFileToReceive = baseNexusFile + "." + std::to_string(mpi_rank);
436 // Note that, unlike a bit earlier, don't need to check prior isGood in 1st receive, because not in loop
437 isGood = mpi_recv_text_file(catFileToReceive.c_str(), mpi_rank, sendingRank);
438 isGood = isGood && mpi_recv_text_file(nexFileToReceive.c_str(), mpi_rank, sendingRank);
439 }
440 }
441
442 // Wait when appropriate
443 if (blockAll) { MPI_Barrier(MPI_COMM_WORLD); }
444
445 // Sync status among the ranks also, if appropriate
446 if (syncReturnStatus) {
447 return mpiSyncStatusAnd(isGood, mpi_rank, mpi_num_procs, "distributing subdivided hydrofabric files");
448 }
449 // Otherwise, just return the local status value
450 else {
451 return isGood;
452 }
453 }

470 bool subdivide_hydrofabric(int mpi_rank, int mpi_num_procs, const std::string &catchmentDataFile,
471 const std::string &nexusDataFile, const std::string &partitionConfigFile)
472 {
473 // Track whether things are good, meaning ok to continue and, at the end, whether successful
474 // Start with a value of true
475 bool isGood = true;
476
477 #ifndef ACTIVATE_PYTHON
478 // We can't be good to proceed with this, because Python is not active
479 isGood = false;
480 std::cerr << "Driver is unable to perform required hydrofabric subdividing when Python integration is not active." << std::endl;
481
482
483 // Sync with the rest of the ranks and bail if any aren't ready to proceed for any reason
484 if (!mpiSyncStatusAnd(isGood, mpi_rank, mpi_num_procs, "initializing hydrofabric subdivider")) {
485 return false;
486 }
487 #else // i.e., #ifdef ACTIVATE_PYTHON
488 // Have rank 0 handle the generation task for all files/partitions
489 std::unique_ptr<utils::ngenPy::HydrofabricSubsetter> subdivider;
490 // Have rank 0 handle the generation task for all files/partitions
491 if (mpi_rank == 0) {
492 try {
493 subdivider = std::make_unique<utils::ngenPy::HydrofabricSubsetter>(catchmentDataFile, nexusDataFile,
494 partitionConfigFile);
495 }
496 catch (const std::exception &e) {
497 std::cerr << e.what() << std::endl;
498 // Set not good if the subdivider object couldn't be instantiated
499 isGood = false;
500 }
501 }
502 // Sync ranks and bail if any aren't ready to proceed for any reason
503 if (!mpiSyncStatusAnd(isGood, mpi_rank, mpi_num_procs, "initializing hydrofabric subdivider")) {
504 return false;
505 }
506
507 if (mpi_rank == 0) {
508 // Try to perform the subdividing
509 try {
510 isGood = subdivider->execSubdivision();
511 }
512 catch (const std::exception &e) {
513 std::cerr << e.what() << std::endl;
514 // Set not good if the subdivider object couldn't be instantiated
515 isGood = false;
516 }
517 }
518 // Sync ranks again here on whether subdividing was successful, having them all exit at this point if not
519 if (!mpiSyncStatusAnd(isGood, mpi_rank, mpi_num_procs, "executing hydrofabric subdivision")) {
520 return false;
521 }
522 // But if the subdividing went fine ...
523 else {
524 // ... figure out what ranks are on hosts with each other by getting an id for host of each rank
525 int hostIdForRank[mpi_num_procs];
526 get_hosts_array(mpi_rank, mpi_num_procs, hostIdForRank);
527
528 // ... then (when necessary) transferring files around
529 return distribute_subdivided_hydrofabric_files(catchmentDataFile, nexusDataFile, 0, mpi_rank,
530 mpi_num_procs, hostIdForRank, true, true);
531
532 }
533 #endif // ACTIVATE_PYTHON
534 }
