80  inline void dynamic_library_load() {
81  if (bmi_registration_function.empty()) {
82  this->init_exception_msg =
83  "Can't init " + this->model_name + "; empty name given for library's registration function.";
84  throw std::runtime_error(this->init_exception_msg);
85  }
86  if (dyn_lib_handle != nullptr) {
87  this->output.put("WARNING: ignoring attempt to reload dynamic shared library '" + bmi_lib_file +
88  "' for " + this->model_name);
89  return;

94  throw std::runtime_error(this->init_exception_msg);
95  }
96 
97  // Load up the necessary library dynamically

123  inline void *dynamic_load_symbol(const std::string &symbol_name, bool is_null_valid) {
124  if (dyn_lib_handle == nullptr) {
125  throw std::runtime_error("Cannot load symbol " + symbol_name + " without handle to shared library");
126  }
127  // Call first to ensure any previous error is cleared before trying to load the symbol
128  dlerror();
129  void *symbol = dlsym(dyn_lib_handle, symbol_name.c_str());
130  // Now call again to see if there was an error (if there was, this will not be null)
131  char *err_message = dlerror();
132  if (symbol == nullptr && (err_message != nullptr || !is_null_valid)) {
133  this->init_exception_msg = "Cannot load shared lib symbol '" + symbol_name + "' for model " + this->model_name;
134  if (err_message != nullptr) {
135  this->init_exception_msg += " (" + std::string(err_message) + ")";
136  }
137  throw ::external::ExternalIntegrationException(this->init_exception_msg);
138  }
139  return symbol;
140  }
