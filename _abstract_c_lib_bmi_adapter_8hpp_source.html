79 inline void dynamic_library_load() {
80 if (bmi_registration_function.empty()) {
81 this->init_exception_msg =
82 "Can't init " + this->model_name + "; empty name given for library's registration function.";
83 throw std::runtime_error(this->init_exception_msg);
84 }
85 if (dyn_lib_handle != nullptr) {
86 this->output.put("WARNING: ignoring attempt to reload dynamic shared library '" + bmi_lib_file +
87 "' for " + this->model_name);
88 return;
89 }
90 if (!utils::FileChecker::file_is_readable(bmi_lib_file)) {
91 //Try alternative extension...
92 size_t idx = bmi_lib_file.rfind(".");
93 if(idx == std::string::npos){
94 idx = bmi_lib_file.length()-1;
95 }
96 std::string alt_bmi_lib_file;
97 if(bmi_lib_file.length() == 0){
98 this->init_exception_msg =
99 "Can't init " + this->model_name + "; library file path is empty";
100 throw std::runtime_error(this->init_exception_msg);
101 }
102 if(bmi_lib_file.substr(idx) == ".so"){
103 alt_bmi_lib_file = bmi_lib_file.substr(0,idx) + ".dylib";
104 } else if(bmi_lib_file.substr(idx) == ".dylib"){
105 alt_bmi_lib_file = bmi_lib_file.substr(0,idx) + ".so";
106 } else {
107 // Try appending instead of replacing...
108 #ifdef __APPLE__
109 alt_bmi_lib_file = bmi_lib_file + ".dylib";
110 #else
111 #ifdef __GNUC__
112 alt_bmi_lib_file = bmi_lib_file + ".so";
113 #endif // __GNUC__
114 #endif // __APPLE__
115 }
116 //TODO: Try looking in e.g. /usr/lib, /usr/local/lib, $LD_LIBRARY_PATH... try pre-pending "lib"...
117 if (utils::FileChecker::file_is_readable(alt_bmi_lib_file)) {
118 bmi_lib_file = alt_bmi_lib_file;
119 } else {
120 this->init_exception_msg =
121 "Can't init " + this->model_name + "; unreadable shared library file '" + bmi_lib_file + "'";
122 throw std::runtime_error(this->init_exception_msg);
123 }
124
125 }

127 // Call first to ensure any previous error is cleared before trying to load the symbol
128 dlerror();
129 // Load up the necessary library dynamically
130 dyn_lib_handle = dlopen(bmi_lib_file.c_str(), RTLD_NOW | RTLD_LOCAL);
131 // Now call again to see if there was an error (if there was, this will not be null)
132 char *err_message = dlerror();
133 if (dyn_lib_handle == nullptr && err_message != nullptr) {
134 this->init_exception_msg = "Cannot load shared lib '" + bmi_lib_file + "' for model " + this->model_name;
135 if (err_message != nullptr) {

164 inline void *dynamic_load_symbol(const std::string &symbol_name, bool is_null_valid) {
165 if (dyn_lib_handle == nullptr) {
166 throw std::runtime_error("Cannot load symbol '" + symbol_name + "' without handle to shared library (bmi_lib_file = '" + bmi_lib_file + "')");
167 }
168 // Call first to ensure any previous error is cleared before trying to load the symbol
169 dlerror();
170 void *symbol = dlsym(dyn_lib_handle, symbol_name.c_str());
171 // Now call again to see if there was an error (if there was, this will not be null)
172 char *err_message = dlerror();
173 if (symbol == nullptr && (err_message != nullptr || !is_null_valid)) {
174 this->init_exception_msg = "Cannot load shared lib symbol '" + symbol_name + "' for model " + this->model_name;
175 if (err_message != nullptr) {
176 this->init_exception_msg += " (" + std::string(err_message) + ")";
177 }
178 throw ::external::ExternalIntegrationException(this->init_exception_msg);
179 }
180 return symbol;
181 }
