80 inline void dynamic_library_load() {
81 if (bmi_registration_function.empty()) {
82 this->init_exception_msg =
83 "Can't init " + this->model_name + "; empty name given for library's registration function.";
84 throw std::runtime_error(this->init_exception_msg);
85 }
86 if (dyn_lib_handle != nullptr) {
87 this->output.put("WARNING: ignoring attempt to reload dynamic shared library '" + bmi_lib_file +
88 "' for " + this->model_name);
89 return;
90 }
91 if (!utils::FileChecker::file_is_readable(bmi_lib_file)) {
92 //Try alternative extension...
93 size_t idx = bmi_lib_file.rfind(".");
94 if(idx == std::string::npos){
95 idx = bmi_lib_file.length()-1;
96 }
97 std::string alt_bmi_lib_file;
98 if(bmi_lib_file.length() == 0){
99 this->init_exception_msg =
100 "Can't init " + this->model_name + "; library file path is empty";
101 throw std::runtime_error(this->init_exception_msg);
102 }
103 if(bmi_lib_file.substr(idx) == ".so"){
104 alt_bmi_lib_file = bmi_lib_file.substr(0,idx) + ".dylib";
105 } else if(bmi_lib_file.substr(idx) == ".dylib"){
106 alt_bmi_lib_file = bmi_lib_file.substr(0,idx) + ".so";
107 } else {
108 // Try appending instead of replacing...
109 #ifdef __APPLE__
110 alt_bmi_lib_file = bmi_lib_file + ".dylib";
111 #else
112 #ifdef __GNUC__
113 alt_bmi_lib_file = bmi_lib_file + ".so";
114 #endif // __GNUC__
115 #endif // __APPLE__
116 }
117 //TODO: Try looking in e.g. /usr/lib, /usr/local/lib, $LD_LIBRARY_PATH... try pre-pending "lib"...
118 if (utils::FileChecker::file_is_readable(alt_bmi_lib_file)) {
119 bmi_lib_file = alt_bmi_lib_file;
120 } else {
121 this->init_exception_msg =
122 "Can't init " + this->model_name + "; unreadable shared library file '" + bmi_lib_file + "'";
123 throw std::runtime_error(this->init_exception_msg);
124 }
125
126 }
127
128 // Call first to ensure any previous error is cleared before trying to load the symbol
129 dlerror();
130 // Load up the necessary library dynamically
131 dyn_lib_handle = dlopen(bmi_lib_file.c_str(), RTLD_NOW | RTLD_LOCAL);
132 // Now call again to see if there was an error (if there was, this will not be null)
133 char *err_message = dlerror();
134 if (dyn_lib_handle == nullptr && err_message != nullptr) {
135 this->init_exception_msg = "Cannot load shared lib '" + bmi_lib_file + "' for model " + this->model_name;
136 if (err_message != nullptr) {
137 this->init_exception_msg += " (" + std::string(err_message) + ")";
138 }
139 throw ::external::ExternalIntegrationException(this->init_exception_msg);
140 }

165 inline void *dynamic_load_symbol(const std::string &symbol_name, bool is_null_valid) {
166 if (dyn_lib_handle == nullptr) {
167 throw std::runtime_error("Cannot load symbol " + symbol_name + " without handle to shared library");
168 }
169 // Call first to ensure any previous error is cleared before trying to load the symbol
170 dlerror();
171 void *symbol = dlsym(dyn_lib_handle, symbol_name.c_str());
172 // Now call again to see if there was an error (if there was, this will not be null)
173 char *err_message = dlerror();
174 if (symbol == nullptr && (err_message != nullptr || !is_null_valid)) {
175 this->init_exception_msg = "Cannot load shared lib symbol '" + symbol_name + "' for model " + this->model_name;
176 if (err_message != nullptr) {
177 this->init_exception_msg += " (" + std::string(err_message) + ")";
178 }
179 throw ::external::ExternalIntegrationException(this->init_exception_msg);
180 }
181 return symbol;
182 }
