14//############################################################*
15// subroutine to calculate evapotranspiration using *
16// Chow, Maidment, and Mays textbook *
17// F.L. Ogden, NOAA National Weather Service, 2020 *
18//############################################################*
19extern double evapotranspiration_energy_balance_method
20(
21 evapotranspiration_options *et_options,
22 evapotranspiration_params *et_params,
23 evapotranspiration_forcing *et_forcing
24)
25{
26 // local varibles
27 double water_latent_heat_of_vaporization_J_per_kg;
28 double liquid_water_density_kg_per_m3;
29 double lambda_et;
30 double radiation_balance_evapotranspiration_rate_m_per_s;
31
32 // from FAO document: cp specific heat at constant pressure, 1.013 10-3 [MJ kg-1 °C-1],
33
34 // IF SOIL WATER TEMPERATURE NOT PROVIDED, USE A SANE VALUE
35 if(100.0 > et_forcing->water_temperature_C) et_forcing->water_temperature_C=22.0; // growing season
36
37 // CALCULATE VARS NEEDED FOR THE ALL METHODS:
38
39 liquid_water_density_kg_per_m3 = calc_liquid_water_density_kg_per_m3(et_forcing->water_temperature_C); // rho_w
40
41 water_latent_heat_of_vaporization_J_per_kg=2.501e+06-2370.0*et_forcing->water_temperature_C; // eqn 2.7.6 Chow etal.
42 // aka 'lambda'
43
44 // We need this in all options except for aerodynamic or Penman-Monteith methods.
45 // Radiation balance is the simplest method. Involves only radiation calculations, no aerodynamic calculations.
46
47 lambda_et=0.0;
48 if( (et_options->use_aerodynamic_method == FALSE ) && (et_options->use_penman_monteith_method==FALSE) )
49 {
50 // This is equation 3.5.9 from Chow, Maidment, and Mays textbook.
51 lambda_et=et_forcing->net_radiation_W_per_sq_m;
52 radiation_balance_evapotranspiration_rate_m_per_s=lambda_et/
53 (liquid_water_density_kg_per_m3*water_latent_heat_of_vaporization_J_per_kg);
54 }
55 return(radiation_balance_evapotranspiration_rate_m_per_s);
56}
