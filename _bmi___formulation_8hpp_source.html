154  void inner_create_formulation(geojson::PropertyMap properties, bool needs_param_validation) {
155  if (needs_param_validation) {
156  validate_parameters(properties);
157  }
158  // Required parameters first
159  set_bmi_init_config(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG).as_string());
160  set_bmi_main_output_var(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MAIN_OUT_VAR).as_string());
161  set_forcing_file_path(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__FORCING_FILE).as_string());
162  set_model_type_name(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MODEL_TYPE).as_string());
163  set_bmi_using_forcing_file(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__USES_FORCINGS).as_boolean());
164 
165  // Then optional ...
166 
167  if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END) != properties.end()) {
168  set_allow_model_exceed_end_time(
169  properties.at(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END).as_boolean());
170  }
171  if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP) != properties.end()) {
172  set_bmi_model_time_step_fixed(
173  properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP).as_boolean());
174  }
175 
176  // Do this next, since after checking whether other input variables are present in the properties, we can
177  // now construct the adapter and init the model
178  set_bmi_model(construct_model(properties));
179 
180  // Output variable subset and order, if present
181  auto out_var_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_VARS);
182  if (out_var_it != properties.end()) {
183  std::vector<geojson::JSONProperty> out_vars_json_list = out_var_it->second.as_list();
184  std::vector<std::string> out_vars(out_vars_json_list.size());
185  for (int i = 0; i < out_vars_json_list.size(); ++i) {
186  out_vars[i] = out_vars_json_list[i].as_string();
187  }
188  set_output_variable_names(out_vars);
189  }
190  // Otherwise, just take what literally is provided by the model
191  else {
192  set_output_variable_names(get_bmi_model()->GetOutputVarNames());
193  }
194 
195  // Output header fields, if present
196  auto out_headers_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_HEADER_FIELDS);
197  if (out_headers_it != properties.end()) {
198  std::vector<geojson::JSONProperty> out_headers_json_list = out_var_it->second.as_list();
199  std::vector<std::string> out_headers(out_headers_json_list.size());
200  for (int i = 0; i < out_headers_json_list.size(); ++i) {
201  out_headers[i] = out_headers_json_list[i].as_string();
202  }
203  set_output_header_fields(out_headers);
204  }
205  else {
206  set_output_header_fields(get_output_variable_names());
207  }
208  }
