223 const std::string get_analogous_cxx_type(const std::string &py_type_name, const size_t item_size) override {
224 /*
225 * Note that an implementation using a "switch" statement would be problematic. It could be done by
226 * rewriting to separate the integer and non-integer type, then having cases based on size. However,
227 * this might cause trouble on certain systems, since (depending on the particular sizes of types) that
228 * could produce duplicate "case" values.
229 */
230 //TODO: include other numpy type strings, https://numpy.org/doc/stable/user/basics.types.html
231 if ( (py_type_name == "int" || py_type_name == "int16") && item_size == sizeof(short)) {
232 return "short";
233 } else if ( (py_type_name == "int" || py_type_name == "int32" )&& item_size == sizeof(int)) {
234 return "int";
235 } else if (py_type_name == "int" && item_size == sizeof(long)) {
236 return "long";
237 } else if ( (py_type_name == "int" || py_type_name == "int64") && item_size == sizeof(long long)) {
238 return "long long";
239 } else if (py_type_name == "longlong" && item_size == sizeof(long long)) {
240 return "long long"; //numpy type
241 } else if (py_type_name == "float" && item_size == sizeof(float)) {
242 return "float";
243 } else if ((py_type_name == "float" || py_type_name == "float64" || py_type_name == "np.float64" ||
244 py_type_name == "numpy.float64") && item_size == sizeof(double)) {
245 return "double";
246 } else if (py_type_name == "float" && item_size == sizeof(long double)) {
247 return "long double";
248 } else {
249 throw std::runtime_error(
250 "(Bmi_Py_Adapter) Failed determining analogous C++ type for Python model '" + py_type_name +
251 "' type with size " + std::to_string(item_size) + " bytes.");
252 }
253 }

421 size_t val_item_size = (size_t)GetVarItemsize(name);
422 std::vector<std::string> in_v = GetInputVarNames();

425 if (val_type == "int" && val_item_size == sizeof(short))

434 if (val_item_size == sizeof(float))

469 template <typename T>
470 py::array_t<T> get_via_numpy_array(const std::string& name, void *dest, const int *indices, int item_count,
471 size_t item_size, bool is_all_indices)
472 {
473 std::string val_type = GetVarType(name);
474 py::array_t<T, py::array::c_style> dest_array
475 = np.attr("zeros")(item_count, "dtype"_a = val_type, "order"_a = "C");
476 if (is_all_indices) {
477 bmi_model->attr("get_value")(name, dest_array);
478 }
479 else {
480 py::array_t<int, py::array::c_style> indices_np_array
481 = np.attr("zeros")(item_count, "dtype"_a = "int32", "order"_a = "C");
482 auto indices_mut_direct = indices_np_array.mutable_unchecked<1>();
483 for (py::size_t i = 0; i < (py::size_t) item_count; ++i)
484 indices_mut_direct(i) = indices[i];
485 /* Leaving this here for now in case mutable_unchecked method doesn't behave as expected in testing
486 py::buffer_info buffer_info = indices_np_array.request();
487 int *indices_np_arr_ptr = (int*)buffer_info.ptr;
488 for (int i = 0; i < item_count; ++i)
489 indices_np_arr_ptr[i] = indices[i];
490 */
491 bmi_model->attr("get_value_at_indices")(name, dest_array, indices_np_array);
492 }
493
494 auto direct_access = dest_array.template unchecked<1>();
495 for (py::size_t i = 0; i < (py::size_t) item_count; ++i)
496 ((T *) dest)[i] = direct_access(i);
497
498 return dest_array;
499 }

514 void SetValue(std::string name, void *src) override {
515 int itemSize = GetVarItemsize(name);
516 std::string py_type = GetVarType(name);
517 std::string cxx_type = get_analogous_cxx_type(py_type, (size_t) itemSize);
518
519 if (cxx_type == "short") {
520 set_value<short>(name, (short *) src);
521 } else if (cxx_type == "int") {
522 set_value<int>(name, (int *) src);
523 } else if (cxx_type == "long") {
524 set_value<long>(name, (long *) src);
525 } else if (cxx_type == "long long") {
526 //FIXME this gets dicey -- if a python numpy array is of type np.int64 (long long),
527 //but a c++ int* is passed to this function as src, it will fail in undefined ways...
528 //the template type overload may be perferred for doing SetValue from framework components
529 //such as forcing providers...
530 set_value<long long>(name, (long long *) src);
531 } else if (cxx_type == "float") {
532 set_value<float>(name, (float *) src);
533 } else if (cxx_type == "double") {
534 set_value<double>(name, (double *) src);
535 } else if (cxx_type == "long double") {
536 set_value<long double>(name, (long double *) src);
537 } else {
538 throw std::runtime_error("Bmi_Py_Adapter cannot set values for variable '" + name +
539 "' that has unrecognized C++ type '" + cxx_type + "'");
540 }

552 void set_value(const std::string &name, std::vector<T> src) {
553 int nbytes = GetVarNbytes(name);
554 int itemSize = GetVarItemsize(name);
555 int length = nbytes / itemSize;
556
557 if (length != src.size()) {
558 throw std::runtime_error(

563 py::array_t<T> model_var_array = bmi_model->attr("get_value_ptr")(name);
564 auto mutable_unchecked_proxy = model_var_array.template mutable_unchecked<1>();
565 for (size_t i = 0; i < length; ++i) {
566 mutable_unchecked_proxy(i) = src[i];

722 inline void separate_package_and_simple_name() {
723 if (!model_initialized) {
724 std::vector<std::string> split_name;
725 std::string delimiter = ".";
726 std::string name_string = bmi_type_py_full_name;
727
728 size_t pos = 0;
729 std::string token;
730 while ((pos = name_string.find(delimiter)) != std::string::npos) {
731 token = name_string.substr(0, pos);
732 split_name.emplace_back(token);
733 name_string.erase(0, pos + delimiter.length());
734 }
735 if (split_name.empty()) {
736 throw std::runtime_error("Cannot interpret BMI Python model type '" + bmi_type_py_full_name
737 + "'; expected format is <python_module>.<python_class>");
738 }
739 // What's left should be the class name
740 bmi_type_py_class_name = std::make_shared<std::string>(name_string);
741 //split_name.pop_back();
742 // And then the split name should contain the module
743 // TODO: going to need to look at this again in the future; right now, assuming the format
744 // <python_module>.<python_class> works fine as long as a model class is always in a top-level
745 // module, but the current logic is going to interpret any complex parent module name as a single
746 // top-level namespace package; e.g., ngen.namespacepackage.model works if ngen.namespacepackage is
747 // a namespace package, but there would be problems with something like ngenpkg.innermodule1.model.
748 bmi_type_py_module_name = std::make_shared<std::string>(boost::algorithm::join(split_name, delimiter));
749 }
