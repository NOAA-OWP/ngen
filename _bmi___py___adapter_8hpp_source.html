234  void get_value_at_indices(const string& name, void *dest, int *inds, int count, bool is_all_indices) {
235  string val_type = GetVarType(name);
236  size_t val_item_size = (size_t)GetVarItemsize(name);
237  vector<string> in_v = GetInputVarNames();
238 
239  // The available types and how they are handled here should match what is in SetValueAtIndices
240  if (val_type == "int" && val_item_size == sizeof(short))
241  get_via_numpy_array<short>(name, dest, inds, count, val_item_size, is_all_indices);
242  else if (val_type == "int" && val_item_size == sizeof(int))
243  get_via_numpy_array<int>(name, dest, inds, count, val_item_size, is_all_indices);
244  else if (val_type == "int" && val_item_size == sizeof(long))
245  get_via_numpy_array<long>(name, dest, inds, count, val_item_size, is_all_indices);
246  else if (val_type == "int" && val_item_size == sizeof(long long))
247  get_via_numpy_array<long long>(name, dest, inds, count, val_item_size, is_all_indices);
248  else if (val_type == "float" && val_item_size == sizeof(float))
249  get_via_numpy_array<float>(name, dest, inds, count, val_item_size, is_all_indices);
250  else if (val_type == "float" && val_item_size == sizeof(double))
251  get_via_numpy_array<double>(name, dest, inds, count, val_item_size, is_all_indices);
252  else if (val_type == "float" && val_item_size == sizeof(long double))
