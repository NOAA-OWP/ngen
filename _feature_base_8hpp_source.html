490 int get_contributor_count() {
491 int count = 0;
492
493 std::vector<FeatureBase*> to_search;
494 std::vector<FeatureBase*> total;
495
496 FeatureBase *current = this;
497 total.push_back(current);
498
499 for (FeatureBase *origination_feature : current->origination) {
500 to_search.push_back(origination_feature);
501 }
502
503 while (to_search.size() > 0) {
504 current = to_search.back();
505 to_search.pop_back();
506 total.push_back(current);
507 for (FeatureBase *origination_feature : current->origination) {
508 if (not contains(total, origination_feature) and not contains(to_search, origination_feature)) {
509 to_search.push_back(origination_feature);
510 }
511 }
512 }
513
514 return total.size() - 1;
515 }

557 inline bool operator==(const FeatureBase& rhs) {
558 if (this->get_id() != rhs.get_id()) {
559 return false;
560 }
561
562 if (not (this->get_type() == rhs.get_type())) {
563 return false;
564 }
565
566 if (this->bounding_box != rhs.bounding_box) {
567 return false;
568 }
569
570 if (this->get_type() == FeatureType::GeometryCollection) {
571 if (this->geometry_collection.size() != rhs.geometry_collection.size()) {
572 return false;
573 }
574
575 for (int geometry_index = 0; geometry_index < this->geometry_collection.size(); geometry_index++) {
576 if (not bg::equals(this->geometry_collection[geometry_index], rhs.geometry_collection[geometry_index])) {
577 return false;
578 }
579 }
580 }
581 else if (not bg::equals(this->geom, rhs.geom)) {
582 return false;
583 }
584
585 if (this->keys().size() != rhs.keys().size() or this->property_keys().size() != rhs.property_keys().size()) {
586 return false;
587 }
588
589 std::vector<std::string> these_keys = this->keys();
590 std::vector<std::string> these_property_keys = this->property_keys();
591
592 for (std::string key : rhs.keys()) {
593 if (not contains(these_keys, key) or this->get(key) != rhs.get(key)) {
594 return false;
595 }
596 }
597
598 for (std::string key : rhs.property_keys()) {
599 if (not contains(these_keys, key) or this->get_property(key) != rhs.get_property(key)) {
600 return false;
601 }
602 }
603
604 if (
605 not (
606 this->origination.size() == rhs.origination.size()
607 or this->destination.size() == rhs.destination.size()
608 or this->neighbors.size() == rhs.neighbors.size()
609 )
610 ) {
611 return false;
612 }
613
614 for (auto feature : rhs.origination) {
615 if (not contains(this->origination, feature)) {
616 return false;
617 }
618 }
619
620 for (auto feature : rhs.destination) {
621 if (not contains(this->destination, feature)) {
622 return false;
623 }
624 }
625
626 for (auto feature : rhs.neighbors) {
627 if (not contains(this->neighbors, feature)) {
628 return false;
629 }
630 }
631
632 return true;

640 virtual void break_links() {
641 // Go through all of the originators and remove the reference to this feature
642 for (auto originator : this->origination) {
643 originator->remove_destination(this);
644 }
645
646 // Now the list of originators can be cleared because they no longer link to this feature
647 this->origination.clear();
648 for (auto target : this->destination) {
649 target->remove_origination(this);
650 }
651
652 this->destination.clear();
653
654 for (auto neighbor : this->neighbors) {
655 neighbor->remove_neighbor(this);
656 }
657
658 this->neighbors.clear();
659 }
