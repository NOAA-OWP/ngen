495 int get_contributor_count() {
496 int count = 0;
497
498 std::vector<FeatureBase*> to_search;
499 std::vector<FeatureBase*> total;
500
501 FeatureBase *current = this;
502 total.push_back(current);
503
504 for (FeatureBase *origination_feature : current->origination) {
505 to_search.push_back(origination_feature);
506 }
507
508 while (to_search.size() > 0) {
509 current = to_search.back();
510 to_search.pop_back();
511 total.push_back(current);
512 for (FeatureBase *origination_feature : current->origination) {
513 if (not contains(total, origination_feature) and not contains(to_search, origination_feature)) {
514 to_search.push_back(origination_feature);
515 }
516 }
517 }
518
519 return total.size() - 1;
520 }

562 inline bool operator==(const FeatureBase& rhs) {
563 if (this->get_id() != rhs.get_id()) {
564 return false;
565 }
566
567 if (not (this->get_type() == rhs.get_type())) {
568 return false;
569 }
570
571 if (this->bounding_box != rhs.bounding_box) {
572 return false;
573 }
574
575 if (this->get_type() == FeatureType::GeometryCollection) {
576 if (this->geometry_collection.size() != rhs.geometry_collection.size()) {
577 return false;
578 }
579
580 for (int geometry_index = 0; geometry_index < this->geometry_collection.size(); geometry_index++) {
581 if (not bg::equals(this->geometry_collection[geometry_index], rhs.geometry_collection[geometry_index])) {
582 return false;
583 }
584 }
585 }
586 else if (not bg::equals(this->geom, rhs.geom)) {
587 return false;
588 }
589
590 if (this->keys().size() != rhs.keys().size() or this->property_keys().size() != rhs.property_keys().size()) {
591 return false;
592 }
593
594 std::vector<std::string> these_keys = this->keys();
595 std::vector<std::string> these_property_keys = this->property_keys();
596
597 for (std::string key : rhs.keys()) {
598 if (not contains(these_keys, key) or this->get(key) != rhs.get(key)) {
599 return false;
600 }
601 }
602
603 for (std::string key : rhs.property_keys()) {
604 if (not contains(these_keys, key) or this->get_property(key) != rhs.get_property(key)) {
605 return false;
606 }
607 }
608
609 if (
610 not (
611 this->origination.size() == rhs.origination.size()
612 or this->destination.size() == rhs.destination.size()
613 or this->neighbors.size() == rhs.neighbors.size()
614 )
615 ) {
616 return false;
617 }
618
619 for (auto feature : rhs.origination) {
620 if (not contains(this->origination, feature)) {
621 return false;
622 }
623 }
624
625 for (auto feature : rhs.destination) {
626 if (not contains(this->destination, feature)) {
627 return false;
628 }
629 }
630
631 for (auto feature : rhs.neighbors) {
632 if (not contains(this->neighbors, feature)) {
633 return false;
634 }
635 }
636
637 return true;

645 virtual void break_links() {
646 // Go through all of the originators and remove the reference to this feature
647 for (auto originator : this->origination) {
648 originator->remove_destination(this);
649 }
650
651 // Now the list of originators can be cleared because they no longer link to this feature
652 this->origination.clear();
653 for (auto target : this->destination) {
654 target->remove_origination(this);
655 }
656
657 this->destination.clear();
658
659 for (auto neighbor : this->neighbors) {
660 neighbor->remove_neighbor(this);
661 }
662
663 this->neighbors.clear();
664 }
