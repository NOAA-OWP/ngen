141 void addToPath(const std::string &directoryPath) {
142 std::tuple<py::list, std::vector<std::string>> sysPathTuple = getSystemPath();
143 py::list sys_path = std::get<0>(sysPathTuple);
144 std::vector<std::string> sys_path_vector = std::get<1>(sysPathTuple);
145
146 py::object requestedDirPath = Path(directoryPath);
147 if (py::bool_(requestedDirPath.attr("is_dir")())) {
148 if (std::find(sys_path_vector.begin(), sys_path_vector.end(), directoryPath) == sys_path_vector.end()) {
149 sys_path.attr("insert")(sys_path_vector.size(), py::str(directoryPath));
150 }
151 }
152 else {
153 std::string dirPath = py::str(requestedDirPath);
154 throw std::runtime_error("Cannot add non-existing directory '" + dirPath + "' to Python PATH");

217 py::object getModule(const std::vector<std::string> &moduleLevelNames) {
218 // Start with top-level module name, then descend through attributes as needed
219 py::object module = getModule(moduleLevelNames[0]);
220 for (size_t i = 1; i < moduleLevelNames.size(); ++i) {
221 module = module.attr(moduleLevelNames[i].c_str());

233 py::list getVenvPackagesDirOptions() {
234 // Look for a local virtual environment directory also, if there is one
235 const char* env_var_venv = std::getenv("VIRTUAL_ENV");
236 py::object venv_dir = env_var_venv != nullptr ? Path(env_var_venv): py::none();
