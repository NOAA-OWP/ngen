163  py::list getVenvPackagesDirOptions() {
164  // Add the package dir from a local virtual environment directory also, if there is one
165  py::object venv_dir = Path("./.venv");
166  // Try switching to a secondary if primary isn't a directory
167  if (!(py::bool_(venv_dir.attr("is_dir")()))) {
