120 void addToPath(const std::string &directoryPath) {
121 std::tuple<py::list, std::vector<std::string>> sysPathTuple = getSystemPath();
122 py::list sys_path = std::get<0>(sysPathTuple);
123 std::vector<std::string> sys_path_vector = std::get<1>(sysPathTuple);
124
125 py::object requestedDirPath = Path(directoryPath);
126 if (py::bool_(requestedDirPath.attr("is_dir")())) {
127 if (std::find(sys_path_vector.begin(), sys_path_vector.end(), directoryPath) == sys_path_vector.end()) {
128 sys_path.attr("insert")(sys_path_vector.size(), py::str(directoryPath));
129 }
130 }
131 else {
132 std::string dirPath = py::str(requestedDirPath);
133 throw std::runtime_error("Cannot add non-existing directory '" + dirPath + "' to Python PATH");

217 py::object searchForVenvDir() {
218 py::object current_dir = Path.attr("cwd")();
219 std::vector<py::object> parent_options = {
220 current_dir,
221 current_dir.attr("parent"),
222 current_dir.attr("parent").attr("parent")
223 };
224 std::vector<std::string> dir_name_options = {".venv", "venv"};
225 for (py::object &parent_option : parent_options) {
226 for (const std::string &dir_name_option : dir_name_options) {
227 py::object venv_dir_candidate = parent_option.attr("joinpath")(dir_name_option);
228 if (py::bool_(venv_dir_candidate.attr("is_dir")())) {
229 return venv_dir_candidate;
230 }
231 }
232 }
233 return py::none();
234 }

241 py::list getVenvPackagesDirOptions() {
242 // Look for a local virtual environment directory also, if there is one
243 const char* env_var_venv = std::getenv("VIRTUAL_ENV");
244 py::object venv_dir = env_var_venv != nullptr ? Path(env_var_venv): searchForVenvDir();
245
246 if (!venv_dir.is_none() && py::bool_(venv_dir.attr("is_dir")())) {
247 // Resolve the full path
248 venv_dir = venv_dir.attr("resolve")();
249 // Get options for the packages dir
250 py::list site_packages_options = py::list(venv_dir.attr("glob")("**/site-packages/"));
251 return site_packages_options;
252 }
253
254 return py::list();
255 }

264 inline std::tuple<py::list, std::vector<std::string>> getSystemPath() {
265 py::list sys_path = importedTopLevelModules.find("sys")->second.attr("path");
266 std::vector<std::string> sys_path_vector(sys_path.size());
