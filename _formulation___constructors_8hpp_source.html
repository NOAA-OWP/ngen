54 std::string identifier,
55 forcing_params &forcing_config,
56 utils::StreamHandler output_stream
57 ) {
58 constructor formulation_constructor = formulations.at(formulation_type);
59 std::shared_ptr<data_access::GenericDataProvider> fp;
60 if (forcing_config.provider == "CsvPerFeature" || forcing_config.provider == ""){
61 fp = std::make_shared<CsvPerFeatureForcingProvider>(forcing_config);
62 }
63#ifdef NETCDF_ACTIVE
64 else if (forcing_config.provider == "NetCDF"){
65 fp = data_access::NetCDFPerFeatureDataProvider::get_shared_provider(forcing_config.path, forcing_config.simulation_start_t, forcing_config.simulation_end_t, output_stream);
66 }
67#endif
68 else if (forcing_config.provider == "NullForcingProvider"){
69 fp = std::make_shared<NullForcingProvider>();
70 }
71 else { // Some unknown string in the provider field?
72 throw std::runtime_error(
73 "Invalid formulation forcing provider configuration! identifier: \"" + identifier +
74 "\", formulation_type: \"" + formulation_type +
75 "\", provider: \"" + forcing_config.provider + "\"");
76 }
77 return formulation_constructor(identifier, fp, output_stream);
78 };
