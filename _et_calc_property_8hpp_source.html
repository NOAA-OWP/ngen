54 //############################################################*
55 // subroutine to calculate net radiation from all components *
56 // of the radiation budget, using values provided by WRF *
57 // Reference, Bras, R.L., HYDROLOGY an Introduction to *
58 // Hydrologic Science, Addison Wesley, 1990. *
59 // F.L. Ogden, NOAA National Weather Service, 2020 *
60 //############################################################*
61 extern double calculate_net_radiation_W_per_sq_m
62 (
63  evapotranspiration_options *et_options, // et_options.yes_aorc == TRUE if we are using aorc forcing values.
64  surface_radiation_params *surf_rad_params,
65  surface_radiation_forcing *surf_rad_forcing
66 )
67 
68 {
69  // local variables
70  double net_radiation_W_per_sq_m;
71  double outgoing_longwave_radiation_W_per_sq_m;
72  double stefan_boltzmann_constant = SB; //W/m^2/K^4
73  double atmosphere_longwave_emissivity; // dimensionless, based on water vapor content of air
74  double saturation_water_vapor_partial_pressure_Pa;
75  double actual_water_vapor_partial_pressure_Pa;
76  double actual_water_vapor_mixing_ratio;
77  double cloud_base_temperature_C;
78  double surface_longwave_albedo;
79  double N,Klw;
80 
81  // CALCULATE OUTGOING LONGWAVE RADIATION FLUX FROM SURFACE
82  outgoing_longwave_radiation_W_per_sq_m=surf_rad_params->surface_longwave_emissivity*stefan_boltzmann_constant*
83  pow(surf_rad_forcing->surface_skin_temperature_C+TK,4.0);
84  // must convert C to K
85 
86  if(0.999 < surf_rad_params->surface_longwave_emissivity)
87  {
88  surface_longwave_albedo=0.0; // soil, rock, concrete, asphalt, vegetation, snow
89  }
90  else
91  {
92  surface_longwave_albedo=0.03; // water - actually not this simple, but close enough for now
93  }
94 
95  if(et_options->yes_aorc==FALSE) // we must calculate longwave incoming from the atmosphere
96  saturation_water_vapor_partial_pressure_Pa=calc_air_saturation_vapor_pressure_Pa(surf_rad_forcing->air_temperature_C);
97 
98  actual_water_vapor_partial_pressure_Pa=surf_rad_forcing->relative_humidity_percent/100.0*
99  saturation_water_vapor_partial_pressure_Pa;
100 
101  if(et_options->yes_aorc==FALSE)
102  {
103  // CALCULATE DOWNWELLING LONGWAVE RADIATION FLUX FROM ATMOSPHERE, W/m2.
104  if(0.90 < surf_rad_forcing->cloud_cover_fraction) // very nearly overcast or overcast
105  {
106  // calculate longwave downwelling using overcast equation, with emissivity of cloud base =1.0
107  cloud_base_temperature_C=surf_rad_forcing->air_temperature_C+
108  surf_rad_forcing->ambient_temperature_lapse_rate_deg_C_per_km*
109  surf_rad_forcing->cloud_base_height_m/1000.0;
110  surf_rad_forcing->incoming_longwave_radiation_W_per_sq_m=stefan_boltzmann_constant*
111  pow(cloud_base_temperature_C+TK,4.0);
112  }
113  else // not overcast, use TVA (1972) formulation, taken from Bras R.L., textbook, pg. 44.
114  {
115  // use cloudy skies formulation
116  // clear sky emissivity
117  atmosphere_longwave_emissivity = 0.740 + 0.0049*actual_water_vapor_partial_pressure_Pa/100.0; //conv. Pa to mb
118  N=surf_rad_forcing->cloud_cover_fraction;
119  Klw=(1.0+0.17*N*N); // effect of cloud cover from TVA (1972)
120  surf_rad_forcing->incoming_longwave_radiation_W_per_sq_m=atmosphere_longwave_emissivity*Klw*
121  stefan_boltzmann_constant*
122  pow(surf_rad_forcing->air_temperature_C+TK,4.0);
123  }
124  }
125 
126  net_radiation_W_per_sq_m=(1.0-surf_rad_params->surface_shortwave_albedo)*
127  surf_rad_forcing->incoming_shortwave_radiation_W_per_sq_m +
128  (1.0-surface_longwave_albedo)*
129  surf_rad_forcing->incoming_longwave_radiation_W_per_sq_m -
130  outgoing_longwave_radiation_W_per_sq_m; // this is plus, negative grnd ht flx is downward
131 
132  return(net_radiation_W_per_sq_m);
133 }

135 //############################################################*
136 // subroutine to calculate aerodynamic resistance term needed *
137 // in both the Penman-Monteith FAO reference ET procedure. *
138 // and the aerodynamic, combination, and Priestley-Taylor *
139 // ET calculation methods. *
140 // Reference: http://www.fao.org/3/X0490E/x0490e06.htm *
141 // F.L. Ogden, NOAA National Weather Service, 2020 *
142 //############################################################*
143 extern double calculate_aerodynamic_resistance
144 (
145  double wind_speed_measurement_height_m, // default =2.0 [m]
146  double humidity_measurement_height_m, // default =2.0 [m],
147  double zero_plane_displacement_height_m, // depends on surface roughness [m],
148  double momentum_transfer_roughness_length_m, // [m],
149  double heat_transfer_roughness_length_m, // [m],
150  double wind_speed_m_per_s // [m s-1].
151 )
152 {
153  // define local variables to ease computations:
154 
155  double ra,zm,zh,d,zom,zoh,k,uz;
156  double von_karman_constant_squared=KV2; // this is dimensionless universal constant [-], K=0.41, squared.
157 
158  // input sanity checks.
159  if(1.0e-06 >=wind_speed_measurement_height_m ) wind_speed_measurement_height_m=2.0; // standard measurement height
160  if(1.0e-06 >=humidity_measurement_height_m ) humidity_measurement_height_m=2.0; // standard measurement height
161  if(1.0e-06 >= momentum_transfer_roughness_length_m)
162  fprintf(stderr,"momentum_transfer_roughness_length_m is tiny in calculate_aerodynamic_resistance(). Should not be tiny.\n");
163  if(1.0e-06 >= heat_transfer_roughness_length_m ) //warn. Should not be tiny.
164  fprintf(stderr,"heat_transfer_roughness_length_m is tiny in calculate_aerodynamic_resistance(). Should not be tiny.\n");
165 
166  // convert to smaller local variable names to keep equation readable
167  zm=wind_speed_measurement_height_m;
168  zh=humidity_measurement_height_m;
169  d= zero_plane_displacement_height_m;
170  zom=momentum_transfer_roughness_length_m;
171  zoh=heat_transfer_roughness_length_m;
172  uz=wind_speed_m_per_s;
173 
174  // here log is the natural logarithm.
175 
176  ra=log((zm-d)/zom)*log((zh-d)/zoh)/(von_karman_constant_squared*uz); // this is the equation for the aero. resist.
177  // from the FAO reference ET document.
178 
179  return(ra);
180 }

183 // function to calculate saturation vapor pressure of air *
184 // based on the exponential relationship defined in Chow, *
185 // Maidment, and Mays,textbook. Input is air temp C *
186 // F.L. Ogden, NOAA National Weather Service, 2020 *

190  double air_sat_vap_press_Pa= 611.0*exp(17.27*air_temperature_C/(237.3+air_temperature_C)); // it is 237.3

196 //#################################################################*
197 // function to calculate slope saturation vapor pressure curve *
198 // based on the exponential relationship defined in Chow, *
199 // Maidment, and Mays,textbook, Eqn. 3.2.10. Input is air temp C *
200 // calls function calc_air_saturation_vapor_pressure_Pa(). *
201 // F.L. Ogden, NOAA National Weather Service, 2020 *

211 //############################################################*
212 // function to calculate density of liquid water by empirical *
213 // equation, as a function of water temperature in C *
214 // fit of water density vs. temperature data by FLO. Data *
215 // from water properties table in Chow, Maidment, and Mays *
216 // textbook. Fit using tblcurve program. r^2=0.9975 *
217 // this fit produces rho_w=1000.151 kg/m3 at T=0C, so I limit *
218 // it to 1000.0. Doesn't accurately predict max. density at *
219 // T=4C, so this equation is an approximation. TODO maybe a *
220 // better model exists. But, this value is not required at *
221 // super high precision to convert latent heat flux into a *
222 // depth of water. *
223 // F.L. Ogden, NOAA National Weather Service, 2020 *
224 //############################################################*
225 double calc_liquid_water_density_kg_per_m3(double water_temperature_C)
226 {
227  double a=0.0009998492; // this precision is necessary
228  double b=4.9716595e-09; // ditto.
229 
230  double water_density_kg_per_m3=1.0/(a+b*water_temperature_C*water_temperature_C);
231 
232  if(988> water_density_kg_per_m3) fprintf(stderr,"strange water density value!\n");
233  if(1000<water_density_kg_per_m3) water_density_kg_per_m3=1000.0; // this empirical function yield 1000.151 at 0C.
234 
235  return(water_density_kg_per_m3);
236 }

239 //############################################################/
240 // subroutine to calculate the short-wave solar radiation /
241 // reaching the land surface. Mostly from Hydrology textbook /
242 // by R.L. Bras., with some updates on calculation of local /
243 // hour angle, optical air mass, and atmospheric extinction. /
244 // Inputs: air temp., cloud cover fraction, air turbidity, /
245 // day of year, zulu time /
246 // outputs solar radiation, solar elev. angle, solar azimuth. /
247 // F.L. Ogden, 2009, NOAA National Weather Service, 2020 /
248 //############################################################/
249 
250 void calculate_solar_radiation(solar_radiation_options *options, solar_radiation_parameters *params,
251  solar_radiation_forcing *forcing, solar_radiation_results *results)
252 {
253  double delta,r,equation_of_time_minutes,M,phi;
254  double sinalpha,tau,alpha,cosalpha,azimuth;
255  double Io,Ic,kshort,Ips;
256  double b,fh1;
257 
258  // constants
259  double solar_constant_W_per_sq_m;
260 
261 
262  double solar_declination_angle_degrees;
263  double solar_declination_angle_radians;
264  double earth_sun_distance_ratio;
265  double local_hour_angle_degrees;
266  double local_hour_angle_radians;
267  double antipodal_hour_angle_degrees;
268  double antipodal_obs_longitude_degrees;
269  double obs_x,obs_y,sun_x,sun_y;
270  double zulu_time_h;
271  double optical_air_mass;
272 
273  solar_constant_W_per_sq_m = 1361.6; // Dudock de Wit et al. 2017 GRL, approx. avg. value
274 
275  solar_declination_angle_degrees=23.45*M_PI/180.0*cos(2.0*M_PI/365.0*(172.0-forcing->day_of_year));
276  solar_declination_angle_radians=solar_declination_angle_degrees*M_PI/180.0;
277 
278  earth_sun_distance_ratio=1.0+0.017*cos(2.0*M_PI/365*(186.0-forcing->day_of_year));
279 
280  // calculate the local hour angle using a unit circle centered on the observer, with Obs. at x=1, y=0.
281  // Note: G=Greenwich, A=180E=180W==antipode, O=obs., S=sun.
282  //--------------------------------------------------------------------------------------------------------------
283  //
284  // y ^
285  // | Antipode
286  // | /
287  // --- / all angles measured as in trigonometry
288  // _-- --_ ^
289  // / \ \ If the sun were here, this would be a neg. LHA
290  // | rot | |
291  // | EARTH ^ | |
292  // | + | |O ----------------------> x (all angles measured ccw from here)
293  // | N.Pole | | Observer \ t
294  // | | \ t
295  // \ / |
296  // --_ _-- |
297  // / -_- \ This is a positive local hour angle
298  // / \ (LHA), after
299  // / \ local noon. This text prevents
300  // v v / this from being
301  // G S / a cont. comment
302  // Greenwich vector pointing
303  // Prime Meridian <TO SUN>
304  //
305 
306  // this is the "equation of time" that accounts for the analemma effect
307  M=2.0*M_PI*forcing->day_of_year/365.242; // mean anomaly of sun from wikipedia, works for leap years
308  equation_of_time_minutes=-7.655*sin(M)+9.873*sin(2.0*M+3.588); // an approximation of the equation of time, minutes
309  zulu_time_h=forcing->zulu_time_h-equation_of_time_minutes/1440.0; // adjust the position of the sun for analemma effect
310 
311  // here I use the antipode as the time origin, because that is where the sun is overhead at 00:00Z
312  antipodal_hour_angle_degrees = zulu_time_h*15.0; // see note on above figure
313 
314  // here I convert longitude of the observer to the same coordinate system to eliminate the problem of +-180 deg. long.
315  antipodal_obs_longitude_degrees = 180.0-params->longitude_degrees;
316 
317  // here I convert these angles to points on a unit circle using the above coordinate system to go to a purely
318  // geometric representation. This helps deal with problems related to +-180 deg.
319  obs_x=1.0;
320  obs_y=0.0;
321 
322  sun_x=cos(M_PI/180.0*(360.0-(antipodal_hour_angle_degrees-antipodal_obs_longitude_degrees)));
323  sun_y=sin(M_PI/180.0*(360.0-(antipodal_hour_angle_degrees-antipodal_obs_longitude_degrees)));
324 
325  local_hour_angle_degrees= acos(obs_x*sun_x+obs_y*sun_y)*180.0/M_PI; // acos(dot-product).
326 
327  if(sun_y>0.0) local_hour_angle_degrees *= -1.0; // Before local noon, the hour angle is defined as negative.
328 
329  local_hour_angle_radians=local_hour_angle_degrees*M_PI/180.0;
330 
331  // USE SIMPLER VARIABLE NAMES FOR THESE DENSE CALCULATIONS
332  tau=local_hour_angle_radians; // local hour angle, radians
333  delta=solar_declination_angle_radians; // solar declination angle, radians
334  phi=params->latitude_degrees*M_PI/180.0; // latitude, radians
335 
336  // calculate solar elevation angle, sin(alpha)
337 
338  sinalpha=sin(delta)*sin(phi)+cos(delta)*cos(phi)*cos(tau);
339  alpha=asin(sinalpha); // radians
340  cosalpha=cos(alpha);
341 
342  // azimuth pointing to the sun (radians)
343  azimuth=acos(sin(delta)/(cosalpha*cos(phi))-tan(alpha)*tan(phi));
344  if(tau>0.0) // after local noon
345  {
346  azimuth=2.0*M_PI-azimuth;
347  }
348 
349  results->solar_elevation_angle_degrees=alpha*180.0/M_PI; // convert to degrees
350 
351  results->solar_azimuth_angle_degrees=azimuth*180.0/M_PI; // convert to degrees
352 
353  results->solar_local_hour_angle_degrees=tau*180.0/M_PI; // convert to degrees
354 
355  if(alpha>0.0) // the sun is over the horizon
356  {
357  // ==================== SHORTWAVE RADIATION CALCULATIONS =======================
358  r=earth_sun_distance_ratio; // the effect of non-circularity (eccentricity) of Earth's orbit
359  Io=solar_constant_W_per_sq_m/(r*r); // J/(m^2 s) at top of atmosphere, adjusted for orbital eccentricity
360 
361  optical_air_mass=(1.002432*pow(sinalpha,2.0)+0.148386*sinalpha+0.0096467)/ // after Young 1994
362  (pow(sinalpha,3.0)+0.149864*pow(sinalpha,2.0)+0.0102963*sinalpha+0.000303978);
363 
364  // the following comes from Ineichen and Perez, 2002
365  fh1=exp(-1.0*params->site_elevation_m/8000.0); // elev. in meters, effect of atmos. thickness on air mass
366  b=0.664+0.163/fh1;
367 
368  // note, atm_turbidity is equal to Tlk in Ineichen and Perez, 2002.
369  Ic=b*Io*exp(-0.09*optical_air_mass*(forcing->atmospheric_turbidity_factor-1.0)); // clear sky radiation
370 
371  // adjust for cloudiness effects using procedure from Bras' Hydrology text
372  if(options->cloud_base_height_known==TRUE)
373  {
374  // percent of cloudless insolation, z= cloud base elev km.
375  kshort=0.18+0.0853*forcing->cloud_base_height_m/1000.0; // convert cloud base height to km for this calc.
376  Ips=Ic*(1.0-(1.0-kshort)*forcing->cloud_cover_fraction); // insolation considering clouds, Eagleson, 1970.
377  }
378  else
379  {
380  // cloud base elevation not known.
381  kshort=0.65*forcing->cloud_cover_fraction*forcing->cloud_cover_fraction; // (TVA, 1972)
382  Ips=Ic*(1.0-kshort);
383  }
384 
385  // all these results are calculated near the land surface, but above the canopy or snow pack.
386  results->solar_radiation_flux_W_per_sq_m= Ic; // no clouds. This is on a plane perpendicular to earth-sun line.
387  results->solar_radiation_horizontal_flux_W_per_sq_m=Ic*sinalpha; // this is on a horizontal plane
388  results->solar_radiation_cloudy_flux_W_per_sq_m=Ips; // Considers clouds, on a plane perpendicular to earth-sun line
389  results->solar_radiation_horizontal_cloudy_flux_W_per_sq_m=Ips*sinalpha; // on a horizontal plane tangent to earth
390 
391  // I comment this out because it is more appripriate in a vegetation effect routine
392  // Ipsg=Kt*Ips;
393 
394  // I comment this out because it is part of net radiation calculations done elsewhere
395  // Ieff=Ipsg*(1.0-Albedo); effective incoming shortwave radiation
396  }
397 
398  return;
399 }

401 // Function to calculate hydrological variables needed for evapotranspiration calculation
402 extern void calculate_intermediate_variables
403 (
404  evapotranspiration_options *et_options,
405  evapotranspiration_params *et_params,
406  evapotranspiration_forcing *et_forcing,
407  intermediate_vars *inter_vars
408 )
409 {
410  // local variables
411  double aerodynamic_resistance_sec_per_m; // value [s per m], computed in: calculate_aerodynamic_resistance()
412  double instantaneous_et_rate_m_per_s;
413  double aerodynamic_resistance_s_per_m;
414  double psychrometric_constant_Pa_per_C;
415  double slope_sat_vap_press_curve_Pa_s;
416  double air_saturation_vapor_pressure_Pa;
417  double air_actual_vapor_pressure_Pa;
418  double moist_air_density_kg_per_m3;
419  double water_latent_heat_of_vaporization_J_per_kg;
420  double moist_air_gas_constant_J_per_kg_K;
421  double moist_air_specific_humidity_kg_per_m3;
422  double vapor_pressure_deficit_Pa;
423  double liquid_water_density_kg_per_m3;
424  double delta;
425  double gamma;
426 
427  // IF SOIL WATER TEMPERATURE NOT PROVIDED, USE A SANE VALUE
428  if(100.0 > et_forcing->water_temperature_C) et_forcing->water_temperature_C=22.0; // growing season
429 
430  // CALCULATE VARS NEEDED FOR THE ALL METHODS:
431 
432  liquid_water_density_kg_per_m3 = calc_liquid_water_density_kg_per_m3(et_forcing->water_temperature_C); // rho_w
433 
434  water_latent_heat_of_vaporization_J_per_kg=2.501e+06-2370.0*et_forcing->water_temperature_C; // eqn 2.7.6 Chow etal.
435  // aka 'lambda'
436  // all methods other than radiation balance method involve at least some of the aerodynamic method calculations
437 
438  // IF HEAT/MOMENTUM ROUGHNESS LENGTHS NOT GIVEN, USE DEFAULTS SO THAT THEIR RATIO IS EQUAL TO 1.
439  if((1.0e-06> et_params->heat_transfer_roughness_length_m) ||
440  (1.0e-06> et_params->momentum_transfer_roughness_length_m)) // zero should be passed down if these are unknown
441  {
442  et_params->heat_transfer_roughness_length_m =1.0; // decent default values, and the ratio of these is 1.0
443  et_params->momentum_transfer_roughness_length_m =1.0;
444  }
445 
446  // e_sat is needed for all aerodynamic and Penman-Monteith methods
447 
448  air_saturation_vapor_pressure_Pa=calc_air_saturation_vapor_pressure_Pa(et_forcing->air_temperature_C);
449 
450  if( (0.0 < et_forcing->relative_humidity_percent) && (100.0 >= et_forcing->relative_humidity_percent) )
451  {
452  // meaningful relative humidity value provided
453  air_actual_vapor_pressure_Pa=et_forcing->relative_humidity_percent/100.0 * air_saturation_vapor_pressure_Pa;
454 
455  // calculate specific humidity, q_v
456  et_forcing->specific_humidity_2m_kg_per_kg=0.622*air_actual_vapor_pressure_Pa/et_forcing->air_pressure_Pa;
457  }
458  else
459  {
460  // if here, we must be using AORC forcing that provides specific humidity instead of relative humidity
461  air_actual_vapor_pressure_Pa=et_forcing->specific_humidity_2m_kg_per_kg*et_forcing->air_pressure_Pa/0.622;
462  if(air_actual_vapor_pressure_Pa > air_saturation_vapor_pressure_Pa)
463  {
464  // this is bad. Actual vapor pressure of air should not be higher than saturated value.
465  // warn and reset to something meaningful
466  fprintf(stderr,"Invalid value of specific humidity with no supplied rel. humidity in ET calc. function:\n");
467  fprintf(stderr,"Relative Humidity: %lf percent\n",et_forcing->relative_humidity_percent);
468  fprintf(stderr,"Specific Humidity: %lf kg/kg\n",et_forcing->specific_humidity_2m_kg_per_kg);
469  air_actual_vapor_pressure_Pa=0.65*air_saturation_vapor_pressure_Pa;
470  }
471  }
472 
473  // VPD
474  vapor_pressure_deficit_Pa = air_saturation_vapor_pressure_Pa - air_actual_vapor_pressure_Pa;
475 
476  moist_air_gas_constant_J_per_kg_K=287.0*(1.0+0.608*et_forcing->specific_humidity_2m_kg_per_kg); //R_a
