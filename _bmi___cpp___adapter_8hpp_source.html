336 inline void execModuleCreation() {
337 if (get_dyn_lib_handle() == nullptr) {
338 if (model_create_fname.empty()) {
339 this->init_exception_msg =
340 "Can't init BMI C++ model; empty name given for module's create function.";
341 throw std::runtime_error(this->init_exception_msg);
342 }
343 if (model_destroy_fname.empty()){
344 this->init_exception_msg =
345 "Can't init BMI C++ model; empty name given for module's destroy function.";
346 throw std::runtime_error(this->init_exception_msg);
347 }
348 dynamic_library_load();
349 }
350 void *symbol;
351 //Cpp_Bmi *(*dynamic_create_bmi)(Cpp_Bmi *model);
352 ModelCreator dynamic_creator;
353
354 try {
355 // Acquire the BMI creator function
356 symbol = dynamic_load_symbol(model_create_fname);
357 dynamic_creator = (ModelCreator) symbol;
358 bmi_model = std::shared_ptr<Cpp_Bmi>(
359 dynamic_creator(),
360 [](Cpp_Bmi* p) {
361 // DO NOTHING.
362 /* The best practice pattern for using C++ objects with dlopen is to declare a creator function
363 * AND a destroyer function. It is not wise to create the object in the library and destroy it
364 * in the calling environment because reasons (https://stackoverflow.com/a/40109212). Originally
365 * here I had a call to the destroyer function, but that blew up because in our implementation
366 * dlclose() gets called before this shared_ptr gets decremented to zero, so segfault happens
367 * when calling the library-resident destroyer function. It would probably be simpler to skip
368 * shared_ptr entirely and just keep a raw pointer to the object, but that makes it look like
369 * we're being ignorant of smart pointers...so this is left here as an explicit NO-OP and the
370 * destroyer is called in Finalize() above, which occurs in the right sequence. Better ideas?
371 *
372 * Note also that the creator function is called in our *constructor*, not in Initialize(), so
373 * this is somewhat asymmetric...however, BMI's documentation only stipulates that Initialize()
374 * "should perform all tasks that are to take place before entering the model’s time loop," so
375 * it seems legal to e.g. call GetInputVarNames() before Initialize()--which would require the
376 * backing model to already exist--so creating it in Initialize() is not an option.
377 */
378 }
379 );
380 }
381 catch (const ::external::ExternalIntegrationException &e) {
382 // "Override" the default message in this case
383 this->init_exception_msg =
384 "Cannot init " + this->model_name + ", failed to acquire creator and/or destroyer functions: " +
385 this->init_exception_msg;
386 throw ::external::ExternalIntegrationException(this->init_exception_msg);
387 }
388 }

454 void finalizeForCppAdapter() {
455 if (bmi_model != nullptr && model_initialized) {
456 model_initialized = false;
457
458 //C++ Models should throw their own exceptions...do we want to wrap in State_Exception?
459 // It looks like the Python adapter just marshals native exceptions, so guessing not.
460 bmi_model->Finalize();
461
462 // Acquire and call the BMI destroyer function
463 ModelDestroyer dynamic_destroyer;

465 dynamic_destroyer = (ModelDestroyer) symbol;
466 dynamic_destroyer(this->bmi_model.get());
