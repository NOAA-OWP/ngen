332 inline void execModuleCreation() {
333 if (get_dyn_lib_handle() == nullptr) {
334 if (model_create_fname.empty()) {
335 this->init_exception_msg =
336 "Can't init BMI C++ model; empty name given for module's create function.";
337 throw std::runtime_error(this->init_exception_msg);
338 }
339 if (model_destroy_fname.empty()){
340 this->init_exception_msg =
341 "Can't init BMI C++ model; empty name given for module's destroy function.";
342 throw std::runtime_error(this->init_exception_msg);
343 }
344 dynamic_library_load();
345 }
346 void *symbol;
347 //Cpp_Bmi *(*dynamic_create_bmi)(Cpp_Bmi *model);
348 ModelCreator dynamic_creator;
349
350 try {
351 // Acquire the BMI creator function
352 symbol = dynamic_load_symbol(model_create_fname);
353 dynamic_creator = (ModelCreator) symbol;
354 bmi_model = std::shared_ptr<Cpp_Bmi>(
355 dynamic_creator(),
356 [](Cpp_Bmi* p) {
357 // DO NOTHING.
358 /* The best practice pattern for using C++ objects with dlopen is to declare a creator function
359 * AND a destroyer function. It is not wise to create the object in the library and destroy it
360 * in the calling environment because reasons (https://stackoverflow.com/a/40109212). Originally
361 * here I had a call to the destroyer function, but that blew up because in our implementation
362 * dlclose() gets called before this shared_ptr gets decremented to zero, so segfault happens
363 * when calling the library-resident destroyer function. It would probably be simpler to skip
364 * shared_ptr entirely and just keep a raw pointer to the object, but that makes it look like
365 * we're being ignorant of smart pointers...so this is left here as an explicit NO-OP and the
366 * destroyer is called in Finalize() above, which occurs in the right sequence. Better ideas?
367 *
368 * Note also that the creator function is called in our *constructor*, not in Initialize(), so
369 * this is somewhat asymmetric...however, BMI's documentation only stipulates that Initialize()
370 * "should perform all tasks that are to take place before entering the model’s time loop," so
371 * it seems legal to e.g. call GetInputVarNames() before Initialize()--which would require the
372 * backing model to already exist--so creating it in Initialize() is not an option.
373 */
374 }
375 );
376 }
377 catch (const ::external::ExternalIntegrationException &e) {
378 // "Override" the default message in this case
379 this->init_exception_msg =
380 "Cannot init " + this->model_name + ", failed to acquire creator and/or destroyer functions: " +
381 this->init_exception_msg;
382 throw ::external::ExternalIntegrationException(this->init_exception_msg);
383 }
384 }

450 void finalizeForCppAdapter() {
451 if (bmi_model != nullptr && model_initialized) {
452 model_initialized = false;
453
454 //C++ Models should throw their own exceptions...do we want to wrap in State_Exception?
455 // It looks like the Python adapter just marshals native exceptions, so guessing not.
456 bmi_model->Finalize();
457
458 // Acquire and call the BMI destroyer function
459 ModelDestroyer dynamic_destroyer;
