323 inline void execModuleCreation() {
324 if (get_dyn_lib_handle() == nullptr) {
325 if (model_create_fname.empty()) {
326 this->init_exception_msg =
327 "Can't init BMI C++ model; empty name given for module's create function.";
328 throw std::runtime_error(this->init_exception_msg);
329 }
330 if (model_destroy_fname.empty()){
331 this->init_exception_msg =
332 "Can't init BMI C++ model; empty name given for module's destroy function.";
333 throw std::runtime_error(this->init_exception_msg);
334 }
335 dynamic_library_load();
336 }
337 void *symbol;
338 //Cpp_Bmi *(*dynamic_create_bmi)(Cpp_Bmi *model);
339 ModelCreator dynamic_creator;
340
341 try {
342 // Acquire the BMI creator function
343 symbol = dynamic_load_symbol(model_create_fname);
344 dynamic_creator = (ModelCreator) symbol;
345 bmi_model = std::shared_ptr<Cpp_Bmi>(
346 dynamic_creator(),
347 [](Cpp_Bmi* p) {
348 // DO NOTHING.
349 /* The best practice pattern for using C++ objects with dlopen is to declare a creator function
350 * AND a destroyer function. It is not wise to create the object in the library and destroy it
351 * in the calling environment because reasons (https://stackoverflow.com/a/40109212). Originally
352 * here I had a call to the destroyer function, but that blew up because in our implementation
353 * dlclose() gets called before this shared_ptr gets decremented to zero, so segfault happens
354 * when calling the library-resident destroyer function. It would probably be simpler to skip
355 * shared_ptr entirely and just keep a raw pointer to the object, but that makes it look like
356 * we're being ignorant of smart pointers...so this is left here as an explicit NO-OP and the
357 * destroyer is called in Finalize() above, which occurs in the right sequence. Better ideas?
358 *
359 * Note also that the creator function is called in our *constructor*, not in Initialize(), so
360 * this is somewhat asymmetric...however, BMI's documentation only stipulates that Initialize()
361 * "should perform all tasks that are to take place before entering the model’s time loop," so
362 * it seems legal to e.g. call GetInputVarNames() before Initialize()--which would require the
363 * backing model to already exist--so creating it in Initialize() is not an option.
364 */
365 }
366 );
367 }
368 catch (const ::external::ExternalIntegrationException &e) {
369 // "Override" the default message in this case
370 this->init_exception_msg =
371 "Cannot init " + this->model_name + ", failed to acquire creator and/or destroyer functions: " +
372 this->init_exception_msg;
373 throw ::external::ExternalIntegrationException(this->init_exception_msg);
374 }
375 }

441 void finalizeForCppAdapter() {
442 if (bmi_model != nullptr && model_initialized) {
443 model_initialized = false;
444
445 //C++ Models should throw their own exceptions...do we want to wrap in State_Exception?
446 // It looks like the Python adapter just marshals native exceptions, so guessing not.
447 bmi_model->Finalize();

450 ModelDestroyer dynamic_destroyer;
