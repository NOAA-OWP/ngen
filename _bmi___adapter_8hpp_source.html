93 void acquire_time_conversion_factor(double &time_convert_factor) {
94 std::string time_units = GetTimeUnits();
95 if (time_units == "s" || time_units == "sec" || time_units == "second" || time_units == "seconds")
96 time_convert_factor = 1.0;
97 else if (time_units == "m" || time_units == "min" || time_units == "minute" ||
98 time_units == "minutes")
99 time_convert_factor = 60.0;
100 else if (time_units == "h" || time_units == "hr" || time_units == "hour" || time_units == "hours")
101 time_convert_factor = 3600.0;
102 else if (time_units == "d" || time_units == "day" || time_units == "days")
103 time_convert_factor = 86400.0;
104 else
105 throw std::runtime_error(
106 "Invalid model time step units ('" + time_units + "') in " + model_name + ".");
107 }

174 void Initialize() {
175 // If there was previous init attempt but w/ failure exception, throw runtime error and include previous
176 // message
177 if (model_initialized && !init_exception_msg.empty()) {
178 throw std::runtime_error(
179 "Previous " + model_name + " init attempt had exception: \n\t" + init_exception_msg);
180 }
181 // If there was previous init attempt w/ (implicitly) no exception on previous attempt, just return
182 else if (model_initialized) {
183 return;
184 }
185 else if (!utils::FileChecker::file_is_readable(bmi_init_config)) {
186 init_exception_msg = "Cannot initialize " + model_name + " using unreadable file '"
187 + bmi_init_config + "'";
188 throw std::runtime_error(init_exception_msg);
189 }
190 else {
191 try {
192 // TODO: make this same name as used with other testing (adjust name in docstring above also)
193 construct_and_init_backing_model();
194 // Make sure this is set to 'true' after this function call finishes
195 model_initialized = true;
196 acquire_time_conversion_factor(bmi_model_time_convert_factor);
197 }
198 // Record the exception message before re-throwing to handle subsequent function calls properly
199 catch (std::exception& e) {
200 // Make sure this is set to 'true' after this function call finishes
201 model_initialized = true;
202 throw e;
203 }
204 }
205 }

225 void Initialize(std::string config_file) override {
226 if (config_file != bmi_init_config && model_initialized) {
227 throw std::runtime_error(
228 "Model init previously attempted; cannot change config from " + bmi_init_config + " to "
229 + config_file);
230 }
231
232 if (config_file != bmi_init_config && !model_initialized) {
233 output.put("Warning: initialization call changes model config from " + bmi_init_config + " to "
234 + config_file);
235 bmi_init_config = config_file;
236 }
237 try {
238 Initialize();
239 }
240 catch (models::external::State_Exception &e) {
241 throw e;
242 }
243 catch (std::exception &e) {
244 throw std::runtime_error(e.what());
245 }
