164 void Initialize() {
165 // If there was previous init attempt but w/ failure exception, throw runtime error and include previous
166 // message
167 if (model_initialized && !init_exception_msg.empty()) {
168 throw std::runtime_error(
169 "Previous " + model_name + " init attempt had exception: \n\t" + init_exception_msg);
170 }
171 // If there was previous init attempt w/ (implicitly) no exception on previous attempt, just return
172 else if (model_initialized) {
173 return;
174 }
175 else if (!utils::FileChecker::file_is_readable(bmi_init_config)) {
176 init_exception_msg = "Cannot initialize " + model_name + " using unreadable file '"
177 + bmi_init_config + "'";
178 throw std::runtime_error(init_exception_msg);
179 }
180 else {
181 try {
182 // TODO: make this same name as used with other testing (adjust name in docstring above also)
183 construct_and_init_backing_model();
184 // Make sure this is set to 'true' after this function call finishes
185 model_initialized = true;
186 bmi_model_time_convert_factor = get_time_convert_factor();
187 }
188 // Record the exception message before re-throwing to handle subsequent function calls properly
189 catch (std::exception& e) {
190 // Make sure this is set to 'true' after this function call finishes
191 model_initialized = true;
192 throw e;
193 }
194 }
195 }

215 void Initialize(std::string config_file) override {
216 if (config_file != bmi_init_config && model_initialized) {
217 throw std::runtime_error(
218 "Model init previously attempted; cannot change config from " + bmi_init_config + " to "
219 + config_file);
220 }
221
222 if (config_file != bmi_init_config && !model_initialized) {
223 output.put("Warning: initialization call changes model config from " + bmi_init_config + " to "
224 + config_file);
225 bmi_init_config = config_file;
226 }
227 try {
228 Initialize();
229 }
230 catch (models::external::State_Exception &e) {
231 throw e;
232 }
233 catch (std::exception &e) {
234 throw std::runtime_error(e.what());
235 }
236 }
