95  void acquire_time_conversion_factor(double &time_convert_factor) {
96  string time_units = GetTimeUnits();
97  if (time_units == "s" || time_units == "sec" || time_units == "second" || time_units == "seconds")
98  time_convert_factor = 1.0;
99  else if (time_units == "m" || time_units == "min" || time_units == "minute" ||
100  time_units == "minutes")
101  time_convert_factor = 60.0;
102  else if (time_units == "h" || time_units == "hr" || time_units == "hour" || time_units == "hours")
103  time_convert_factor = 3600.0;
104  else if (time_units == "d" || time_units == "day" || time_units == "days")
105  time_convert_factor = 86400.0;
106  else
107  throw runtime_error(
108  "Invalid model time step units ('" + time_units + "') in " + model_name + ".");
109  }

176  void Initialize() {
177  // If there was previous init attempt but w/ failure exception, throw runtime error and include previous
178  // message
179  if (model_initialized && !init_exception_msg.empty()) {
180  throw runtime_error(
181  "Previous " + model_name + " init attempt had exception: \n\t" + init_exception_msg);
182  }
183  // If there was previous init attempt w/ (implicitly) no exception on previous attempt, just return
184  else if (model_initialized) {
185  return;
186  }
187  else if (!utils::FileChecker::file_is_readable(bmi_init_config)) {
188  init_exception_msg = "Cannot initialize " + model_name + " using unreadable file '"
189  + bmi_init_config + "'";
190  throw runtime_error(init_exception_msg);
191  }
192  else {
193  try {
194  // TODO: make this same name as used with other testing (adjust name in docstring above also)
195  construct_and_init_backing_model();
196  // Make sure this is set to 'true' after this function call finishes
197  model_initialized = true;
198  acquire_time_conversion_factor(bmi_model_time_convert_factor);
199  }
200  // Record the exception message before re-throwing to handle subsequent function calls properly
201  catch (exception& e) {
202  // Make sure this is set to 'true' after this function call finishes
203  model_initialized = true;
204  throw e;
205  }
206  }
207  }

227  void Initialize(string config_file) override {
228  if (config_file != bmi_init_config && model_initialized) {
229  throw runtime_error(

231  + config_file);

235  output.put("Warning: initialization call changes model config from " + bmi_init_config + " to "
236  + config_file);

238  }
