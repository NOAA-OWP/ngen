379 template<class T>
380 void SetValue(std::string name, std::vector<T> src) {
381 int item_size, total_bytes;
382 try {
383 item_size = GetVarItemsize(name);
384 total_bytes = GetVarNbytes(name);
385 }
386 catch (std::runtime_error &e) {
387 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
388 "; unable to test item and array sizes are equal (does variable " + name +
389 " exist for model " + model_name + "?)");
390 }
391 if (item_size != sizeof(src[0])) {
392 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
393 " with values of different item size");
394 }
395 if (src.size() != total_bytes / item_size) {
396 throw std::runtime_error(
397 "Cannot set " + name + " variable of " + model_name + " from vector of size " +
398 std::to_string(src.size) + " (expected size " + std::to_string(total_bytes / item_size) +
399 ")");
400 }
401 SetValue(std::move(name), static_cast<void *>(src.data()));
402 }

686 std::shared_ptr<std::vector<std::string>> inner_get_variable_names(bool is_input_variables) {
687 // Writing this def once here at beginning: fewer lines, but this may or may not be used.
688 std::string varType = (is_input_variables) ? "input" : "output";
689
690 // Obtain this via inner functions, which should use the model directly and not other member functions
691 int variableCount;
692 try {
693 variableCount = (is_input_variables) ? inner_get_input_item_count() : inner_get_output_item_count();
694 }
695 catch (const std::exception &e) {
696 throw std::runtime_error(model_name + " failed to count of " + varType + " variable names array.");
697 }
698
699 // With variable count now obtained, create the vector
700 std::shared_ptr<std::vector<std::string>> var_names = std::make_shared<std::vector<std::string>>(
701 std::vector<std::string>(variableCount));
702
703 // Must get the names from the model as an array of C strings
704 // The array can be on the stack ...
705 char* names_array[variableCount];
706 // ... but allocate the space for the individual C strings (i.e., the char * elements)
707 for (int i = 0; i < variableCount; i++) {
708 names_array[i] = static_cast<char *>(malloc(sizeof(char) * BMI_MAX_VAR_NAME));
709 }
710
711 // With the necessary char** in hand, get the names from the model
712 int names_result;
713 if (is_input_variables) {
714 names_result = get_input_var_names(bmi_model.get(), names_array);
715 }
716 else {
717 names_result = get_output_var_names(bmi_model.get(), names_array);
718 }
719 if (names_result != BMI_SUCCESS) {
720 throw std::runtime_error(model_name + " failed to get array of " + varType + " variables names.");
721 }
722
723 // Then convert from array of C strings to vector of strings, freeing the allocated space as we go
724 for (int i = 0; i < variableCount; ++i) {
725 (*var_names)[i] = std::string(names_array[i]);
726 free(names_array[i]);
727 }
728
729 return var_names;
730 }
