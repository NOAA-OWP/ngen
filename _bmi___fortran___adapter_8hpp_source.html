352 template<class T>
353 void SetValue(std::string name, std::vector<T> src) {
354 int item_size, total_bytes;
355 try {
356 item_size = GetVarItemsize(name);
357 total_bytes = GetVarNbytes(name);
358 }
359 catch (std::runtime_error &e) {
360 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
361 "; unable to test item and array sizes are equal (does variable " + name +
362 " exist for model " + model_name + "?)");
363 }
364 if (item_size != sizeof(src[0])) {
365 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
366 " with values of different item size");
367 }
368 if (src.size() != total_bytes / item_size) {
369 throw std::runtime_error(
370 "Cannot set " + name + " variable of " + model_name + " from vector of size " +
371 std::to_string(src.size()) + " (expected size " + std::to_string(total_bytes / item_size) +
372 ")");
373 }
374 SetValue(std::move(name), static_cast<void *>(src.data()));
375 }

659 std::shared_ptr<std::vector<std::string>> inner_get_variable_names(bool is_input_variables) {
660 // Writing this def once here at beginning: fewer lines, but this may or may not be used.
661 std::string varType = (is_input_variables) ? "input" : "output";
662
663 // Obtain this via inner functions, which should use the model directly and not other member functions
664 int variableCount;
665 try {
666 variableCount = (is_input_variables) ? inner_get_input_item_count() : inner_get_output_item_count();
667 }
668 catch (const std::exception &e) {
669 throw std::runtime_error(model_name + " failed to count of " + varType + " variable names array.");
670 }
671
672 // With variable count now obtained, create the vector
673 std::shared_ptr<std::vector<std::string>> var_names = std::make_shared<std::vector<std::string>>(
674 std::vector<std::string>(variableCount));
675
676 // Must get the names from the model as an array of C strings
677 // The array can be on the stack ...
678 char* names_array[variableCount];
679 // ... but allocate the space for the individual C strings (i.e., the char * elements)
680 for (int i = 0; i < variableCount; i++) {
681 names_array[i] = static_cast<char *>(malloc(sizeof(char) * BMI_MAX_VAR_NAME));
682 }
683
684 // With the necessary char** in hand, get the names from the model
685 int names_result;
686 if (is_input_variables) {
687 names_result = get_input_var_names(bmi_model.get(), names_array);
688 }
689 else {
690 names_result = get_output_var_names(bmi_model.get(), names_array);
691 }
692 if (names_result != BMI_SUCCESS) {
693 throw std::runtime_error(model_name + " failed to get array of " + varType + " variables names.");
694 }
695
696 // Then convert from array of C strings to vector of strings, freeing the allocated space as we go
697 for (int i = 0; i < variableCount; ++i) {
698 (*var_names)[i] = std::string(names_array[i]);
699 free(names_array[i]);
700 }
701
702 return var_names;
703 }
