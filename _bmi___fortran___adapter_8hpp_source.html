355 template<class T>
356 void SetValue(std::string name, std::vector<T> src) {
357 int item_size, total_bytes;
358 try {
359 item_size = GetVarItemsize(name);
360 total_bytes = GetVarNbytes(name);
361 }
362 catch (std::runtime_error &e) {
363 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
364 "; unable to test item and array sizes are equal (does variable " + name +
365 " exist for model " + model_name + "?)");
366 }
367 if (item_size != sizeof(src[0])) {
368 throw std::runtime_error("Cannot set " + name + " variable of " + model_name +
369 " with values of different item size");
370 }
371 if (src.size() != total_bytes / item_size) {
372 throw std::runtime_error(
373 "Cannot set " + name + " variable of " + model_name + " from vector of size " +
374 std::to_string(src.size()) + " (expected size " + std::to_string(total_bytes / item_size) +
375 ")");
376 }
377 SetValue(std::move(name), static_cast<void *>(src.data()));
378 }

662 std::shared_ptr<std::vector<std::string>> inner_get_variable_names(bool is_input_variables) {
663 // Writing this def once here at beginning: fewer lines, but this may or may not be used.
664 std::string varType = (is_input_variables) ? "input" : "output";
665
666 // Obtain this via inner functions, which should use the model directly and not other member functions
667 int variableCount;
668 try {
669 variableCount = (is_input_variables) ? inner_get_input_item_count() : inner_get_output_item_count();
670 }
671 catch (const std::exception &e) {
672 throw std::runtime_error(model_name + " failed to count of " + varType + " variable names array.");
673 }
674
675 // With variable count now obtained, create the vector
676 std::shared_ptr<std::vector<std::string>> var_names = std::make_shared<std::vector<std::string>>(
677 std::vector<std::string>(variableCount));
678
679 // Must get the names from the model as an array of C strings
680 // The array can be on the stack ...
681 std::vector<char*> names_array(variableCount);
682 // ... but allocate the space for the individual C strings (i.e., the char * elements)
683 for (int i = 0; i < variableCount; i++) {
684 names_array[i] = static_cast<char *>(malloc(sizeof(char) * BMI_MAX_VAR_NAME));
685 }
686
687 // With the necessary char** in hand, get the names from the model
688 int names_result;
689 if (is_input_variables) {
690 names_result = get_input_var_names(bmi_model.get(), names_array.data());
691 }
692 else {
693 names_result = get_output_var_names(bmi_model.get(), names_array.data());
694 }
695 if (names_result != BMI_SUCCESS) {
696 throw std::runtime_error(model_name + " failed to get array of " + varType + " variables names.");
697 }
698
699 // Then convert from array of C strings to vector of strings, freeing the allocated space as we go
700 for (int i = 0; i < variableCount; ++i) {
701 (*var_names)[i] = std::string(names_array[i]);
702 free(names_array[i]);
703 }
704
705 return var_names;
706 }
