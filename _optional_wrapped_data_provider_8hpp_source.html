193 double get_value(const CatchmentAggrDataSelector& selector, data_access::ReSampleMethod m) override
194 {
195 std::string output_name = selector.get_variable_name();
196 time_t init_time = selector.get_init_time();
197 const long duration_s = selector.get_duration_secs();
198 const std::string output_units = selector.get_output_units();
199
200 // Balk if not in this instance's collection of outputs
201 if (find(providedOutputs.begin(), providedOutputs.end(), output_name) == providedOutputs.end()) {
202 throw std::runtime_error("Unknown output " + output_name + " requested from wrapped provider");
203 }
204 // Balk if this instance is not ready
205 if (!isReadyToProvideData()) {
206 throw std::runtime_error("Cannot get value for " + output_name
207 + " from optional wrapped provider before it is ready");
208 }
209
210 // Remember: when the instance is ready to provide data, one or both of these **must** be true:
211 // * the desired output has a default value available
212 // * the desired output is provided by the backing wrapped provider
213
214 // If this output is not available from a wrapped provider OR we should override, then return the default
215 // Remember: isDefaultOverride() is not the same as "should use default" (see function docstring)
216 if (!isSuppliedByWrappedProvider(output_name) || isDefaultOverride(output_name)) {
217 // Take note if/how often default gets used in case that information is needed later
218 recordUsingDefault(output_name);
219 return defaultValues.at(output_name);
220 }
221 // Otherwise (i.e., available from wrapped provider and no override), get from backing wrapped provider
222 else {
223 return wrapped_provider->get_value(selector,m);
224 }
225 }

249 bool isDefaultOverride(const std::string &output_name) {
250 // First, there must be a default, and there must be something to override
251 if (!isSuppliedWithDefault(output_name) || !isSuppliedByWrappedProvider(output_name)) {
252 return false;
253 }
254 // Must have positive wait indicator value
255 const auto &wait_it = defaultUsageWaits.find(output_name);
256 return wait_it != defaultUsageWaits.end() && wait_it->second > 0;
257 }

311 bool setWrappedProvider(
312 GenericDataProvider *provider) override {
313 // Disallow re-setting the provider if already validly set
314 // TODO: consider if this is still appropriate, given backing providers can be valid without providing
315 // everything (because of defaults), and so a new provider may supply more or less of what is needed
316 if (isWrappedProviderSet()) {
317 setMessage = "Cannot change wrapped provider after a valid provide has already been set";
318 return false;
319 }
320
321 // Confirm this will provide anything
322 if (provider == nullptr) {
323 setMessage = "Cannot set provider as null, as this cannot provide the values this type must proxy";
324 return false;
325 }
326
327 // Check this provides everything needed, accounting for defaults (and also tallying the outputs provided)
328 unsigned short providedByProviderCount = 0;
329 for (const std::string &requiredName : providedOutputs) {
330 // If supplied by the provider, increment our count and continue to the next required output name
331 if (isSuppliedByProvider(requiredName, provider)) {
332 ++providedByProviderCount;
333 continue;
334 }
335 // If something isn't supplied by the provider (i.e., the "else"), there has to be a default, or we bail
336 else if (!isSuppliedWithDefault(requiredName)) {
337 setMessage = "Given provider does not provide the required " + requiredName;
338 return false;
339 }
340 }
341
342 // Also make sure it provides at least 1 relevant output (possible no outputs were "needed" due to defaults)
343 if (providedByProviderCount == 0) {
344 setMessage = "Given provider does not provide any required outputs (all satisfied via defaults)";
345 return false;
346 }
347
348 // If this is good, set things and return true
349 wrapped_provider = provider;
350 setMessage.clear();
351 return true;
352 }

373 virtual void recordUsingDefault(const std::string &output_name) {
374 // Don't bother doing anything if there aren't waits assigned for this output
375 // Also, in this implementation, don't count usages until there is a backing provider that can provide this
376 auto waits_it = defaultUsageWaits.find(output_name);
377 if (waits_it != defaultUsageWaits.end() && isSuppliedByWrappedProvider(output_name)) {
378 // If this was not the last required default usage, simply decrement the wait count
379 if (waits_it->second > 1) {
380 waits_it->second -= 1;
381 }
382 // If this was the last required default usage wait, clear this entry from the collection
383 // Also, right now we are only supporting positive counts, so remove anything else as well
384 else {
385 defaultUsageWaits.erase(waits_it);
386 }
387 }
