16// evapotranspiration (ET) module,
17// Version 1.0 by Fred L. Ogden, NOAA-NWS-OWP, May, 2020.
18//
19// includes five different methods to clculate ET, from Chow, Maidment & Mays Textbook, and UNFAO Penman-Monteith:
20// 1. energy balance method
21// 2. aerodynamic method
22// 3. combination method, which combines 1 & 2.
23// 4. Priestley-Taylor method, which assumes the ratio between 1 & 2, and only calculates 1.
24// 5. Penman-Monteith method, which requires a value of canopy resistance term, and does not rely on 1 or 2.
25//
26// This subroutine requires a considerable amount of meteorological data as input.
27// a. temperature and (relative-humidity or specific humidity) and the heights at which they are measured.
28// b. near surface wind speed measurement and the height at which it was measured.
29// c. the ambient atmospheric temperature lapse rate
30// d. the fraction of the sky covered by clouds
31// e. (optional) the height above ground to the cloud base. If not provided, then assumed.
32// f. the day of the year (1-366) and time of day (UTC only!)
33// g. the skin temperature of the earth's surface, TODO: should come from another module to calc. soil or veg. temp.
34// h. the zero-plane roughness height of the atmospheric boundary layer assuming log-law behavior (from land cover)
35// i. the average root zone soil temperature, or near-surface water temperature in the case of lake evaporation.
36// j. the incoming solar (shortwave) radiation. If not provided it is computed from d,e,f, using an
37// updated method similar to the one presented in Bras, R.L. Hydrology. Requires value of the Linke atmospheric
38// turbidity factor, which varies from 2 for clear mountain air to 5 for smoggy air. According to Hove & Manyumbu
39// 2012, who calculated values over Zimbabwe that varied from 2.14 to 3.71. Other values exist in the literature.
40// TODO: This turbidity factor could be calculated from satellite obs. or maybe NOAA already does this?
41//
42// All radiation calculations needed for 1, 3, 4, and 5 require net radiation calculations at the land surface.
43// the net radiation is calculated using a, c, d, e, f, g, j, plus the Linke turbidity factor, which can be estimate
44// from satellite observations.
45//
46
47// NOTE THE VALUE OF evapotranspiration_params.zero_plane_displacement_height COMES FROM LAND COVER DATA.
48// Taken from: https://websites.pmc.ucsc.edu/~jnoble/wind/extrap/
49//
50//Roughness Roughness Landscape Type
51// Class Length (m)
52//----------------------------------------------------------------------------------------------------------------
53// 0 0.0002 Smooth water surface
54// 0.2 0.0005 Inlet water
55// 0.5 0.0024 Completely open terrain, smooth surface, e.g. concrete runways in airports, mowed grass, etc.
56// 1 0.03 Open agricultural area without fences and hedgerows and very scattered buildings. Only softly
57// rounded hills
58// 1.5 0.055 Agricultural land with some houses and 8 metre tall sheltering hedgerows with a distance of
59// approximately 1250 metres
60// 2 0.1 Agricultural land with some houses and 8 metre tall sheltering hedgerows with a distance of
61// approximately 500 metres
62// 2.5 0.2 Agricultural land with many houses, shrubs and plants, or 8 metre tall sheltering hedgerows
63// with a distance of approximately 250 metres
64// 3 0.4 Villages, small towns, agricultural land with many or tall sheltering hedgerows, forests
65// and very rough and uneven terrain
66// 3.5 0.8 Larger cities with tall buildings
67// 4 1.6 Very large cities with tall buildings and skyscrapers
68// Roughness definitions according to the European Wind Atlas.
69//
70// According to the UN FAO Penman-Monteith example here:
71// http://www.fao.org/3/X0490E/x0490e06.htm#aerodynamic%20resistance%20(ra)
72// the zero plane roughness length,"d" can be approximated as 2/3 of the vegetation height (H): d=2/3*H.
73// the momentum roughness height "zom" can be estimated as 0.123*H.
74// the heat transfer roughness height "zoh" can be approximated as 0.1 * zom.

317//############################################################*
318// subroutine to calculate evapotranspiration using *
319// Chow, Maidment, and Mays textbook *
320// F.L. Ogden, NOAA National Weather Service, 2020 *
321//############################################################*
322extern double evapotranspiration_energy_balance_method
323(
324evapotranspiration_options *et_options,
325evapotranspiration_params *et_params,
326evapotranspiration_forcing *et_forcing
327)
328{
329// local varibles
330double water_latent_heat_of_vaporization_J_per_kg;
331double liquid_water_density_kg_per_m3;
332double lambda_et;
333double radiation_balance_evapotranspiration_rate_m_per_s;
334
335// from FAO document: cp specific heat at constant pressure, 1.013 10-3 [MJ kg-1 C-1],
336
337// IF SOIL WATER TEMPERATURE NOT PROVIDED, USE A SANE VALUE
338if(100.0 > et_forcing->water_temperature_C) et_forcing->water_temperature_C=22.0; // growing season
339
340// CALCULATE VARS NEEDED FOR THE ALL METHODS:
341
342liquid_water_density_kg_per_m3 = calc_liquid_water_density_kg_per_m3(et_forcing->water_temperature_C); // rho_w
343
344water_latent_heat_of_vaporization_J_per_kg=2.501e+06-2370.0*et_forcing->water_temperature_C; // eqn 2.7.6 Chow etal.
345 // aka 'lambda'
346
347// We need this in all options except for aerodynamic or Penman-Monteith methods.
348// Radiation balance is the simplest method. Involves only radiation calculations, no aerodynamic calculations.
349
350lambda_et=0.0;
351if( (et_options->use_aerodynamic_method == FALSE ) && (et_options->use_penman_monteith_method==FALSE) )
352 {
353 // This is equation 3.5.9 from Chow, Maidment, and Mays textbook.
354 lambda_et=et_forcing->net_radiation_W_per_sq_m;
355 radiation_balance_evapotranspiration_rate_m_per_s=lambda_et/
356 (liquid_water_density_kg_per_m3*water_latent_heat_of_vaporization_J_per_kg);
357 }
358return(radiation_balance_evapotranspiration_rate_m_per_s);
359}

367(

369evapotranspiration_params *et_params,
370evapotranspiration_forcing *et_forcing,
371intermediate_vars *inter_vars

390liquid_water_density_kg_per_m3 = inter_vars->liquid_water_density_kg_per_m3;

401 {

403 mass_flux = 0.622*von_karman_constant_squared*moist_air_density_kg_per_m3* // kg per sq. meter per sec.
404 vapor_pressure_deficit_Pa*et_forcing->wind_speed_m_per_s/
405 (et_forcing->air_pressure_Pa*
406 pow(log(et_params->wind_speed_measurement_height_m/et_params->zero_plane_displacement_height_m),2.0));
407
408 aerodynamic_method_evapotranspiration_rate_m_per_s=mass_flux/liquid_water_density_kg_per_m3;
409 }
410
411return(aerodynamic_method_evapotranspiration_rate_m_per_s);
412}

415// subroutine to calculate evapotranspiration using *
416// Chow, Maidment, and Mays textbook *
417// F.L. Ogden, NOAA National Weather Service, 2020 *
418//############################################################*
419extern double evapotranspiration_combination_method
420(
421evapotranspiration_options *et_options,
422evapotranspiration_params *et_params,
423evapotranspiration_forcing *et_forcing,
424intermediate_vars *inter_vars
425)
426{
427// local varibles
428double psychrometric_constant_Pa_per_C;
429double slope_sat_vap_press_curve_Pa_s;
430double air_saturation_vapor_pressure_Pa;
431double air_actual_vapor_pressure_Pa;
432double moist_air_density_kg_per_m3;
433double water_latent_heat_of_vaporization_J_per_kg;
434double moist_air_gas_constant_J_per_kg_K;
435double vapor_pressure_deficit_Pa;
436double liquid_water_density_kg_per_m3;
437double lambda_et;
438double radiation_balance_evapotranspiration_rate_m_per_s;
439double aerodynamic_method_evapotranspiration_rate_m_per_s;
440double instantaneous_et_rate_m_per_s;
441double mass_flux;
442double delta;
443double gamma;
444double von_karman_constant_squared=(double)KV2; // a constant equal to 0.41 squared
445
446calculate_intermediate_variables(et_options, et_params, et_forcing, inter_vars);
447
448liquid_water_density_kg_per_m3 = inter_vars->liquid_water_density_kg_per_m3;
449water_latent_heat_of_vaporization_J_per_kg=inter_vars->water_latent_heat_of_vaporization_J_per_kg;
450air_saturation_vapor_pressure_Pa=inter_vars->air_saturation_vapor_pressure_Pa;
451air_actual_vapor_pressure_Pa=inter_vars->air_actual_vapor_pressure_Pa;
452vapor_pressure_deficit_Pa=inter_vars->vapor_pressure_deficit_Pa;
453moist_air_gas_constant_J_per_kg_K=inter_vars->moist_air_gas_constant_J_per_kg_K;
454moist_air_density_kg_per_m3=inter_vars->moist_air_density_kg_per_m3;
455slope_sat_vap_press_curve_Pa_s=inter_vars->slope_sat_vap_press_curve_Pa_s;
456water_latent_heat_of_vaporization_J_per_kg=inter_vars->water_latent_heat_of_vaporization_J_per_kg;
457psychrometric_constant_Pa_per_C=inter_vars->psychrometric_constant_Pa_per_C;
458
459delta=slope_sat_vap_press_curve_Pa_s;
460gamma=psychrometric_constant_Pa_per_C;
461
462if(et_options->use_combination_method==TRUE)
463lambda_et=0.0;
464if( (et_options->use_aerodynamic_method == FALSE ) && (et_options->use_penman_monteith_method==FALSE) )
465 {
466 // This is equation 3.5.9 from Chow, Maidment, and Mays textbook.
467 lambda_et=et_forcing->net_radiation_W_per_sq_m;
468 radiation_balance_evapotranspiration_rate_m_per_s=lambda_et/
469 (liquid_water_density_kg_per_m3*water_latent_heat_of_vaporization_J_per_kg);
470 mass_flux = 0.622*von_karman_constant_squared*moist_air_density_kg_per_m3* // kg per sq. meter per sec.
471 vapor_pressure_deficit_Pa*et_forcing->wind_speed_m_per_s/
472 (et_forcing->air_pressure_Pa*
473 pow(log(et_params->wind_speed_measurement_height_m/et_params->zero_plane_displacement_height_m),2.0));
474
475 aerodynamic_method_evapotranspiration_rate_m_per_s=mass_flux/liquid_water_density_kg_per_m3;
476 }
477 {
478 // This is equation 3.5.26 from Chow, Maidment, and Mays textbook
479 instantaneous_et_rate_m_per_s=
480 delta/(delta+gamma)*radiation_balance_evapotranspiration_rate_m_per_s+
481 gamma/(delta+gamma)*aerodynamic_method_evapotranspiration_rate_m_per_s;
482 }
483return (instantaneous_et_rate_m_per_s);
484}

603//#####################################################################################*
604// subroutine to calculate latent heat flux (lambda*et) *
605// using the Penman-Monteith equation as described by FAO *
606// see: http://www.fao.org/3/X0490E/x0490e06.htm#aerodynamic%20resistance%20(ra) *
607// After Allen et al. ASCE Reference ET calculation method *
608// F.L. Ogden, NOAA National Weather Service, 2020 *
609//#####################################################################################*
610extern double penman_monteith_et_calculation
611(
612double delta,
613double gamma,
614double moist_air_density_kg_per_m3,
615double vapor_pressure_deficit_Pa,
616evapotranspiration_params *et_params,
617evapotranspiration_forcing *et_forcing
618)
619
620{
621// local varibles
622double numerator;
623double denominator;
624double aerodynamic_resistance_s_per_m;
625
626// this method requires more calculations
627if(is_fabs_less_than_epsilon(et_params->vegetation_height_m,1.0e-06)==TRUE)
628 {
629 // the vegetation height was not specified. TODO should warn??
630 fprintf(stderr,"WARNING: Vegetation height not specified in the Penman-Monteith routine. Using 0.5m.\n");
631 et_params->vegetation_height_m=0.5; // use a reasonable assumed value
632 }
633
634// use approximations from UN FAO: http://www.fao.org/3/X0490E/x0490e06.htm#aerodynamic%20resistance%20(ra)
635et_params->zero_plane_displacement_height_m=2.0/3.0*et_params->vegetation_height_m;
636et_params->momentum_transfer_roughness_length_m=0.123*et_params->vegetation_height_m;
637et_params->heat_transfer_roughness_length_m=0.1*et_params->vegetation_height_m;
638
639aerodynamic_resistance_s_per_m= calculate_aerodynamic_resistance
640 ( et_params->wind_speed_measurement_height_m,
641 et_params->humidity_measurement_height_m,
642 et_params->zero_plane_displacement_height_m,
643 et_params->momentum_transfer_roughness_length_m,
644 et_params->heat_transfer_roughness_length_m,
645 et_forcing->wind_speed_m_per_s);
646
647// all the ingredients have been prepared. Make Penman-Monteith soufle...
648// from: http://www.fao.org/3/X0490E/x0490e06.htm#aerodynamic%20resistance%20(ra)
649
650numerator = delta* (et_forcing->net_radiation_W_per_sq_m - et_forcing->ground_heat_flux_W_per_sq_m) +
651 moist_air_density_kg_per_m3 * CP *
652 vapor_pressure_deficit_Pa/aerodynamic_resistance_s_per_m;
653
654denominator = delta + gamma * (1.0+et_forcing->canopy_resistance_sec_per_m/aerodynamic_resistance_s_per_m);
655
656return(numerator/denominator); // Latent heat flux in Watts per sq. m., or J per (s m2)
657}

659//############################################################*
660// subroutine to calculate net radiation from all components *
661// of the radiation budget, using values provided by WRF *
662// Reference, Bras, R.L., HYDROLOGY an Introduction to *
663// Hydrologic Science, Addison Wesley, 1990. *
664// F.L. Ogden, NOAA National Weather Service, 2020 *
665//############################################################*
666extern double calculate_net_radiation_W_per_sq_m
667(
668evapotranspiration_options *et_options, // et_options.yes_aorc == TRUE if we are using aorc forcing values.
669surface_radiation_params *surf_rad_params,
670surface_radiation_forcing *surf_rad_forcing
671)
672
673{
674// local variables
675double net_radiation_W_per_sq_m;
676double outgoing_longwave_radiation_W_per_sq_m;
677double stefan_boltzmann_constant = SB; //W/m^2/K^4
678double atmosphere_longwave_emissivity; // dimensionless, based on water vapor content of air
679double saturation_water_vapor_partial_pressure_Pa;
680double actual_water_vapor_partial_pressure_Pa;
681double actual_water_vapor_mixing_ratio;
682double cloud_base_temperature_C;
683double surface_longwave_albedo;
684double N,Klw;
685
686
687
688// CALCULATE OUTGOING LONGWAVE RADIATION FLUX FROM SURFACE
689outgoing_longwave_radiation_W_per_sq_m=surf_rad_params->surface_longwave_emissivity*stefan_boltzmann_constant*
690 pow(surf_rad_forcing->surface_skin_temperature_C+TK,4.0);
691 // must convert C to K
692
693if(0.999 < surf_rad_params->surface_longwave_emissivity)
694 {
695 surface_longwave_albedo=0.0; // soil, rock, concrete, asphalt, vegetation, snow
696 }
697else
698 {
699 surface_longwave_albedo=0.03; // water - actually not this simple, but close enough for now
700 }
701
702if(et_options->yes_aorc==FALSE) // we must calculate longwave incoming from the atmosphere
703saturation_water_vapor_partial_pressure_Pa=calc_air_saturation_vapor_pressure_Pa(surf_rad_forcing->air_temperature_C);
704
705actual_water_vapor_partial_pressure_Pa=surf_rad_forcing->relative_humidity_percent/100.0*
706 saturation_water_vapor_partial_pressure_Pa;
707
708if(et_options->yes_aorc==FALSE)
709 {
710 // CALCULATE DOWNWELLING LONGWAVE RADIATION FLUX FROM ATMOSPHERE, W/m2.
711 if(0.90 < surf_rad_forcing->cloud_cover_fraction) // very nearly overcast or overcast
712 {
713 // calculate longwave downwelling using overcast equation, with emissivity of cloud base =1.0
714 cloud_base_temperature_C=surf_rad_forcing->air_temperature_C+
715 surf_rad_forcing->ambient_temperature_lapse_rate_deg_C_per_km*
716 surf_rad_forcing->cloud_base_height_m/1000.0;
717 surf_rad_forcing->incoming_longwave_radiation_W_per_sq_m=stefan_boltzmann_constant*
718 pow(cloud_base_temperature_C+TK,4.0);
719 }
720 else // not overcast, use TVA (1972) formulation, taken from Bras R.L., textbook, pg. 44.
721 {
722 // use cloudy skies formulation
723 // clear sky emissivity
724 atmosphere_longwave_emissivity = 0.740 + 0.0049*actual_water_vapor_partial_pressure_Pa/100.0; //conv. Pa to mb
725 N=surf_rad_forcing->cloud_cover_fraction;
726 Klw=(1.0+0.17*N*N); // effect of cloud cover from TVA (1972)
727 surf_rad_forcing->incoming_longwave_radiation_W_per_sq_m=atmosphere_longwave_emissivity*Klw*
728 stefan_boltzmann_constant*
729 pow(surf_rad_forcing->air_temperature_C+TK,4.0);
730 }
731 }
732
733net_radiation_W_per_sq_m=(1.0-surf_rad_params->surface_shortwave_albedo)*
734 surf_rad_forcing->incoming_shortwave_radiation_W_per_sq_m +
735 (1.0-surface_longwave_albedo)*
736 surf_rad_forcing->incoming_longwave_radiation_W_per_sq_m -
737 outgoing_longwave_radiation_W_per_sq_m; // this is plus, negative grnd ht flx is downward
738
739
740return(net_radiation_W_per_sq_m);
741}

743//############################################################*
744// subroutine to calculate aerodynamic resistance term needed *
745// in both the Penman-Monteith FAO reference ET procedure. *
746// and the aerodynamic, combination, and Priestley-Taylor *
747// ET calculation methods. *
748// Reference: http://www.fao.org/3/X0490E/x0490e06.htm *
749// F.L. Ogden, NOAA National Weather Service, 2020 *
750//############################################################*
751extern double calculate_aerodynamic_resistance
752(
753double wind_speed_measurement_height_m, // default =2.0 [m]
754double humidity_measurement_height_m, // default =2.0 [m],
755double zero_plane_displacement_height_m, // depends on surface roughness [m],
756double momentum_transfer_roughness_length_m, // [m],
757double heat_transfer_roughness_length_m, // [m],
758double wind_speed_m_per_s // [m s-1].
759)
760{
761// define local variables to ease computations:
762
763double ra,zm,zh,d,zom,zoh,k,uz;
764double von_karman_constant_squared=KV2; // this is dimensionless universal constant [-], K=0.41, squared.
765
766// input sanity checks.
767if(1.0e-06 >=wind_speed_measurement_height_m ) wind_speed_measurement_height_m=2.0; // standard measurement height
768if(1.0e-06 >=humidity_measurement_height_m ) humidity_measurement_height_m=2.0; // standard measurement height
769if(1.0e-06 >= momentum_transfer_roughness_length_m)
770 fprintf(stderr,"momentum_transfer_roughness_length_m is tiny in calculate_aerodynamic_resistance(). Should not be tiny.\n");
771if(1.0e-06 >= heat_transfer_roughness_length_m ) //warn. Should not be tiny.
772 fprintf(stderr,"heat_transfer_roughness_length_m is tiny in calculate_aerodynamic_resistance(). Should not be tiny.\n");
773
774// convert to smaller local variable names to keep equation readable
775zm=wind_speed_measurement_height_m;
776zh=humidity_measurement_height_m;
777d= zero_plane_displacement_height_m;
778zom=momentum_transfer_roughness_length_m;
779zoh=heat_transfer_roughness_length_m;
780uz=wind_speed_m_per_s;
781
782// here log is the natural logarithm.
783
784ra=log((zm-d)/zom)*log((zh-d)/zoh)/(von_karman_constant_squared*uz); // this is the equation for the aero. resist.
785 // from the FAO reference ET document.
786
787return(ra);
788}

845//############################################################/
846// subroutine to calculate the short-wave solar radiation /
847// reaching the land surface. Mostly from Hydrology textbook /
848// by R.L. Bras., with some updates on calculation of local /
849// hour angle, optical air mass, and atmospheric extinction. /
850// Inputs: air temp., cloud cover fraction, air turbidity, /
851// day of year, zulu time /
852// outputs solar radiation, solar elev. angle, solar azimuth. /
853// F.L. Ogden, 2009, NOAA National Weather Service, 2020 /
854//############################################################/
855
856void calculate_solar_radiation(solar_radiation_options *options, solar_radiation_parameters *params,
857 solar_radiation_forcing *forcing, solar_radiation_results *results)
858{
859double delta,r,equation_of_time_minutes,M,phi;
860double sinalpha,tau,alpha,cosalpha,azimuth;
861double Io,Ic,kshort,Ips;
862double b,fh1;
863
864// constants
865double solar_constant_W_per_sq_m;
866
867
868double solar_declination_angle_degrees;
869double solar_declination_angle_radians;
870double earth_sun_distance_ratio;
871double local_hour_angle_degrees;
872double local_hour_angle_radians;
873double antipodal_hour_angle_degrees;
874double antipodal_obs_longitude_degrees;
875double obs_x,obs_y,sun_x,sun_y;
876double zulu_time_h;
877double optical_air_mass;
878
879solar_constant_W_per_sq_m = 1361.6; // Dudock de Wit et al. 2017 GRL, approx. avg. value
880
881solar_declination_angle_degrees=23.45*M_PI/180.0*cos(2.0*M_PI/365.0*(172.0-forcing->day_of_year));
882solar_declination_angle_radians=solar_declination_angle_degrees*M_PI/180.0;
883
884earth_sun_distance_ratio=1.0+0.017*cos(2.0*M_PI/365*(186.0-forcing->day_of_year));
885
886// calculate the local hour angle using a unit circle centered on the observer, with Obs. at x=1, y=0.
887// Note: G=Greenwich, A=180E=180W==antipode, O=obs., S=sun.
888//--------------------------------------------------------------------------------------------------------------
889//
890// y ^
891// | Antipode
892// | /
893// --- / all angles measured as in trigonometry
894// _-- --_ ^
895// / \ \ If the sun were here, this would be a neg. LHA
896// | rot | |
897// | EARTH ^ | |
898// | + | |O ----------------------> x (all angles measured ccw from here)
899// | N.Pole | | Observer \ t
900// | | \ t
901// \ / |
902// --_ _-- |
903// / -_- \ This is a positive local hour angle
904// / \ (LHA), after
905// / \ local noon. This text prevents
906// v v / this from being
907// G S / a cont. comment
908// Greenwich vector pointing
909// Prime Meridian <TO SUN>
910//
911
912// this is the "equation of time" that accounts for the analemma effect
913M=2.0*M_PI*forcing->day_of_year/365.242; // mean anomaly of sun from wikipedia, works for leap years
914equation_of_time_minutes=-7.655*sin(M)+9.873*sin(2.0*M+3.588); // an approximation of the equation of time, minutes
915zulu_time_h=forcing->zulu_time_h-equation_of_time_minutes/1440.0; // adjust the position of the sun for analemma effect
916
917// here I use the antipode as the time origin, because that is where the sun is overhead at 00:00Z
918antipodal_hour_angle_degrees = zulu_time_h*15.0; // see note on above figure
919
920// here I convert longitude of the observer to the same coordinate system to eliminate the problem of +-180 deg. long.
921antipodal_obs_longitude_degrees = 180.0-params->longitude_degrees;
922
923// here I convert these angles to points on a unit circle using the above coordinate system to go to a purely
924// geometric representation. This helps deal with problems related to +-180 deg.
925obs_x=1.0;
926obs_y=0.0;
927
928sun_x=cos(M_PI/180.0*(360.0-(antipodal_hour_angle_degrees-antipodal_obs_longitude_degrees)));
929sun_y=sin(M_PI/180.0*(360.0-(antipodal_hour_angle_degrees-antipodal_obs_longitude_degrees)));
930
931local_hour_angle_degrees= acos(obs_x*sun_x+obs_y*sun_y)*180.0/M_PI; // acos(dot-product).
932
933if(sun_y>0.0) local_hour_angle_degrees *= -1.0; // Before local noon, the hour angle is defined as negative.
934
935local_hour_angle_radians=local_hour_angle_degrees*M_PI/180.0;
936
937// USE SIMPLER VARIABLE NAMES FOR THESE DENSE CALCULATIONS
938tau=local_hour_angle_radians; // local hour angle, radians
939delta=solar_declination_angle_radians; // solar declination angle, radians
940phi=params->latitude_degrees*M_PI/180.0; // latitude, radians
941
942// calculate solar elevation angle, sin(alpha)
943
944sinalpha=sin(delta)*sin(phi)+cos(delta)*cos(phi)*cos(tau);
945alpha=asin(sinalpha); // radians
946cosalpha=cos(alpha);
947
948// azimuth pointing to the sun (radians)
949azimuth=acos(sin(delta)/(cosalpha*cos(phi))-tan(alpha)*tan(phi));
950if(tau>0.0) // after local noon
951 {
952 azimuth=2.0*M_PI-azimuth;
953 }
954
955results->solar_elevation_angle_degrees=alpha*180.0/M_PI; // convert to degrees
956
957results->solar_azimuth_angle_degrees=azimuth*180.0/M_PI; // convert to degrees
958
959results->solar_local_hour_angle_degrees=tau*180.0/M_PI; // convert to degrees
960
961if(alpha>0.0) // the sun is over the horizon
962 {
963 // ==================== SHORTWAVE RADIATION CALCULATIONS =======================
964 r=earth_sun_distance_ratio; // the effect of non-circularity (eccentricity) of Earth's orbit
965 Io=solar_constant_W_per_sq_m/(r*r); // J/(m^2 s) at top of atmosphere, adjusted for orbital eccentricity
966
967 optical_air_mass=(1.002432*pow(sinalpha,2.0)+0.148386*sinalpha+0.0096467)/ // after Young 1994
968 (pow(sinalpha,3.0)+0.149864*pow(sinalpha,2.0)+0.0102963*sinalpha+0.000303978);
969
970 // the following comes from Ineichen and Perez, 2002
971 fh1=exp(-1.0*params->site_elevation_m/8000.0); // elev. in meters, effect of atmos. thickness on air mass
972 b=0.664+0.163/fh1;
973
974 // note, atm_turbidity is equal to Tlk in Ineichen and Perez, 2002.
975 Ic=b*Io*exp(-0.09*optical_air_mass*(forcing->atmospheric_turbidity_factor-1.0)); // clear sky radiation
976

983 }
984 else
985 {
986 // cloud base elevation not known.
987 kshort=0.65*forcing->cloud_cover_fraction*forcing->cloud_cover_fraction; // (TVA, 1972)
988 Ips=Ic*(1.0-kshort);
989 }

991 // all these results are calculated near the land surface, but above the canopy or snow pack.
992 results->solar_radiation_flux_W_per_sq_m= Ic; // no clouds. This is on a plane perpendicular to earth-sun line.
993 results->solar_radiation_horizontal_flux_W_per_sq_m=Ic*sinalpha; // this is on a horizontal plane
994 results->solar_radiation_cloudy_flux_W_per_sq_m=Ips; // Considers clouds, on a plane perpendicular to earth-sun line
995 results->solar_radiation_horizontal_cloudy_flux_W_per_sq_m=Ips*sinalpha; // on a horizontal plane tangent to earth
996
997// I comment this out because it is more appripriate in a vegetation effect routine
998// Ipsg=Kt*Ips;
999
1000// I comment this out because it is part of net radiation calculations done elsewhere
1001// Ieff=Ipsg*(1.0-Albedo); effective incoming shortwave radiation
1002 }
1003
1004return;
1005}
