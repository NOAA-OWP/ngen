624 inline void init_deferred_associations() {
625 for (int d = 0; d < deferredProviders.size(); ++d) {
626 std::shared_ptr<data_access::OptionalWrappedDataProvider> &deferredProvider = deferredProviders[d];
627 // Skip doing anything for any deferred provider that already has its backing provider set
628 if (deferredProvider->isWrappedProviderSet())
629 continue;
630
631 // TODO: improve this later; since defaults can be used, it is technically possible to grab something
632 // valid when something more appropriate would later be available
633 // Iterate through available data providers and set association once a sufficient one is found
634 std::map<std::string, std::shared_ptr<data_access::GenericDataProvider>>::iterator avail_it;
635 for (avail_it = availableData.begin(); avail_it != availableData.end(); avail_it++) {
636 // If this satisfies everything this deferred provider needs to provide, and thus can be/was set ...
637 if (deferredProvider->setWrappedProvider(avail_it->second.get())) {
638 break;
639 }
640 }
641
642 // If none of the available data providers could be used for this deferred one, throw exception
643 if (!deferredProvider->isWrappedProviderSet()) {
644 // TODO: this probably needs to be some kind of custom configuration exception
645 std::string msg = "Multi BMI formulation cannot be created from config: cannot find available data "
646 "provider to satisfy set of deferred provisions for nested module at index "
647 + std::to_string(deferredProviderModuleIndices[d]) + ": {";
648 // There must always be at least 1; get manually to help with formatting
649 msg += deferredProvider->get_available_variable_names()[0];
650 // And here make sure to start at 1 instead of 0
651 for (int i = 1; i < deferredProvider->get_available_variable_names().size(); ++i)
652 msg += ", " + deferredProvider->get_available_variable_names()[i];
653 msg += "}";
654 throw realization::ConfigurationException(msg);
655 }
656 }

674 template<class T>
675 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
676 std::shared_ptr<data_access::GenericDataProvider> wfp = std::make_shared<data_access::WrappedDataProvider>(this);
677 std::shared_ptr<T> mod = std::make_shared<T>(identifier, wfp, output);
678
679 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
680 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
681 "{{id}}", id);
682
683 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
684 mod->create_formulation(properties);
685
686 // Set this up for placing in the module_variable_maps member variable
687 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
688 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
689 for (const std::string &var_name : mod->get_bmi_input_variables()) {
690 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
691 (*var_aliases)[framework_alias] = var_name;
692 // If framework_name is not yet in collection from which we have available data sources ...
693 if (availableData.count(framework_alias) != 1) {
694 setup_nested_deferred_provider(var_name, framework_alias, mod, mod_index);
695 }
696 else {
697 mod->input_forcing_providers[var_name] = availableData[framework_alias];
698 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
699 }
700 }
701
702 // Also add the output variable aliases
703 for (const std::string &var_name : mod->get_bmi_output_variables()) {
704 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
705 (*var_aliases)[framework_alias] = var_name;
706 if (availableData.count(framework_alias) > 0) {
707 throw std::runtime_error(
708 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
709 " with output variable " + framework_alias +
710 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
711 " because a previous module is using this output variable name/alias.");
712 }
713 availableData[framework_alias] = mod;
714 }
715 module_variable_maps[mod_index] = var_aliases;
716 return mod;
717 }

746 template<class T>
747 void setup_nested_deferred_provider(const std::string &bmi_input_var_name,
748 const std::string &framework_output_name,
749 std::shared_ptr<T> mod,
750 int mod_index) {
751 // TODO: probably don't actually need bmi_input_var_name, and just can deal with framework_output_name
752 // Create deferred, optional provider for providing this
753 // Only include BMI variable name, as that's what'll be visible when associating to backing provider
754 // It's "deferred" in that we'll set the backing later.
755 // It's "optional" in that it waits to use backing provider, using the default some number of times
756 std::shared_ptr<data_access::OptionalWrappedDataProvider> provider;
757 // TODO: make sure only alias is needed
758 auto defs_it = default_output_values.find(framework_output_name);
759 if (defs_it != default_output_values.end()) {
760 // TODO: consider also reading wait count from config
761 provider = std::make_shared<data_access::OptionalWrappedDataProvider>(framework_output_name, defs_it->second, 1);
762 }
763 else {
764 provider = std::make_shared<data_access::OptionalWrappedDataProvider>(framework_output_name);
765 }
766
767 // Add deferred to collection and module index to collection
768 deferredProviders.push_back(provider);
769 deferredProviderModuleIndices.push_back(mod_index);
770 // Assign as provider within module
771 // TODO: per TODO at top, probably can replace bmi_input_var_name here with framework_output_name
772 mod->input_forcing_providers[bmi_input_var_name] = provider;
773 }
