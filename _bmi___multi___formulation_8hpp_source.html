594 inline void init_deferred_associations() {
595 for (int d = 0; d < deferredProviders.size(); ++d) {
596 std::shared_ptr<data_access::OptionalWrappedDataProvider> &deferredProvider = deferredProviders[d];
597 // Skip doing anything for any deferred provider that already has its backing provider set
598 if (deferredProvider->isWrappedProviderSet())
599 continue;
600
601 // TODO: improve this later; since defaults can be used, it is technically possible to grab something
602 // valid when something more appropriate would later be available
603 // Iterate through available data providers and set association once a sufficient one is found
604 std::map<std::string, std::shared_ptr<data_access::GenericDataProvider>>::iterator avail_it;
605 for (avail_it = availableData.begin(); avail_it != availableData.end(); avail_it++) {
606 // If this satisfies everything this deferred provider needs to provide, and thus can be/was set ...
607 if (deferredProvider->setWrappedProvider(avail_it->second.get())) {
608 break;
609 }
610 }
611
612 // If none of the available data providers could be used for this deferred one, throw exception
613 if (!deferredProvider->isWrappedProviderSet()) {
614 // TODO: this probably needs to be some kind of custom configuration exception
615 std::string msg = "Multi BMI formulation cannot be created from config: cannot find available data "
616 "provider to satisfy set of deferred provisions for nested module at index "
617 + std::to_string(deferredProviderModuleIndices[d]) + ": {";
618 // There must always be at least 1; get manually to help with formatting
619 msg += deferredProvider->get_available_variable_names()[0];
620 // And here make sure to start at 1 instead of 0
621 for (int i = 1; i < deferredProvider->get_available_variable_names().size(); ++i)
622 msg += ", " + deferredProvider->get_available_variable_names()[i];
623 msg += "}";
624 throw realization::ConfigurationException(msg);
625 }
626 }

644 template<class T>
645 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
646 std::shared_ptr<data_access::GenericDataProvider> wfp = std::make_shared<data_access::WrappedDataProvider>(this);
647 std::shared_ptr<T> mod = std::make_shared<T>(identifier, wfp, output);
648
649 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
650 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
651 "{{id}}", id);
652
653 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
654 mod->create_formulation(properties);
655
656 // Set this up for placing in the module_variable_maps member variable
657 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
658 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
659 for (const std::string &var_name : mod->get_bmi_input_variables()) {
660 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
661 (*var_aliases)[framework_alias] = var_name;
662 // If framework_name is not yet in collection from which we have available data sources ...
663 if (availableData.count(framework_alias) != 1) {
664 setup_nested_deferred_provider(var_name, framework_alias, mod, mod_index);
665 }
666 else {
667 mod->input_forcing_providers[var_name] = availableData[framework_alias];
668 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
669 }
670 }
671
672 // Also add the output variable aliases
673 for (const std::string &var_name : mod->get_bmi_output_variables()) {
674 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
675 (*var_aliases)[framework_alias] = var_name;
676 if (availableData.count(framework_alias) > 0) {
677 throw std::runtime_error(
678 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
679 " with output variable " + framework_alias +
680 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
681 " because a previous module is using this output variable name/alias.");
682 }
683 availableData[framework_alias] = mod;
684 }
685 module_variable_maps[mod_index] = var_aliases;
686 return mod;
687 }

717 void setup_nested_deferred_provider(const std::string &bmi_input_var_name,

739 deferredProviderModuleIndices.push_back(mod_index);
