120 size_t determine_et_formulation_index() {
121 std::shared_ptr<data_access::GenericDataProvider> et_provider = nullptr;
122 if (availableData.find(NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != availableData.end()) {
123 et_provider = availableData[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP];
124 }
125 else if (availableData.find(CSDMS_STD_NAME_POTENTIAL_ET) != availableData.end()) {
126 et_provider = availableData[CSDMS_STD_NAME_POTENTIAL_ET];
127 }
128 // The pointers in availableData and modules should be the same, based on create_multi_formulation()
129 // So, if an et_provider was found from availableData, we can compare to determine the right index.
130 if (et_provider != nullptr) {
131 for (size_t i = 0; i < modules.size(); ++i) {
132 if (et_provider == modules[i]) {
133 return i;
134 }
135 }
136 }
137 // Otherwise, check the modules directly for the standard ET output variable names.
138 for (size_t i = 0; i < modules.size(); ++i) {
139 std::vector<std::string> values = modules[i]->get_avaliable_variable_names();
140 if (std::find(values.begin(), values.end(), NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP) != values.end()) {
141 return i;
142 }
143 if (std::find(values.begin(), values.end(), CSDMS_STD_NAME_POTENTIAL_ET) != values.end()) {
144 return i;
145 }
146 }
147 return DEFAULT_ET_FORMULATION_INDEX;

643 inline void init_deferred_associations() {
644 for (int d = 0; d < deferredProviders.size(); ++d) {
645 std::shared_ptr<data_access::OptionalWrappedProvider> &deferredProvider = deferredProviders[d];
646 // Skip doing anything for any deferred provider that already has its backing provider set
647 if (deferredProvider->isWrappedProviderSet())
648 continue;
649
650 // TODO: improve this later; since defaults can be used, it is technically possible to grab something
651 // valid when something more appropriate would later be available
652 // Iterate through available data providers and set association once a sufficient one is found
653 std::map<std::string, std::shared_ptr<data_access::GenericDataProvider>>::iterator avail_it;
654 for (avail_it = availableData.begin(); avail_it != availableData.end(); avail_it++) {
655 // If this satisfies everything this deferred provider needs to provide, and thus can be/was set ...
656 if (deferredProvider->setWrappedProvider(avail_it->second.get())) {
657 break;
658 }
659 }
660
661 // If none of the available data providers could be used for this deferred one, throw exception
662 if (!deferredProvider->isWrappedProviderSet()) {
663 // TODO: this probably needs to be some kind of custom configuration exception
664 std::string msg = "Multi BMI formulation cannot be created from config: cannot find available data "
665 "provider to satisfy set of deferred provisions for nested module at index "
666 + std::to_string(deferredProviderModuleIndices[d]) + ": {";
667 // There must always be at least 1; get manually to help with formatting
668 msg += deferredProvider->get_avaliable_variable_names()[0];
669 // And here make sure to start at 1 instead of 0
670 for (int i = 1; i < deferredProvider->get_avaliable_variable_names().size(); ++i)
671 msg += ", " + deferredProvider->get_avaliable_variable_names()[i];
672 msg += "}";
673 throw realization::ConfigurationException(msg);
674 }
675 }

693 template<class T>
694 std::shared_ptr<T> init_nested_module(int mod_index, std::string identifier, geojson::PropertyMap properties) {
695 std::shared_ptr<data_access::GenericDataProvider> wfp = std::make_shared<data_access::WrappedForcingProvider>(this);
696 std::shared_ptr<T> mod = std::make_shared<T>(identifier, wfp, output);
697
698 // Since this is a nested formulation, support usage of the '{{id}}' syntax for init config file paths.
699 Catchment_Formulation::config_pattern_substitution(properties, BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG,
700 "{{id}}", id);
701
702 // Call create_formulation to perform the rest of the typical initialization steps for the formulation.
703 mod->create_formulation(properties);
704
705 // Set this up for placing in the module_variable_maps member variable
706 std::shared_ptr<std::map<std::string, std::string>> var_aliases;
707 var_aliases = std::make_shared<std::map<std::string, std::string>>(std::map<std::string, std::string>());
708 for (const std::string &var_name : mod->get_bmi_input_variables()) {
709 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
710 (*var_aliases)[framework_alias] = var_name;
711 // If framework_name is not yet in collection from which we have available data sources ...
712 if (availableData.count(framework_alias) != 1) {
713 setup_nested_deferred_provider(var_name, framework_alias, mod, mod_index);
714 }
715 else {
716 mod->input_forcing_providers[var_name] = availableData[framework_alias];
717 mod->input_forcing_providers[framework_alias] = availableData[framework_alias];
718 }
719 }
720
721 // Also add the output variable aliases
722 for (const std::string &var_name : mod->get_bmi_output_variables()) {
723 std::string framework_alias = mod->get_config_mapped_variable_name(var_name);
724 (*var_aliases)[framework_alias] = var_name;
725 if (availableData.count(framework_alias) > 0) {
726 throw std::runtime_error(
727 "Multi BMI cannot be created with module " + mod->get_model_type_name() +
728 " with output variable " + framework_alias +
729 (var_name == framework_alias ? "" : " (an alias of BMI variable " + var_name + ")") +
730 " because a previous module is using this output variable name/alias.");
731 }
732 availableData[framework_alias] = mod;
733 }
734 module_variable_maps[mod_index] = var_aliases;
735 return mod;
736 }

765 template<class T>
766 void setup_nested_deferred_provider(const std::string &bmi_input_var_name,
767 const std::string &framework_output_name,
768 std::shared_ptr<T> mod,
769 int mod_index) {
770 // TODO: probably don't actually need bmi_input_var_name, and just can deal with framework_output_name
771 // Create deferred, optional provider for providing this
772 // Only include BMI variable name, as that's what'll be visible when associating to backing provider
773 // It's "deferred" in that we'll set the backing later.
774 // It's "optional" in that it waits to use backing provider, using the default some number of times
775 std::shared_ptr<data_access::OptionalWrappedProvider> provider;
776 // TODO: make sure only alias is needed
777 auto defs_it = default_output_values.find(framework_output_name);
778 if (defs_it != default_output_values.end()) {
779 // TODO: consider also reading wait count from config
780 provider = std::make_shared<data_access::OptionalWrappedProvider>(framework_output_name, defs_it->second, 1);
781 }
782 else {
783 provider = std::make_shared<data_access::OptionalWrappedProvider>(framework_output_name);
784 }
785
786 // Add deferred to collection and module index to collection
787 deferredProviders.push_back(provider);
788 deferredProviderModuleIndices.push_back(mod_index);
789 // Assign as provider within module
790 // TODO: per TODO at top, probably can replace bmi_input_var_name here with framework_output_name
791 mod->input_forcing_providers[bmi_input_var_name] = provider;
792 }
