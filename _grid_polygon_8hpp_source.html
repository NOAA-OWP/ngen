18polygon_grid_type create_ns_cells(point_t max_corner, point_t min_corner, double x_width, double y_width)
19{
20 // get the values for the first grid cell
21 double y1 = max_corner.y();
22 double y2 = max_corner.y() - y_width;
23
24
25 // calculate the the size of the grid
26 int x_size = static_cast<int>(std::ceil(max_corner.x() - min_corner.x()));
27 int y_size = static_cast<int>(std::ceil(max_corner.y() - min_corner.y()));
28
29 // create the boost multi_array
30 polygon_grid_type cells{boost::extents[x_size][y_size]};
31
32 // set the inital row cell index position
33 int r = 0;
34
35 // move through y dimension
36 while (y1 > min_corner.y())
37 {
38 // move through x dimension
39 double x1 = min_corner.x();
40 double x2 = min_corner.x() + x_width;
41 int c = 0;
42
43 while (x1 < max_corner.x())
44 {
45 polygon_type p;
46 //assign points to polygon
47 bg::append(p, point_t(x1,y1));
48 bg::append(p, point_t(x2,y1));
49 bg::append(p, point_t(x2,y2));
50 bg::append(p, point_t(x1,y2));
51 bg::append(p, point_t(x1,y1));
52
53 // assign polygon to grid and move column index
54 cells[r][c++] = p;
55
56 // advance x
57 x1 = x2;
58 x2 += x_width;
59 }
60
61 // advance y
62 y1 = y2;
63 y2 -= y_width;
64 ++r;
65 }
