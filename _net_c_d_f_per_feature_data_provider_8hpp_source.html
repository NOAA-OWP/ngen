237 time_unit = TIME_NANOSECONDS;

388 double get_value(const CatchmentAggrDataSelector& selector, ReSampleMethod m) override
389 {
390 auto init_time = selector.get_init_time();
391 auto stop_time = init_time + selector.get_duration_secs(); // scope hiding! BAD JUJU!
392
393 size_t idx1 = get_ts_index_for_time(init_time);
394 size_t idx2;
395 try {
396 idx2 = get_ts_index_for_time(stop_time-1); // Don't include next timestep when duration % timestep = 0
397 }
398 catch(const std::out_of_range &e){
399 idx2 = get_ts_index_for_time(this->stop_time-1); //to the edge
400 }
401
402 auto stride = idx2 - idx1;
403
404 std::vector<std::size_t> start, count;
405
406 auto cat_pos = id_pos[selector.get_id()];
407
408
409
410 double t1 = time_vals[idx1];
411 double t2 = time_vals[idx2];
412
413 double rvalue = 0.0;
414
415 auto ncvar = get_ncvar(selector.get_variable_name());
416
417 std::string native_units = get_ncvar_units(selector.get_variable_name());
418
419 auto read_len = idx2 - idx1 + 1;
420
421 std::vector<double> raw_values;
422 raw_values.resize(read_len);
423
424 //TODO: Currently assuming a whole variable cache slice across all catchments for a single timestep...but some stuff here to support otherwise.
425 size_t cache_slices_t_n = read_len / cache_slice_t_size; // Integer division!
426 // For reference: https://stackoverflow.com/a/72030286
427 for( size_t i = 0; i < cache_slices_t_n; i++ ) {
428 std::shared_ptr<std::vector<double>> cached;
429 int cache_t_idx = (idx1 - (idx1 % cache_slice_t_size) + i);
430 std::string key = ncvar.getName() + "|" + std::to_string(cache_t_idx);
431 if(value_cache.contains(key)){
432 cached = value_cache.get(key).get();
433 } else {
434 cached = std::make_shared<std::vector<double>>(cache_slice_c_size * cache_slice_t_size);
435 start.clear();
436 start.push_back(0); // only always 0 when cache_slice_c_size = numids!
437 start.push_back(cache_t_idx * cache_slice_t_size);
438 count.clear();
439 count.push_back(cache_slice_c_size);
440 count.push_back(cache_slice_t_size); // Must be 1 for now!...probably...
441 ncvar.getVar(start,count,&(*cached)[0]);
442 value_cache.insert(key, cached);
443 }
444 for( size_t j = 0; j < cache_slice_t_size; j++){
445 raw_values[i+j] = cached->at((j*cache_slice_t_size) + cat_pos);
446 }
447 }
448
449
450 rvalue = 0.0;
451
452 double a , b = 0.0;
453
454 a = 1.0 - ( (t1 - init_time) / time_stride );
455 rvalue += (a * raw_values[0]);
456
457 for( size_t i = 1; i < raw_values.size() -1; ++i )
458 {
459 rvalue += raw_values[i];
460 }
461
462 if ( raw_values.size() > 1) // likewise the last data value may not be fully in the window
463 {
464 b = (stop_time - t2) / time_stride;
465 rvalue += (b * raw_values.back() );
466 }
467
468 // account for the resampling methods
469 switch(m)
470 {
471 case SUM: // we allready have the sum so do nothing
472 ;
473 break;
474
475 case MEAN:
476 {
477 // This is getting a length weighted mean
478 // the data values where allready scaled for where there was only partial use of a data value
479 // so we just need to do a final scale to account for the differnce between time_stride and duration_s
480
481 double scale_factor = (selector.get_duration_secs() > time_stride ) ? (time_stride / selector.get_duration_secs()) : (1.0 / (a + b));
482 rvalue *= scale_factor;
483 }
484 break;
485
486 default:
487 ;
488 }
489
490 try
491 {
492 return UnitsHelper::get_converted_value(native_units, rvalue, selector.get_output_units());
493 }
494 catch (const std::runtime_error& e)
495 {
496 #ifndef UDUNITS_QUIET
497 std::cerr<<"WARN: Unit conversion unsuccessful - Returning unconverted value! (\""<<e.what()<<"\")"<<std::endl;
498 #endif
499 return rvalue;
500 }
501
502 return rvalue;
503 }
