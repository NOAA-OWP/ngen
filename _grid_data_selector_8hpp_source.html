110 GridDataSelector(
111 SelectorConfig config,
112 const GridSpecification& grid,
113 boost::span<const geojson::coordinate_t> points
114 )
115 : config_(std::move(config))
116 {
117 // Using a std::set since points may be close enough that they are within
118 // the same grid cell. This ensures that each cell is uniquely indexed.
119 std::set<Cell> cells;
120 for (const auto& point : points) {
121 cells.emplace(Cell{
122 /*x=*/position_(point.get<0>(), grid.extent.xmin(), grid.extent.xmax(), grid.columns),
123 /*y=*/position_(point.get<1>(), grid.extent.ymin(), grid.extent.ymax(), grid.rows),
124 /*z=*/0UL,
125 /*value=*/NAN
126 });
127 }
128
129 cells_.assign(cells.begin(), cells.end());

142 GridDataSelector(
143 SelectorConfig config,
144 const GridSpecification& grid,
145 const geojson::polygon_t& polygon
146 )
147 : config_(std::move(config))
148 {
149 const auto xmin = grid.extent.xmin();
150 const auto xmax = grid.extent.xmax();
151 const auto ymin = grid.extent.ymin();
152 const auto ymax = grid.extent.ymax();
153 const auto ydiff = static_cast<double>(grid.rows) / (ymax - ymin);
154 const auto xdiff = static_cast<double>(grid.columns) / (xmax - xmin);
155
156 const auto bbox = BoundingBox{ boost::geometry::return_envelope<box_t>(polygon) };
157 for (double row = bbox.ymin(); row < bbox.ymax() - ydiff; row += ydiff) {
158 for (double col = bbox.xmin(); col < bbox.xmax() - xdiff; row += xdiff) {
159 const box_t cell_box = {
160 /*min_corner=*/{ col, row },
161 /*max_corner=*/{ col + xdiff, row + ydiff }
162 };
163
164 if (boost::geometry::intersects(cell_box, polygon)) {
165 auto x = position_(col, xmin, xmax, grid.columns);
166 auto y = position_(row, ymin, ymax, grid.rows);
167 cells_.emplace_back(Cell{x, y, /*z=*/0UL, /*value=*/NAN});
168 }
169 }
170 }
171 }

180 GridDataSelector(
181 SelectorConfig config,
182 const GridSpecification& grid,
183 const BoundingBox& extent
184 )
185 : config_(std::move(config))
186 {
187 const auto col_min = position_(extent.xmin(), grid.extent.xmin(), grid.extent.xmax(), grid.columns);
188 const auto col_max = position_(extent.xmax(), grid.extent.xmin(), grid.extent.xmax(), grid.columns);
189 const auto row_min = position_(extent.ymin(), grid.extent.ymin(), grid.extent.ymax(), grid.rows);
190 const auto row_max = position_(extent.ymax(), grid.extent.ymin(), grid.extent.ymax(), grid.rows);
191 const auto ncells = (row_max - row_min) * (col_max - col_min);
192
193 cells_.reserve(ncells);
194 for (auto row = row_min; row < row_max; row++) {
195 for (auto col = col_min; col < col_max; col++) {
196 cells_.emplace_back(Cell{/*x=*/col, /*y=*/row, /*z=*/0UL, /*value=*/NAN});
197 }
198 }
