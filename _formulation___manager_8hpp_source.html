397 forcing_params get_forcing_params(const geojson::PropertyMap &forcing_prop_map, std::string identifier, simulation_time_params &simulation_time_config) {
398 std::string path = "";
399 if(forcing_prop_map.count("path") != 0){
400 path = forcing_prop_map.at("path").as_string();
401 }
402 std::string provider;
403 if(forcing_prop_map.count("provider") != 0){
404 provider = forcing_prop_map.at("provider").as_string();
405 }
406 if (forcing_prop_map.count("file_pattern") == 0) {
407 return forcing_params(
408 path,
409 provider,
410 simulation_time_config.start_time,
411 simulation_time_config.end_time
412 );
413 }
414
415 if (path.empty()) {
416 throw std::runtime_error("Error with NGEN config - 'path' in forcing params must be set to a "
417 "non-empty parent directory path when 'file_pattern' is used.");
418 }
419
420 // Since we are given a pattern, we need to identify the directory and pull out anything that matches the pattern
421 if (path.compare(path.size() - 1, 1, "/") != 0) {
422 path += "/";
423 }
424
425 std::string filepattern = forcing_prop_map.at("file_pattern").as_string();
426
427 int id_index = filepattern.find("{{id}}");
428
429 // If an index for '{{id}}' was found, we can count on that being where the id for this realization can be found.
430 // For instance, if we have a pattern of '.*{{id}}_14_15.csv' and this is named 'cat-87',
431 // this will match on 'stuff_example_cat-87_14_15.csv'
432 if (id_index != std::string::npos) {
433 filepattern = filepattern.replace(id_index, sizeof("{{id}}") - 1, identifier);
434 }
435
436 // Create a regular expression used to identify proper file names
437 std::regex pattern(filepattern);
438
439 // A stream providing the functions necessary for evaluating a directory:
440 // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
441 DIR *directory = nullptr;
442
443 // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
444 struct dirent *entry = nullptr;
445
446 // Attempt to open the directory for evaluation
447 directory = opendir(path.c_str());
448 // Allow for a few retries in certain failure situations
449 size_t attemptCount = 0;
450 std::string errMsg;
451 while (directory == nullptr && attemptCount++ < 5) {
452 // For several error codes, we should break immediately and not retry
453 if (errno == ENOENT) {
454 errMsg = "No such file or directory.";
455 break;
456 }
457 if (errno == ENXIO) {
458 errMsg = "No such device or address.";
459 break;
460 }
461 if (errno == EACCES) {
462 errMsg = "Permission denied.";
463 break;
464 }
465 if (errno == EPERM) {
466 errMsg = "Operation not permitted.";
467 break;
468 }
469 if (errno == ENOTDIR) {
470 errMsg = "File at provided path is not a directory.";
471 break;
472 }
473 if (errno == EMFILE) {
474 errMsg = "The current process has too many open files.";
475 break;
476 }
477 if (errno == ENFILE) {
478 errMsg = "The system has too many open files.";
479 break;
480 }
481 sleep(2);
482 directory = opendir(path.c_str());
483 errMsg = "Received system error number " + std::to_string(errno);
484 }
485
486 // If the directory could be found and opened, we can go ahead and iterate
487 if (directory != nullptr) {
488 bool match;
489 while ((entry = readdir(directory))) {
490 match = std::regex_match(entry->d_name, pattern);
491 if( match ) {
492 // If the entry is a regular file or symlink AND the name matches the pattern,
493 // we can consider this ready to be interpretted as valid forcing data (even if it isn't)
494 #ifdef _DIRENT_HAVE_D_TYPE
495 if ( entry->d_type == DT_REG or entry->d_type == DT_LNK ) {
496 return forcing_params(
497 path + entry->d_name,
498 provider,
499 simulation_time_config.start_time,
500 simulation_time_config.end_time
501 );
502 }
503 else if ( entry->d_type == DT_UNKNOWN )
504 #endif
505 {
506 //dirent is not guaranteed to provide propoer file type identification in d_type
507 //so if a system returns unknown or it isn't supported, need to use stat to determine if it is a file
508 struct stat st;
509 if( stat((path+entry->d_name).c_str(), &st) != 0) {
510 throw std::runtime_error("Could not stat file "+path+entry->d_name);
511 }
512 if( S_ISREG(st.st_mode) ) {
513 //Sinde we used stat and not lstat, we get the result of the target of links as well
514 //so this covers both cases we are interested in.
515 return forcing_params(
516 path + entry->d_name,
517 provider,
518 simulation_time_config.start_time,
519 simulation_time_config.end_time
520 );
521 }
522 throw std::runtime_error("Forcing data is path "+path+entry->d_name+" is not a file");
523 }
524 } //no match found, try next entry
525 } // end while iter dir
526 } //end if directory
527 else {
528 // The directory wasn't found or otherwise couldn't be opened; forcing data cannot be retrieved
529 throw std::runtime_error("Error opening forcing data dir '" + path + "' after " + std::to_string(attemptCount) + " attempts: " + errMsg);
530 }
531
532 closedir(directory);
533
534 throw std::runtime_error("Forcing data could not be found for '" + identifier + "'");
535 }

544 void parse_external_model_params(boost::property_tree::ptree& model_params, const geojson::Feature catchment_feature) {
545 boost::property_tree::ptree attr {};
546 for (decltype(auto) param : model_params) {
547 if (param.second.count("source") == 0) {
548 attr.put_child(param.first, param.second);
549 continue;
550 }
551
552 decltype(auto) param_source = param.second.get_child("source");
553 decltype(auto) param_source_name = param_source.get_value<std::string>();
554 if (param_source_name != "hydrofabric") {
555 // temporary until the logic for alternative sources is designed
556 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
557 }
558
559 decltype(auto) param_name = param.second.find("from") == param.second.not_found()
560 ? param.first
561 : param.second.get_child("from").get_value<std::string>();
562
563 if (catchment_feature->has_property(param_name)) {
564 auto catchment_attribute = catchment_feature->get_property(param_name);
565 switch (catchment_attribute.get_type()) {
566 case geojson::PropertyType::Natural:
567 attr.put(param.first, catchment_attribute.as_natural_number());
568 break;
569 case geojson::PropertyType::Boolean:
570 attr.put(param.first, catchment_attribute.as_boolean());
571 break;
572 case geojson::PropertyType::Real:
573 attr.put(param.first, catchment_attribute.as_real_number());
574 break;
575 case geojson::PropertyType::String:
576 attr.put(param.first, catchment_attribute.as_string());
577 break;
578
579 case geojson::PropertyType::List:
580 case geojson::PropertyType::Object:
581 default:
582 std::cerr << "WARNING: property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
583 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
584 break;
585 }
586 } else {
587 std::cerr << "WARNING Failed to parse external parameter: catchment `"
588 << catchment_feature->get_id()
589 << "` does not contain the property `"
590 << param_name << "`\n";
591 }
592 }
593
594 model_params.swap(attr);
595 }

604 void parse_external_model_params(geojson::PropertyMap& model_params, const geojson::Feature catchment_feature) {
605 geojson::PropertyMap attr {};
606 for (decltype(auto) param : model_params) {
607 // Check for type to short-circuit. If param.second is not an object, `.has_key()` will throw
608 if (param.second.get_type() != geojson::PropertyType::Object || !param.second.has_key("source")) {
609 attr.emplace(param.first, param.second);
610 continue;
611 }
612
613 decltype(auto) param_source = param.second.at("source");
614 decltype(auto) param_source_name = param_source.as_string();
615 if (param_source_name != "hydrofabric") {
616 // TODO: temporary until the logic for alternative sources is designed
617 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
618 }
619
620 // Property name in the feature properties is either
621 // the value of key "from", or has the same name as
622 // the expected model parameter key
623 decltype(auto) param_name = param.second.has_key("from")
624 ? param.second.at("from").as_string()
625 : param.first;
626
627 if (catchment_feature->has_property(param_name)) {
628 auto catchment_attribute = catchment_feature->get_property(param_name);
629
630 // Use param.first in the `.emplace` calls instead of param_name since
631 // the expected name is given by the key of the model_params values.
632 switch (catchment_attribute.get_type()) {
633 case geojson::PropertyType::Natural:
634 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_natural_number()));
635 break;
636 case geojson::PropertyType::Boolean:
637 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_boolean()));
638 break;
639 case geojson::PropertyType::Real:
640 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_real_number()));
641 break;
642 case geojson::PropertyType::String:
643 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_string()));
644 break;
645
646 case geojson::PropertyType::List:
647 case geojson::PropertyType::Object:
648 default:
649 // TODO: Should list/object values be passed to model parameters?
650 // Typically, feature properties *should* be scalars.
651 std::cerr << "WARNING: property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
652 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
653 break;
654 }
655 }
656 }
657
658 model_params.swap(attr);
659 }
