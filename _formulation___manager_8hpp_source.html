310  forcing_params get_global_forcing_params(std::string identifier, simulation_time_params &simulation_time_config) {
311  std::string path = this->global_forcing.at("path").as_string();
312 
313  if (this->global_forcing.count("file_pattern") == 0) {
314  return forcing_params(
315  path,
316  simulation_time_config.start_time,
317  simulation_time_config.end_time
318  );
319  }
320 
321  // Since we are given a pattern, we need to identify the directory and pull out anything that matches the pattern
322  if (path.compare(path.size() - 1, 1, "/") != 0) {
323  path += "/";
324  }
325 
326  std::string filepattern = this->global_forcing.at("file_pattern").as_string();
327 
328  int id_index = filepattern.find("{{id}}");
329 
330  // If an index for '{{id}}' was found, we can count on that being where the id for this realization can be found.
331  // For instance, if we have a pattern of '.*{{id}}_14_15.csv' and this is named 'cat-87',
332  // this will match on 'stuff_example_cat-87_14_15.csv'
333  if (id_index != std::string::npos) {
334  filepattern = filepattern.replace(id_index, sizeof("{{id}}") - 1, identifier);
335  }
336 
337  // Create a regular expression used to identify proper file names
338  std::regex pattern(filepattern);
339 
340  // A stream providing the functions necessary for evaluating a directory:
341  // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
342  DIR *directory = nullptr;
343 
344  // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
345  struct dirent *entry = nullptr;
346 
347  // Attempt to open the directory for evaluation
348  directory = opendir(path.c_str());
349 
350  // If the directory could be found, we can go ahead and iterate
351  if (directory != nullptr) {
352  while ((entry = readdir(directory))) {
353  // If the entry is a regular file or symlink AND the name matches the pattern,
354  // we can consider this ready to be interpretted as valid forcing data (even if it isn't)
355  if ((entry->d_type == DT_REG or entry->d_type == DT_LNK) and std::regex_match(entry->d_name, pattern)) {
356  return forcing_params(
357  path + entry->d_name,
358  simulation_time_config.start_time,
359  simulation_time_config.end_time
360  );
361  }
362  }
363  }
364  else {
365  // The directory wasn't found; forcing data cannot be retrieved
366  throw std::runtime_error("No directory for forcing data was found at: " + path);
367  }
368 
369  closedir(directory);
370 
371  throw std::runtime_error("Forcing data could not be found for '" + identifier + "'");
372  }
