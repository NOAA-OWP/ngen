432 std::shared_ptr<Catchment_Formulation> construct_missing_formulation(geojson::Feature& feature, utils::StreamHandler output_stream, simulation_time_params &simulation_time_config){
433 const std::string identifier = feature->get_id();
434
435 std::string formulation_type_key = get_formulation_key(global_formulation_tree.get_child("formulations.."));
436
437 forcing_params forcing_config = this->get_forcing_params(this->global_forcing, identifier, simulation_time_config);
438
439 std::shared_ptr<Catchment_Formulation> missing_formulation = construct_formulation(formulation_type_key, identifier, forcing_config, output_stream);
440 // Need to work with a copy, since it is altered in-place
441 geojson::PropertyMap global_properties_copy = global_formulation_parameters;
442 Catchment_Formulation::config_pattern_substitution(global_properties_copy,
443 BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG, "{{id}}",
444 identifier);
445
446 // parse any external model parameters in this config
447
448 // handle single-bmi
449 if (global_properties_copy.count("model_params") > 0) {
450 decltype(auto) model_params = global_properties_copy.at("model_params");
451 geojson::PropertyMap model_params_copy = model_params.get_values();
452 parse_external_model_params(model_params_copy, feature);
453 global_properties_copy.at("model_params") = geojson::JSONProperty("model_params", model_params_copy);
454 }
455
456 // handle multi-bmi
457 // FIXME: this seems inefficient -- is there a better way?
458 if (global_properties_copy.count("modules") > 0) {
459 decltype(auto) nested_modules = global_properties_copy.at("modules").as_list();
460 for (auto& bmi_module : nested_modules) {
461 geojson::PropertyMap module_def = bmi_module.get_values();
462 geojson::PropertyMap module_params = module_def.at("params").get_values();
463 if (module_params.count("model_params") > 0) {
464 decltype(auto) model_params = module_params.at("model_params");
465 geojson::PropertyMap model_params_copy = model_params.get_values();
466 parse_external_model_params(model_params_copy, feature);
467 module_params.at("model_params") = geojson::JSONProperty("model_params", model_params_copy);
468 }
469 module_def.at("params") = geojson::JSONProperty("params", module_params);
470 bmi_module = geojson::JSONProperty("", module_def);
471 }
472 global_properties_copy.at("modules") = geojson::JSONProperty("", nested_modules);
473 }
474
475 missing_formulation->create_formulation(global_properties_copy);
476 return missing_formulation;
477 }

479 forcing_params get_forcing_params(geojson::PropertyMap &forcing_prop_map, std::string identifier, simulation_time_params &simulation_time_config) {
480 std::string path;
481 if(forcing_prop_map.count("path") != 0){
482 path = forcing_prop_map.at("path").as_string();
483 }
484 std::string provider;
485 if(forcing_prop_map.count("provider") != 0){
486 provider = forcing_prop_map.at("provider").as_string();
487 }
488 if (forcing_prop_map.count("file_pattern") == 0) {
489 return forcing_params(
490 path,
491 provider,
492 simulation_time_config.start_time,
493 simulation_time_config.end_time
494 );
495 }
496
497 if (path.empty()) {
498 throw std::runtime_error("Error with NGEN config - 'path' in forcing params must be set to a "
499 "non-empty parent directory path when 'file_pattern' is used.");
500 }
501
502 // Since we are given a pattern, we need to identify the directory and pull out anything that matches the pattern
503 if (path.compare(path.size() - 1, 1, "/") != 0) {
504 path += "/";
505 }
506
507 std::string filepattern = forcing_prop_map.at("file_pattern").as_string();
508
509 int id_index = filepattern.find("{{id}}");
510
511 // If an index for '{{id}}' was found, we can count on that being where the id for this realization can be found.
512 // For instance, if we have a pattern of '.*{{id}}_14_15.csv' and this is named 'cat-87',
513 // this will match on 'stuff_example_cat-87_14_15.csv'
514 if (id_index != std::string::npos) {
515 filepattern = filepattern.replace(id_index, sizeof("{{id}}") - 1, identifier);
516 }
517
518 // Create a regular expression used to identify proper file names
519 std::regex pattern(filepattern);
520
521 // A stream providing the functions necessary for evaluating a directory:
522 // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
523 DIR *directory = nullptr;
524
525 // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
526 struct dirent *entry = nullptr;
527
528 // Attempt to open the directory for evaluation
529 directory = opendir(path.c_str());
530 // Allow for a few retries in certain failure situations
531 size_t attemptCount = 0;
532 std::string errMsg;
533 while (directory == nullptr && attemptCount++ < 5) {
534 // For several error codes, we should break immediately and not retry
535 if (errno == ENOENT) {
536 errMsg = "No such file or directory.";
537 break;
538 }
539 if (errno == ENXIO) {
540 errMsg = "No such device or address.";
541 break;
542 }
543 if (errno == EACCES) {
544 errMsg = "Permission denied.";
545 break;
546 }
547 if (errno == EPERM) {
548 errMsg = "Operation not permitted.";
549 break;
550 }
551 if (errno == ENOTDIR) {
552 errMsg = "File at provided path is not a directory.";
553 break;
554 }
555 if (errno == EMFILE) {
556 errMsg = "The current process has too many open files.";
557 break;
558 }
559 if (errno == ENFILE) {
560 errMsg = "The system has too many open files.";
561 break;
562 }
563 sleep(2);
564 directory = opendir(path.c_str());
565 errMsg = "Received system error number " + std::to_string(errno);
566 }
567
568 // If the directory could be found and opened, we can go ahead and iterate
569 if (directory != nullptr) {
570 bool match;
571 while ((entry = readdir(directory))) {
572 match = std::regex_match(entry->d_name, pattern);
573 if( match ) {
574 // If the entry is a regular file or symlink AND the name matches the pattern,
575 // we can consider this ready to be interpretted as valid forcing data (even if it isn't)
576 #ifdef _DIRENT_HAVE_D_TYPE
577 if ( entry->d_type == DT_REG or entry->d_type == DT_LNK ) {
578 return forcing_params(
579 path + entry->d_name,
580 provider,
581 simulation_time_config.start_time,
582 simulation_time_config.end_time
583 );
584 }
585 else if ( entry->d_type == DT_UNKNOWN )
586 #endif
587 {
588 //dirent is not guaranteed to provide propoer file type identification in d_type
589 //so if a system returns unknown or it isn't supported, need to use stat to determine if it is a file
590 struct stat st;
591 if( stat((path+entry->d_name).c_str(), &st) != 0) {
592 throw std::runtime_error("Could not stat file "+path+entry->d_name);
593 }
594 if( S_ISREG(st.st_mode) ) {
595 //Sinde we used stat and not lstat, we get the result of the target of links as well
596 //so this covers both cases we are interested in.
597 return forcing_params(
598 path + entry->d_name,
599 provider,
600 simulation_time_config.start_time,
601 simulation_time_config.end_time
602 );
603 }
604 throw std::runtime_error("Forcing data is path "+path+entry->d_name+" is not a file");
605 }
606 } //no match found, try next entry
607 } // end while iter dir
608 } //end if directory
609 else {
610 // The directory wasn't found or otherwise couldn't be opened; forcing data cannot be retrieved
611 throw std::runtime_error("Error opening forcing data dir '" + path + "' after " + std::to_string(attemptCount) + " attempts: " + errMsg);
612 }
613
614 closedir(directory);
615
616 throw std::runtime_error("Forcing data could not be found for '" + identifier + "'");
617 }

626 void parse_external_model_params(boost::property_tree::ptree& model_params, const geojson::Feature catchment_feature) {
627 boost::property_tree::ptree attr {};
628 for (decltype(auto) param : model_params) {
629 if (param.second.count("source") == 0) {
630 attr.put_child(param.first, param.second);
631 continue;
632 }
633
634 decltype(auto) param_source = param.second.get_child("source");
635 decltype(auto) param_source_name = param_source.get_value<std::string>();
636 if (param_source_name != "hydrofabric") {
637 // temporary until the logic for alternative sources is designed
638 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
639 }
640
641 decltype(auto) param_name = param.second.find("from") == param.second.not_found()
642 ? param.first
643 : param.second.get_child("from").get_value<std::string>();
644
645 if (catchment_feature->has_property(param_name)) {
646 auto catchment_attribute = catchment_feature->get_property(param_name);
647 switch (catchment_attribute.get_type()) {
648 case geojson::PropertyType::Natural:
649 attr.put(param.first, catchment_attribute.as_natural_number());
650 break;
651 case geojson::PropertyType::Boolean:
652 attr.put(param.first, catchment_attribute.as_boolean());
653 break;
654 case geojson::PropertyType::Real:
655 attr.put(param.first, catchment_attribute.as_real_number());
656 break;
657 case geojson::PropertyType::String:
658 attr.put(param.first, catchment_attribute.as_string());
659 break;
660
661 case geojson::PropertyType::List:
662 case geojson::PropertyType::Object:
663 default:
664 std::cerr << "WARNING: property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
665 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
666 break;
667 }
668 } else {
669 std::cerr << "WARNING Failed to parse external parameter: catchment `"
670 << catchment_feature->get_id()
671 << "` does not contain the property `"
672 << param_name << "`\n";
673 }
674 }
675
676 model_params.swap(attr);
677 }

686 void parse_external_model_params(geojson::PropertyMap& model_params, const geojson::Feature catchment_feature) {
687 geojson::PropertyMap attr {};
688 for (decltype(auto) param : model_params) {
689 // Check for type to short-circuit. If param.second is not an object, `.has_key()` will throw
690 if (param.second.get_type() != geojson::PropertyType::Object || !param.second.has_key("source")) {
691 attr.emplace(param.first, param.second);
692 continue;
693 }
694
695 decltype(auto) param_source = param.second.at("source");
696 decltype(auto) param_source_name = param_source.as_string();
697 if (param_source_name != "hydrofabric") {
698 // TODO: temporary until the logic for alternative sources is designed
699 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
700 }
701
702 // Property name in the feature properties is either
703 // the value of key "from", or has the same name as
704 // the expected model parameter key
705 decltype(auto) param_name = param.second.has_key("from")
706 ? param.second.at("from").as_string()
707 : param.first;
708
709 if (catchment_feature->has_property(param_name)) {
710 auto catchment_attribute = catchment_feature->get_property(param_name);
711
712 // Use param.first in the `.emplace` calls instead of param_name since
713 // the expected name is given by the key of the model_params values.
714 switch (catchment_attribute.get_type()) {
715 case geojson::PropertyType::Natural:
716 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_natural_number()));
717 break;
