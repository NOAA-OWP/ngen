393 std::shared_ptr<Catchment_Formulation> construct_missing_formulation(geojson::Feature& feature, utils::StreamHandler output_stream, simulation_time_params &simulation_time_config){
394 const std::string identifier = feature->get_id();
395
396 std::string formulation_type_key = get_formulation_key(global_formulation_tree.get_child("formulations.."));
397
398 forcing_params forcing_config = this->get_forcing_params(this->global_forcing, identifier, simulation_time_config);
399
400 std::shared_ptr<Catchment_Formulation> missing_formulation = construct_formulation(formulation_type_key, identifier, forcing_config, output_stream);
401 // Need to work with a copy, since it is altered in-place
402 geojson::PropertyMap global_properties_copy = global_formulation_parameters;
403 Catchment_Formulation::config_pattern_substitution(global_properties_copy,
404 BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG, "{{id}}",
405 identifier);
406
407 // parse any external model parameters in this config
408
409 // handle single-bmi
410 if (global_properties_copy.count("model_params") > 0) {
411 decltype(auto) model_params = global_properties_copy.at("model_params");
412 geojson::PropertyMap model_params_copy = model_params.get_values();
413 parse_external_model_params(model_params_copy, feature);
414 global_properties_copy.at("model_params") = geojson::JSONProperty("model_params", model_params_copy);
415 }
416
417 // handle multi-bmi
418 // FIXME: this seems inefficient -- is there a better way?
419 if (global_properties_copy.count("modules") > 0) {
420 decltype(auto) nested_modules = global_properties_copy.at("modules").as_list();
421 for (auto& bmi_module : nested_modules) {
422 geojson::PropertyMap module_def = bmi_module.get_values();
423 geojson::PropertyMap module_params = module_def.at("params").get_values();
424 if (module_params.count("model_params") > 0) {
425 decltype(auto) model_params = module_params.at("model_params");
426 geojson::PropertyMap model_params_copy = model_params.get_values();
427 parse_external_model_params(model_params_copy, feature);
428 module_params.at("model_params") = geojson::JSONProperty("model_params", model_params_copy);
429 }
430 module_def.at("params") = geojson::JSONProperty("params", module_params);
431 bmi_module = geojson::JSONProperty("", module_def);
432 }
433 global_properties_copy.at("modules") = geojson::JSONProperty("", nested_modules);
434 }
435
436 missing_formulation->create_formulation(global_properties_copy);
437 return missing_formulation;
438 }

440 forcing_params get_forcing_params(geojson::PropertyMap &forcing_prop_map, std::string identifier, simulation_time_params &simulation_time_config) {
441 std::string path;
442 if(forcing_prop_map.count("path") != 0){
443 path = forcing_prop_map.at("path").as_string();
444 }
445 std::string provider;
446 if(forcing_prop_map.count("provider") != 0){
447 provider = forcing_prop_map.at("provider").as_string();
448 }
449 if (forcing_prop_map.count("file_pattern") == 0) {
450 return forcing_params(
451 path,
452 provider,
453 simulation_time_config.start_time,
454 simulation_time_config.end_time
455 );
456 }
457
458 if (path.empty()) {
459 throw std::runtime_error("Error with NGEN config - 'path' in forcing params must be set to a "
460 "non-empty parent directory path when 'file_pattern' is used.");
461 }
462
463 // Since we are given a pattern, we need to identify the directory and pull out anything that matches the pattern
464 if (path.compare(path.size() - 1, 1, "/") != 0) {
465 path += "/";
466 }
467
468 std::string filepattern = forcing_prop_map.at("file_pattern").as_string();
469
470 int id_index = filepattern.find("{{id}}");
471
472 // If an index for '{{id}}' was found, we can count on that being where the id for this realization can be found.
473 // For instance, if we have a pattern of '.*{{id}}_14_15.csv' and this is named 'cat-87',
474 // this will match on 'stuff_example_cat-87_14_15.csv'
475 if (id_index != std::string::npos) {
476 filepattern = filepattern.replace(id_index, sizeof("{{id}}") - 1, identifier);
477 }
478
479 // Create a regular expression used to identify proper file names
480 std::regex pattern(filepattern);
481
482 // A stream providing the functions necessary for evaluating a directory:
483 // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
484 DIR *directory = nullptr;
485
486 // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
487 struct dirent *entry = nullptr;
488
489 // Attempt to open the directory for evaluation
490 directory = opendir(path.c_str());
491 // Allow for a few retries in certain failure situations
492 size_t attemptCount = 0;
493 std::string errMsg;
494 while (directory == nullptr && attemptCount++ < 5) {
495 // For several error codes, we should break immediately and not retry
496 if (errno == ENOENT) {
497 errMsg = "No such file or directory.";
498 break;
499 }
500 if (errno == ENXIO) {
501 errMsg = "No such device or address.";
502 break;
503 }
504 if (errno == EACCES) {
505 errMsg = "Permission denied.";
506 break;
507 }
508 if (errno == EPERM) {
509 errMsg = "Operation not permitted.";
510 break;
511 }
512 if (errno == ENOTDIR) {
513 errMsg = "File at provided path is not a directory.";
514 break;
515 }
516 if (errno == EMFILE) {
517 errMsg = "The current process has too many open files.";
518 break;
519 }
520 if (errno == ENFILE) {
521 errMsg = "The system has too many open files.";
522 break;
523 }
524 sleep(2);
525 directory = opendir(path.c_str());
526 errMsg = "Received system error number " + std::to_string(errno);
527 }
528
529 // If the directory could be found and opened, we can go ahead and iterate
530 if (directory != nullptr) {
531 bool match;
532 while ((entry = readdir(directory))) {
533 match = std::regex_match(entry->d_name, pattern);
534 if( match ) {
535 // If the entry is a regular file or symlink AND the name matches the pattern,
536 // we can consider this ready to be interpretted as valid forcing data (even if it isn't)
537 #ifdef _DIRENT_HAVE_D_TYPE
538 if ( entry->d_type == DT_REG or entry->d_type == DT_LNK ) {
539 return forcing_params(
540 path + entry->d_name,
541 provider,
542 simulation_time_config.start_time,
543 simulation_time_config.end_time
544 );
545 }
546 else if ( entry->d_type == DT_UNKNOWN )
547 #endif
548 {
549 //dirent is not guaranteed to provide propoer file type identification in d_type
550 //so if a system returns unknown or it isn't supported, need to use stat to determine if it is a file
551 struct stat st;
552 if( stat((path+entry->d_name).c_str(), &st) != 0) {
553 throw std::runtime_error("Could not stat file "+path+entry->d_name);
554 }
555 if( S_ISREG(st.st_mode) ) {
556 //Sinde we used stat and not lstat, we get the result of the target of links as well
557 //so this covers both cases we are interested in.
558 return forcing_params(
559 path + entry->d_name,
560 provider,
561 simulation_time_config.start_time,
562 simulation_time_config.end_time
563 );
564 }
565 throw std::runtime_error("Forcing data is path "+path+entry->d_name+" is not a file");
566 }
567 } //no match found, try next entry
568 } // end while iter dir
569 } //end if directory
570 else {
571 // The directory wasn't found or otherwise couldn't be opened; forcing data cannot be retrieved
572 throw std::runtime_error("Error opening forcing data dir '" + path + "' after " + std::to_string(attemptCount) + " attempts: " + errMsg);
573 }
574
575 closedir(directory);
576
577 throw std::runtime_error("Forcing data could not be found for '" + identifier + "'");
578 }

587 void parse_external_model_params(boost::property_tree::ptree& model_params, const geojson::Feature catchment_feature) {
588 boost::property_tree::ptree attr {};
589 for (decltype(auto) param : model_params) {
590 if (param.second.count("source") == 0) {
591 attr.put_child(param.first, param.second);
592 continue;
593 }
594
595 decltype(auto) param_source = param.second.get_child("source");
596 decltype(auto) param_source_name = param_source.get_value<std::string>();
597 if (param_source_name != "hydrofabric") {
598 // temporary until the logic for alternative sources is designed
599 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
600 }
601
602 decltype(auto) param_name = param.second.find("from") == param.second.not_found()
603 ? param.first
604 : param.second.get_child("from").get_value<std::string>();
605
606 if (catchment_feature->has_property(param_name)) {
607 auto catchment_attribute = catchment_feature->get_property(param_name);
608 switch (catchment_attribute.get_type()) {
609 case geojson::PropertyType::Natural:
610 attr.put(param.first, catchment_attribute.as_natural_number());
611 break;
612 case geojson::PropertyType::Boolean:
613 attr.put(param.first, catchment_attribute.as_boolean());
614 break;
615 case geojson::PropertyType::Real:
616 attr.put(param.first, catchment_attribute.as_real_number());
617 break;
618 case geojson::PropertyType::String:
619 attr.put(param.first, catchment_attribute.as_string());
620 break;
621
622 case geojson::PropertyType::List:
623 case geojson::PropertyType::Object:
624 default:
625 std::cerr << "WARNING: property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
626 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
627 break;
628 }
629 } else {
630 std::cerr << "WARNING Failed to parse external parameter: catchment `"
631 << catchment_feature->get_id()
632 << "` does not contain the property `"
633 << param_name << "`\n";
634 }
635 }
636
637 model_params.swap(attr);
638 }

647 void parse_external_model_params(geojson::PropertyMap& model_params, const geojson::Feature catchment_feature) {
648 geojson::PropertyMap attr {};
649 for (decltype(auto) param : model_params) {
650 // Check for type to short-circuit. If param.second is not an object, `.has_key()` will throw
651 if (param.second.get_type() != geojson::PropertyType::Object || !param.second.has_key("source")) {
652 attr.emplace(param.first, param.second);
653 continue;
654 }
655
656 decltype(auto) param_source = param.second.at("source");
657 decltype(auto) param_source_name = param_source.as_string();
658 if (param_source_name != "hydrofabric") {
659 // TODO: temporary until the logic for alternative sources is designed
660 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
661 }
662
663 // Property name in the feature properties is either
664 // the value of key "from", or has the same name as
665 // the expected model parameter key
666 decltype(auto) param_name = param.second.has_key("from")
667 ? param.second.at("from").as_string()
668 : param.first;
669
670 if (catchment_feature->has_property(param_name)) {
671 auto catchment_attribute = catchment_feature->get_property(param_name);
672
673 // Use param.first in the `.emplace` calls instead of param_name since
674 // the expected name is given by the key of the model_params values.
675 switch (catchment_attribute.get_type()) {
676 case geojson::PropertyType::Natural:
677 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_natural_number()));
678 break;
679 case geojson::PropertyType::Boolean:
680 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_boolean()));
681 break;
682 case geojson::PropertyType::Real:
683 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_real_number()));
684 break;
685 case geojson::PropertyType::String:
686 attr.emplace(param.first, geojson::JSONProperty(param.first, catchment_attribute.as_string()));
687 break;
688
689 case geojson::PropertyType::List:
690 case geojson::PropertyType::Object:
691 default:
692 // TODO: Should list/object values be passed to model parameters?
693 // Typically, feature properties *should* be scalars.
694 std::cerr << "WARNING: property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
695 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
696 break;
697 }
698 }
699 }
700
701 model_params.swap(attr);
702 }
