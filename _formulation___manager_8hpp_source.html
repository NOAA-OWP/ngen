332 forcing_params get_forcing_params(geojson::PropertyMap &forcing_prop_map, std::string identifier, simulation_time_params &simulation_time_config) {
333 std::string path = forcing_prop_map.at("path").as_string();
334 std::string provider = "";
335 if(forcing_prop_map.count("provider") != 0){
336 provider = forcing_prop_map.at("provider").as_string();
337 }
338 if (forcing_prop_map.count("file_pattern") == 0) {
339 return forcing_params(
340 path,
341 provider,
342 simulation_time_config.start_time,
343 simulation_time_config.end_time
344 );
345 }
346
347 // Since we are given a pattern, we need to identify the directory and pull out anything that matches the pattern
348 if (path.compare(path.size() - 1, 1, "/") != 0) {
349 path += "/";
350 }
351
352 std::string filepattern = forcing_prop_map.at("file_pattern").as_string();
353
354 int id_index = filepattern.find("{{id}}");
355
356 // If an index for '{{id}}' was found, we can count on that being where the id for this realization can be found.
357 // For instance, if we have a pattern of '.*{{id}}_14_15.csv' and this is named 'cat-87',
358 // this will match on 'stuff_example_cat-87_14_15.csv'
359 if (id_index != std::string::npos) {
360 filepattern = filepattern.replace(id_index, sizeof("{{id}}") - 1, identifier);
361 }
362
363 // Create a regular expression used to identify proper file names
364 std::regex pattern(filepattern);
365
366 // A stream providing the functions necessary for evaluating a directory:
367 // https://www.gnu.org/software/libc/manual/html_node/Opening-a-Directory.html#Opening-a-Directory
368 DIR *directory = nullptr;
369
370 // structure representing the member of a directory: https://www.gnu.org/software/libc/manual/html_node/Directory-Entries.html
371 struct dirent *entry = nullptr;
372
373 // Attempt to open the directory for evaluation
374 directory = opendir(path.c_str());
375 // Allow for a few retries in certain failure situations
376 size_t attemptCount = 0;
377 std::string errMsg;
378 while (directory == nullptr && attemptCount++ < 5) {
379 // For several error codes, we should break immediately and not retry
380 if (errno == ENOENT) {
381 errMsg = "No such file or directory.";
382 break;
383 }
384 if (errno == ENXIO) {
385 errMsg = "No such device or address.";
386 break;
387 }
388 if (errno == EACCES) {
389 errMsg = "Permission denied.";
390 break;
391 }
392 if (errno == EPERM) {
393 errMsg = "Operation not permitted.";
394 break;
395 }
396 if (errno == ENOTDIR) {
397 errMsg = "File at provided path is not a directory.";
398 break;
399 }
400 if (errno == EMFILE) {
401 errMsg = "The current process has too many open files.";
402 break;
403 }
404 if (errno == ENFILE) {
405 errMsg = "The system has too many open files.";
406 break;
407 }
408 sleep(2);
409 directory = opendir(path.c_str());
410 errMsg = "Received system error number " + std::to_string(errno);
411 }
412
413 // If the directory could be found and opened, we can go ahead and iterate
414 if (directory != nullptr) {
415 bool match;
416 while ((entry = readdir(directory))) {
417 match = std::regex_match(entry->d_name, pattern);
418 if( match ) {
419 // If the entry is a regular file or symlink AND the name matches the pattern,
420 // we can consider this ready to be interpretted as valid forcing data (even if it isn't)
421 #ifdef _DIRENT_HAVE_D_TYPE
422 if ( entry->d_type == DT_REG or entry->d_type == DT_LNK ) {
423 return forcing_params(
424 path + entry->d_name,
425 provider,
426 simulation_time_config.start_time,
427 simulation_time_config.end_time
428 );
429 }
430 else if ( entry->d_type == DT_UNKNOWN )
431 #endif
432 {
433 //dirent is not guaranteed to provide propoer file type identification in d_type
434 //so if a system returns unknown or it isn't supported, need to use stat to determine if it is a file
435 struct stat st;
436 if( stat((path+entry->d_name).c_str(), &st) != 0) {
437 throw std::runtime_error("Could not stat file "+path+entry->d_name);
438 }
439 if( S_ISREG(st.st_mode) ) {
440 //Sinde we used stat and not lstat, we get the result of the target of links as well
441 //so this covers both cases we are interested in.
442 return forcing_params(
443 path + entry->d_name,
444 provider,
445 simulation_time_config.start_time,
446 simulation_time_config.end_time
447 );
448 }
449 throw std::runtime_error("Forcing data is path "+path+entry->d_name+" is not a file");
450 }
451 } //no match found, try next entry
452 } // end while iter dir
453 } //end if directory
454 else {
455 // The directory wasn't found or otherwise couldn't be opened; forcing data cannot be retrieved
456 throw std::runtime_error("Error opening forcing data dir '" + path + "' after " + std::to_string(attemptCount) + " attempts: " + errMsg);
457 }
458
459 closedir(directory);
460
461 throw std::runtime_error("Forcing data could not be found for '" + identifier + "'");
462 }
