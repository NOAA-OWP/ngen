195 double get_value(const string &output_name, const time_t &init_time, const long &duration_s,
196 const string &output_units) override
197 {
198 // Balk if not in this instance's collection of outputs
199 if (find(providedOutputs.begin(), providedOutputs.end(), output_name) == providedOutputs.end()) {
200 throw runtime_error("Unknown output " + output_name + " requested from wrapped provider");
201 }
202 // Balk if this instance is not ready
203 if (!isReadyToProvideData()) {
204 throw runtime_error("Cannot get value for " + output_name
205 + " from optional wrapped provider before it is ready");
206 }
207
208 // Remember: when the instance is ready to provide data, one or both of these **must** be true:
209 // * the desired output has a default value available
210 // * the desired output is provided by the backing wrapped provider
211
212 // If this output is not available from a wrapped provider OR we should override, then return the default
213 // Remember: isDefaultOverride() is not the same as "should use default" (see function docstring)
214 if (!isSuppliedByWrappedProvider(output_name) || isDefaultOverride(output_name)) {
215 // Take note if/how often default gets used in case that information is needed later
216 recordUsingDefault(output_name);
217 return defaultValues.at(output_name);
218 }
219 // Otherwise (i.e., available from wrapped provider and no override), get from backing wrapped provider
220 else {
221 return wrapped_provider->get_value(output_name, init_time, duration_s, output_units);
222 }
223 }

247 bool isDefaultOverride(const string &output_name) {
248 // First, there must be a default, and there must be something to override
249 if (!isSuppliedWithDefault(output_name) || !isSuppliedByWrappedProvider(output_name)) {
250 return false;
251 }
252 // Must have positive wait indicator value
253 const auto &wait_it = defaultUsageWaits.find(output_name);
254 return wait_it != defaultUsageWaits.end() && wait_it->second > 0;
255 }

265 bool isReadyToProvideData() override {
266 // When set, all values are either backed or have defaults, so instance is ready
267 if (isWrappedProviderSet()) {
268 return true;
269 }
270 // Otherwise, if there is a default for everything, we are also good
271 // Since the constructor validates there are no unexpected defaults, we can just compare sizes
272 else {
273 return providedOutputs.size() == defaultValues.size();
274 }

304 bool setWrappedProvider(ForcingProvider *provider) override {
305 // Disallow re-setting the provider if already validly set
306 // TODO: consider if this is still appropriate, given backing providers can be valid without providing
307 // everything (because of defaults), and so a new provider may supply more or less of what is needed
308 if (isWrappedProviderSet()) {
309 setMessage = "Cannot change wrapped provider after a valid provide has already been set";
310 return false;
311 }
312
313 // Confirm this will provide anything
314 if (provider == nullptr) {
315 setMessage = "Cannot set provider as null, as this cannot provide the values this type must proxy";
316 return false;
317 }
318
319 // Check this provides everything needed, accounting for defaults (and also tallying the outputs provided)
320 unsigned short providedByProviderCount = 0;
321 for (const string &requiredName : providedOutputs) {
322 // If supplied by the provider, increment our count and continue to the next required output name
323 if (isSuppliedByProvider(requiredName, provider)) {
324 ++providedByProviderCount;
325 continue;
326 }
327 // If something isn't supplied by the provider (i.e., the "else"), there has to be a default, or we bail
328 else if (!isSuppliedWithDefault(requiredName)) {
329 setMessage = "Given provider does not provide the required " + requiredName;
330 return false;
331 }
332 }
333
334 // Also make sure it provides at least 1 relevant output (possible no outputs were "needed" due to defaults)
335 if (providedByProviderCount == 0) {
336 setMessage = "Given provider does not provide any required outputs (all satisfied via defaults)";
337 return false;
338 }
339
340 // If this is good, set things and return true
341 wrapped_provider = provider;
342 setMessage.clear();
343 return true;
344 }

365 virtual void recordUsingDefault(const string &output_name) {
366 // Don't bother doing anything if there aren't waits assigned for this output
367 // Also, in this implementation, don't count usages until there is a backing provider that can provide this
368 auto waits_it = defaultUsageWaits.find(output_name);
369 if (waits_it != defaultUsageWaits.end() && isSuppliedByWrappedProvider(output_name)) {
370 // If this was not the last required default usage, simply decrement the wait count
371 if (waits_it->second > 1) {
372 waits_it->second -= 1;
373 }
374 // If this was the last required default usage wait, clear this entry from the collection
375 // Also, right now we are only supporting positive counts, so remove anything else as well
376 else {
377 defaultUsageWaits.erase(waits_it);
378 }
379 }
