70  size_t get_ts_index_for_time(const time_t &epoch_time) override {
71  if (epoch_time < start_date_time_epoch) {
72  throw std::out_of_range("Forcing had bad pre-start time for index query: " + std::to_string(epoch_time));
73  }
74  size_t i = 0;
75  // 1 hour
76  time_t seconds_in_time_step = 3600;
77  time_t time = start_date_time_epoch;
78  while (epoch_time >= time + seconds_in_time_step && time < end_date_time_epoch) {
79  i++;
80  time += seconds_in_time_step;
81  }
82  // The end_date_time_epoch is the epoch value of the BEGINNING of the last time step, not its end.
83  // I.e., to make sure we cover it, we have to go another time step beyond.
84  if (time >= end_date_time_epoch + 3600) {
85  throw std::out_of_range("Forcing had bad beyond-end time for index query: " + std::to_string(epoch_time));
86  }
87  else {
88  return i;
89  }

104  double get_value(const std::string &output_name, const time_t &init_time, const long &duration_s,
105  const std::string &output_units) override
106  {
107  size_t current_index;
108  long time_remaining = duration_s;
109  try {
110  current_index = get_ts_index_for_time(init_time);
111  }
112  catch (const std::out_of_range &e) {
113  throw std::out_of_range("Forcing had bad init_time " + std::to_string(init_time) + " for value request");
114  }
115 
116  std::vector<double> involved_time_step_values;
117  std::vector<long> involved_time_step_seconds;
118  long ts_involved_s;
119 
120  time_t first_time_step_start_epoch = start_date_time_epoch + (current_index * 3600);
121  // Handle the first time step differently, since we need to do more to figure out how many seconds came from it
122  // Total time step size minus the offset of the beginning, before the init time
123  ts_involved_s = 3600 - (init_time - first_time_step_start_epoch);
124 
125  involved_time_step_seconds.push_back(ts_involved_s);
126  involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
127  time_remaining -= ts_involved_s;
128  current_index++;
129 
130  while (time_remaining > 0) {
131  if(current_index >= time_epoch_vector.size())
132  return involved_time_step_values[involved_time_step_values.size()-1]; //TODO: Is this the right answer? Is returning any value off the end of the range valid?
133  ts_involved_s = time_remaining > 3600 ? 3600 : time_remaining;
134  involved_time_step_seconds.push_back(ts_involved_s);
135  involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
136  time_remaining -= ts_involved_s;
137  current_index++;
138  }
139  double value = 0;
140  for (size_t i = 0; i < involved_time_step_values.size(); ++i) {
141  if (is_param_sum_over_time_step(output_name))
142  value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / 3600.0);
143  else
144  value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / (double)duration_s);
145  }
146  return value;
147  }

277  void read_csv(string file_name)
278  {
279  int time_col_index = 0;
280  //std::map<std::string, int> col_indices;
281  std::vector<std::vector<double>*> local_valvec_index = {};
282 
283  //Call CSVReader constuctor
284  CSVReader reader(file_name);
285 
286  //Get the data from CSV File
287  std::vector<std::vector<std::string> > data_list = reader.getData();
288 
289  // Process the header (first) row..
290  int c = 0;
291  for (const auto& s : data_list[0]){
292  //std::cerr << s << std::endl;
293  if(s == "Time" || s == "time"){
294  time_col_index = c;
295  local_valvec_index.push_back(nullptr); // make sure the column indices line up!
296  } else {
297  std::string n = s;
298  std::string u = "";
299 
300  //TODO: parse units in parens and/or square brackets?
301 
302  if(well_known_fields.count(n) > 0){
303  auto t = well_known_fields[n];
304  n = std::get<0>(t);
305  u = u == "" ? std::get<1>(t) : u;
306  }
307 
308  //col_indices[n] = i;
309  forcing_vectors[n] = {};
310  local_valvec_index.push_back(&(forcing_vectors[n]));
311  available_forcings.push_back(n);
312  available_forcings_units[n] = u;
313  }
314  c++;
315  }
316 
317  time_t current_row_date_time_epoch;
318  //Iterate through CSV starting on the second row
319  int i = 1;
320  for (i = 1; i < data_list.size(); i++)
321  {
322  //Row vector
323  std::vector<std::string>& vec = data_list[i];
324 
325  struct tm current_row_date_time_utc = tm();
326  string time_str = vec[time_col_index];
327  //TODO: Support more time string formats? This is basically ISO8601 but not complete, support TZ?
328  strptime(time_str.c_str(), "%Y-%m-%d %H:%M:%S", &current_row_date_time_utc);
329 
330  //Convert current row date-time UTC to epoch time
331  current_row_date_time_epoch = timegm(&current_row_date_time_utc);
332 
333  //TODO: I am not sure this is a concern of this object. If forcing is retrieved that doesn't cover the
334  //needed time period, isn't that the requester's concern? (Methods exist to check this...)
335  //Ensure that forcing data covers the entire model period. Otherwise, throw an error.
336  if (i == 1 && start_date_time_epoch < current_row_date_time_epoch)
337  {
338  struct tm start_date_tm;
339  gmtime_r(&start_date_time_epoch, &start_date_tm);
340 
341  char tm_buff[128];
342  strftime(tm_buff, 128, "%Y-%m-%d %H:%M:%S", &start_date_tm);
343  throw std::runtime_error("Error: Forcing data " + file_name + " begins after the model start time:" + std::string(tm_buff) + " < " + time_str);
344  }
345 
346 
347  if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
348  {
349  time_epoch_vector.push_back(current_row_date_time_epoch);
350 
351  int c = -1;
352  for (auto& s : vec){
353  c++;
354  if(c == time_col_index)
355  continue;
356  boost::algorithm::trim(s);
357  local_valvec_index[c]->push_back(boost::lexical_cast<double>(s)); // This is supposed to update the vector in the map...
358  }
359 
360  }
361 
362  }
363  if (i <= 1 || current_row_date_time_epoch < end_date_time_epoch)
364  {
366  cout << "WARNING: Forcing data ends before the model end time." << endl;
367  //throw std::runtime_error("Error: Forcing data ends before the model end time.");
368  }
369  }
