82 size_t get_ts_index_for_time(const time_t &epoch_time) const override {
83 if (epoch_time < start_date_time_epoch) {
84 throw std::out_of_range("Forcing had bad pre-start time for index query: " + std::to_string(epoch_time));
85 }
86 size_t i = 0;
87 // 1 hour
88 time_t seconds_in_time_step = 3600;
89 time_t time = start_date_time_epoch;
90 while (epoch_time >= time + seconds_in_time_step && time < end_date_time_epoch) {
91 i++;
92 time += seconds_in_time_step;
93 }
94 // The end_date_time_epoch is the epoch value of the BEGINNING of the last time step, not its end.
95 // I.e., to make sure we cover it, we have to go another time step beyond.
96 if (time >= end_date_time_epoch + 3600) {
97 throw std::out_of_range("Forcing had bad beyond-end time for index query: " + std::to_string(epoch_time));
98 }
99 else {
100 return i;
101 }

114 double get_value(const CatchmentAggrDataSelector& selector, data_access::ReSampleMethod m) override
115 {
116 size_t current_index;
117 long time_remaining = selector.get_duration_secs();
118 auto init_time = selector.get_init_time();
119 auto output_name = selector.get_variable_name();
120 auto output_units = selector.get_output_units();
121
122 try {
123 current_index = get_ts_index_for_time(init_time);
124 }
125 catch (const std::out_of_range &e) {
126 throw std::out_of_range("Forcing had bad init_time " + std::to_string(init_time) + " for value request");
127 }
128
129 std::vector<double> involved_time_step_values;
130
131 std::vector<long> involved_time_step_seconds;
132 long ts_involved_s;
133
134 time_t first_time_step_start_epoch = start_date_time_epoch + (current_index * 3600);
135 // Handle the first time step differently, since we need to do more to figure out how many seconds came from it
136 // Total time step size minus the offset of the beginning, before the init time
137 ts_involved_s = 3600 - (init_time - first_time_step_start_epoch);
138
139 involved_time_step_seconds.push_back(ts_involved_s);
140 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
141 time_remaining -= ts_involved_s;
142 current_index++;
143
144 while (time_remaining > 0) {
145 if(current_index >= time_epoch_vector.size())
146 return involved_time_step_values[involved_time_step_values.size()-1]; //TODO: Is this the right answer? Is returning any value off the end of the range valid?
147 ts_involved_s = time_remaining > 3600 ? 3600 : time_remaining;
148 involved_time_step_seconds.push_back(ts_involved_s);
149 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
150 time_remaining -= ts_involved_s;
151 current_index++;
152
153 }
154 double value = 0;
155 for (size_t i = 0; i < involved_time_step_values.size(); ++i) {
156 if (is_param_sum_over_time_step(output_name))
157 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / 3600.0);
158 else
159 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / (double)selector.get_duration_secs());
160 }
161
162 // Convert units
163 try {
164 return UnitsHelper::get_converted_value(available_forcings_units[output_name], value, output_units);
165 }
166 catch (const std::runtime_error& e){
167 #ifndef UDUNITS_QUIET
168 std::cerr<<"WARN: Unit conversion unsuccessful - Returning unconverted value! (\""<<e.what()<<"\")"<<std::endl;
169 #endif
170 return value;
171 }

287 void read_csv(std::string file_name)
288 {
289 int time_col_index = 0;
290 //std::map<std::string, int> col_indices;
291 std::vector<std::vector<double>*> local_valvec_index = {};
292
293 //Call CSVReader constuctor
294 CSVReader reader(file_name);
295
296 //Get the data from CSV File
297 std::vector<std::vector<std::string> > data_list = reader.getData();
298
299 // Process the header (first) row..
300 int col_num = 0;
301 for (const auto& col_head : data_list[0]){
302 //std::cerr << s << std::endl;
303 if(col_head == "Time" || col_head == "time"){
304 time_col_index = col_num;
305 local_valvec_index.push_back(nullptr); // make sure the column indices line up!
306 } else {
307 std::string var_name = col_head;
308 std::string units = "";
309
310 boost::trim(var_name); // remove leading/trailing ws
311 const auto var_name_close = var_name.back();
312 if (var_name_close == ']' || var_name_close == ')') {
313 // found closing bracket/parenth
314
315 const bool is_bracket = var_name_close == ']';
316 const size_t var_name_open = is_bracket ? var_name.rfind('[') : var_name.rfind('(');
317 if (var_name_open != std::string::npos) {
318 // found matching opening bracket/parenth
319
320 units = var_name.substr(var_name_open + 1);
321 units.pop_back(); // remove closing bracket
322
323 var_name = var_name.substr(0, var_name_open);
324 boost::trim(var_name); // trim again in case of ws between name and units
325 }
326 }
327
328 auto wkf = data_access::WellKnownFields.find(var_name);
329 if(wkf != data_access::WellKnownFields.end()){
330 units = units.empty() ? std::get<1>(wkf->second) : units;
331 available_forcings.push_back(var_name); // Allow lookup by non-canonical name
332 available_forcings_units[var_name] = units; // Allow lookup of units by non-canonical name
333 var_name = std::get<0>(wkf->second); // Use the CSDMS name from here on
334 }
335
336 forcing_vectors[var_name] = {};
337 local_valvec_index.push_back(&(forcing_vectors[var_name]));
338 available_forcings.push_back(var_name);
339 available_forcings_units[var_name] = units;
340 }
341 col_num++;
342 }
343
344 time_t current_row_date_time_epoch;
345 //Iterate through CSV starting on the second row
346 int i = 1;
347 for (i = 1; i < data_list.size(); i++)
348 {
349 //Row vector
350 std::vector<std::string>& vec = data_list[i];
351
352 struct tm current_row_date_time_utc = tm();
353 std::string time_str = vec[time_col_index];
354 //TODO: Support more time string formats? This is basically ISO8601 but not complete, support TZ?
355 strptime(time_str.c_str(), "%Y-%m-%d %H:%M:%S", &current_row_date_time_utc);
356
357 //Convert current row date-time UTC to epoch time
358 current_row_date_time_epoch = timegm(&current_row_date_time_utc);
359
360 //TODO: I am not sure this is a concern of this object. If forcing is retrieved that doesn't cover the
361 //needed time period, isn't that the requester's concern? (Methods exist to check this...)
362 //Ensure that forcing data covers the entire model period. Otherwise, throw an error.
363 if (i == 1 && start_date_time_epoch < current_row_date_time_epoch)
364 {
365 struct tm start_date_tm;
366 gmtime_r(&start_date_time_epoch, &start_date_tm);
367
368 char tm_buff[128];
369 strftime(tm_buff, 128, "%Y-%m-%d %H:%M:%S", &start_date_tm);
370 throw std::runtime_error("Error: Forcing data " + file_name + " begins after the model start time:" + std::string(tm_buff) + " < " + time_str);
371 }
372
373
374 if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
375 {
376 time_epoch_vector.push_back(current_row_date_time_epoch);
377
378 int c = -1;
379 for (auto& s : vec){
380 c++;
381 if(c == time_col_index)
382 continue;
383 boost::algorithm::trim(s);
384 local_valvec_index[c]->push_back(boost::lexical_cast<double>(s)); // This is supposed to update the vector in the map...
385 }
386
387 }
388
389 }
390 if (i <= 1 || current_row_date_time_epoch < end_date_time_epoch)
391 {
393 std::cout << "WARNING: Forcing data ends before the model end time." << std::endl;
394 //throw std::runtime_error("Error: Forcing data ends before the model end time.");
395 }
396 }
