103 polygon_t polygon;
104
105 bool is_outer = true;
106 int inner_index = 0;
107
108 if (coordinates.size() > 1) {
109 polygon.inners().resize(coordinates.size() - 1);
110 }
111
112 for (auto &shape : coordinates) {
113 if (is_outer) {
114 for (auto &shape_coordinates : shape) {
115 polygon.outer().push_back(
116 coordinate_t(shape_coordinates[0], shape_coordinates[1])
117 );
118 }
119 is_outer = false;
120 }
121 else {
122 for (auto &shape_coordinates : shape) {
123 polygon.inners()[inner_index].push_back(
124 coordinate_t(shape_coordinates[0], shape_coordinates[1])
125 );
126 }
127 inner_index++;
128 }
129 }
130
131 return polygon;
132 }
