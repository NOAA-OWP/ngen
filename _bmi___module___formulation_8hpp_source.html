125 // AORC uses wind speeds from 10m. Must convert to 2m.
126 if (et_options.yes_aorc == TRUE) {
127 // wind speed was measured at 10.0 m height, so we need to calculate the wind speed at 2.0m
128 double numerator = log(2.0 / et_params.zero_plane_displacement_height_m);
129 double denominator = log(10.0 / et_params.zero_plane_displacement_height_m);
130 // this is the 2 m value
131 et_forcing.wind_speed_m_per_s = et_forcing.wind_speed_m_per_s * numerator / denominator;
132 }

146 // transfer aorc forcing data into our data structure for surface radiation calculations
147 surf_rad_forcing.incoming_shortwave_radiation_W_per_sq_m = raw_aorc.DSWRF_surface_W_per_meters_squared;
148 surf_rad_forcing.incoming_longwave_radiation_W_per_sq_m = raw_aorc.DLWRF_surface_W_per_meters_squared;
149 surf_rad_forcing.air_temperature_C = et_forcing.air_temperature_C;
150 // compute relative humidity from specific humidity..
151 double saturation_vapor_pressure_Pa, actual_vapor_pressure_Pa;
152 saturation_vapor_pressure_Pa = et::calc_air_saturation_vapor_pressure_Pa(surf_rad_forcing.air_temperature_C);
153 actual_vapor_pressure_Pa = raw_aorc.SPFH_2maboveground_kg_per_kg * raw_aorc.PRES_surface_Pa / 0.622;
154 surf_rad_forcing.relative_humidity_percent = 100.0 * actual_vapor_pressure_Pa / saturation_vapor_pressure_Pa;
155 // sanity check the resulting value. Should be less than 100%. Sometimes air can be supersaturated.
156 if (100.0 < surf_rad_forcing.relative_humidity_percent) {
157 surf_rad_forcing.relative_humidity_percent = 99.0;
158 }
159 }
160 /* Shouldn't need this for now, but might later

163

357 double get_value(const std::string &output_name, const time_t &init_time, const long &duration_s,
358 const std::string &output_units) override
359 {
360 // First make sure this is an available output
361 const std::vector<std::string> forcing_outputs = get_available_forcing_outputs();
362 if (std::find(forcing_outputs.begin(), forcing_outputs.end(), output_name) == forcing_outputs.end()) {
363 throw runtime_error(get_formulation_type() + " received invalid output forcing name " + output_name);
364 }
365 // TODO: do this, or something better, later; right now, just assume anything using this as a provider is
366 // consistent with times
367 /*
368 if (last_model_response_delta == 0 && last_model_response_start_time == 0) {
369 throw runtime_error(get_formulation_type() + " does not properly set output time validity ranges "
370 "needed to provide outputs as forcings");
371 }
372 */
373
374 // Handle ET requests slightly differently
375 //TODO: This should really only happen if neither of these output_names are provided by the module...
376 // But this code should also probably be removed in the near future (see GH #297 second comment).
377 if (output_name == NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP || output_name == CSDMS_STD_NAME_POTENTIAL_ET) {
378 return calc_et();
379 }
380
381 // If not ET, now get correct BMI variable name, which may be the output or something mapped to this output.
382 // TODO: move this to a dedicated function
383 std::string bmi_var_name;
384 std::vector<std::string> output_names = get_bmi_model()->GetOutputVarNames();
385 if (std::find(output_names.begin(), output_names.end(), output_name) != output_names.end()) {
386 bmi_var_name = output_name;
387 }
388 else {
389 for (auto & iter : bmi_var_names_map) {
390 if (iter.second == output_name) {
391 bmi_var_name = iter.first;
392 break;
393 }
394 }
395 }
396
397 double value = get_var_value_as_double(bmi_var_name);
398
399 // Convert units
400 std::string native_units = get_bmi_model()->GetVarUnits(bmi_var_name);
401 try {
402 return UnitsHelper::get_converted_value(native_units, value, output_units);
403 }
404 catch (const std::runtime_error& e){
405 std::cerr<<"Unit conversion error: "<<std::endl<<e.what()<<std::endl<<"Returning unconverted value!"<<std::endl;
406 return value;
407 }

565 void inner_create_formulation(geojson::PropertyMap properties, bool needs_param_validation) {
566 if (needs_param_validation) {
567 validate_parameters(properties);
568 }
569 // Required parameters first
570 set_bmi_init_config(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__INIT_CONFIG).as_string());
571 set_bmi_main_output_var(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MAIN_OUT_VAR).as_string());
572 set_model_type_name(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__MODEL_TYPE).as_string());
573 set_bmi_using_forcing_file(properties.at(BMI_REALIZATION_CFG_PARAM_REQ__USES_FORCINGS).as_boolean());
574
575 // Then optional ...
576
577 // Note that this must be present if bmi_using_forcing_file is true
578 if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE) != properties.end())
579 set_forcing_file_path(properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FORCING_FILE).as_string());
580 else if (is_bmi_using_forcing_file())
581 throw std::runtime_error("Can't create BMI formulation: using_forcing_file true but no file path set");
582
583 if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END) != properties.end()) {
584 set_allow_model_exceed_end_time(
585 properties.at(BMI_REALIZATION_CFG_PARAM_OPT__ALLOW_EXCEED_END).as_boolean());
586 }
587 if (properties.find(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP) != properties.end()) {
588 set_bmi_model_time_step_fixed(
589 properties.at(BMI_REALIZATION_CFG_PARAM_OPT__FIXED_TIME_STEP).as_boolean());
590 }
591
592 auto std_names_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__VAR_STD_NAMES);
593 if (std_names_it != properties.end()) {
594 geojson::PropertyMap names_map = std_names_it->second.get_values();
595 for (auto& names_it : names_map) {
596 bmi_var_names_map.insert(
597 std::pair<std::string, std::string>(names_it.first, names_it.second.as_string()));
598 }
599 }
600
601 // Do this next, since after checking whether other input variables are present in the properties, we can
602 // now construct the adapter and init the model
603 set_bmi_model(construct_model(properties));
604
605 // Make sure that this is able to interpret model time and convert to real time, since BMI model time is
606 // usually starting at 0 and just counting up
607 determine_model_time_offset();
608
609 // Output variable subset and order, if present
610 auto out_var_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_VARS);
611 if (out_var_it != properties.end()) {
612 std::vector<geojson::JSONProperty> out_vars_json_list = out_var_it->second.as_list();
613 std::vector<std::string> out_vars(out_vars_json_list.size());
614 for (int i = 0; i < out_vars_json_list.size(); ++i) {
615 out_vars[i] = out_vars_json_list[i].as_string();
616 }
617 set_output_variable_names(out_vars);
618 }
619 // Otherwise, just take what literally is provided by the model
620 else {
621 set_output_variable_names(get_bmi_model()->GetOutputVarNames());
622 }
623
624 // Output header fields, if present
625 auto out_headers_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUT_HEADER_FIELDS);
626 if (out_headers_it != properties.end()) {
627 std::vector<geojson::JSONProperty> out_headers_json_list = out_var_it->second.as_list();
628 std::vector<std::string> out_headers(out_headers_json_list.size());
629 for (int i = 0; i < out_headers_json_list.size(); ++i) {
630 out_headers[i] = out_headers_json_list[i].as_string();
631 }
632 set_output_header_fields(out_headers);
633 }
634 else {
635 set_output_header_fields(get_output_variable_names());
636 }
637 // Create a reference to this for ET by using a WrappedForcingProvider
638 std::shared_ptr<forcing::ForcingProvider> self = std::make_shared<forcing::WrappedForcingProvider>(this);
639 input_forcing_providers[NGEN_STD_NAME_POTENTIAL_ET_FOR_TIME_STEP] = self;
640 input_forcing_providers[CSDMS_STD_NAME_POTENTIAL_ET] = self;
641
642 // Output precision, if present
643 auto out_precision_it = properties.find(BMI_REALIZATION_CFG_PARAM_OPT__OUTPUT_PRECISION);
644 if (out_precision_it != properties.end()) {
645 set_output_precision(properties.at(BMI_REALIZATION_CFG_PARAM_OPT__OUTPUT_PRECISION).as_natural_number());
646 }
647
648 // Finally, make sure this is set
649 model_initialized = get_bmi_model()->is_model_initialized();
650 }
