54 //The function that parses the json file and build a unordered set and vector of structs for each line in the json list
55 void parse_partition_file() {
56 std::cout << "\nroot_tree: " << tree.size() << std::endl;
57
58 //Declare two partition_data type structs
59 PartitionData part_data;
60
61 //The outter loop iterating through the list of partitions
62 int remote_mpi_rank;
63 std::string remote_nex_id;
64 std::string remote_cat_id;
65 std::string direction;
66 Tuple tmp_tuple;
67 std::vector<Tuple> remote_conn_vec;
68 int part_counter = 0;
69 for(auto &partition: tree.get_child("partitions")) {
70 //Get partition id
71 std::string part_id = (partition.second).get<std::string>("id");
72
73 //Get mpi_world_rank and set the corresponding part_data struct member
74 int mpi_rank = std::stoi(part_id);
75 part_data.mpi_world_rank = mpi_rank;
76
77 geojson::JSONProperty part = geojson::JSONProperty(part_id, partition.second);
78
79 //Get cat_ids list and insert the elements into the unordered_set catchment_ids in part_data struct
80 for (auto &cat_id : part.at("cat-ids").as_list())
81 {
82 //cat_ids.push_back(remote.as_string());
83 catchment_ids.emplace(cat_id.as_string());
84 }
85 part_data.catchment_ids = catchment_ids;
86 catchment_ids.clear();
87
88 //Get nex_ids list and insert the elements into the unordered_set nexus_ids in part_data struct
89 for (auto &nex_id: part.at("nex-ids").as_list())
90 {
91 //nex_ids.push_back(nex_id.as_string());
92 nexus_ids.emplace(nex_id.as_string());
93 }
94 part_data.nexus_ids = nexus_ids;
95 nexus_ids.clear();
96
97 if( part.at("remote-connections").get_type() == geojson::PropertyType::List )
98 {
99 //It is valid to have no remote connections, but the backend property tree parser
100 //can't represent empty lists/objects, so it turns into an empty string (which is iterable)
101 //so we check to ensure the remote connections are a list type (not string) before we attempt
102 //to process the remote-connections. If they are empty, this step gets skipped entirely.
103 //Get remote-connections and set the corresponding part_data struct member
104
105 for (auto &remote_conn : part.at("remote-connections").as_list())
106 {
107 if ( remote_conn.get_type() != geojson::PropertyType::String )
108 {
109 remote_mpi_rank = remote_conn.at("mpi-rank").as_natural_number();
110 remote_nex_id = remote_conn.at("nex-id").as_string();
111 remote_cat_id = remote_conn.at("cat-id").as_string();
112 direction = remote_conn.at("cat-direction").as_string();
113 tmp_tuple = std::make_tuple(remote_mpi_rank, remote_nex_id, remote_cat_id, direction);
114 remote_conn_vec.push_back(tmp_tuple);
115 }
116 }
117 }
118
119 part_data.remote_connections = remote_conn_vec;
120 remote_conn_vec.clear();
121
122 //Push part_data struct the vector
123 partition_ranks.push_back(part_data);
124
125 part_counter++;
126
127 }
128 };
