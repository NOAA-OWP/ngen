30 static coordinate_t build_point(boost::property_tree::ptree &tree) {
31 std::vector<double> point;
32
33 // GeoJSON specifies that the details of the point will lie in a 1D array under "coordinates"
34 for(auto &shape : tree.get_child("coordinates")) {
35 // The specification of the points will lie within string values, so we need to convert the strings
36 // to double values
37 point.push_back(std::stod(shape.second.data()));
38 }
39
40 // The specification of GeoJson specifies that points will be described within a 1D array of length 2
41 // (the longitude, then the latitude). Use both in order to construct the point and return it
42 return coordinate_t(point[0], point[1]);
43 }

73 static polygon_t build_polygon(boost::property_tree::ptree &tree) {
74 three_dimensional_coordinates total_coordinates;
75
76 for(auto &shape : tree.get_child("coordinates")) {
77
78 two_dimensional_coordinates shape_coordinates;
79 for (auto &group : shape.second) {
80 std::vector<double> latLon;
81 for (auto &coordinate : group.second) {
82 latLon.push_back(std::stod(coordinate.second.data()));
83 }
84 shape_coordinates.push_back(std::move(latLon));
85 }
86 total_coordinates.push_back(std::move(shape_coordinates));
87 }
88
89 polygon_t polygon;
90
91 if (total_coordinates.size() > 1) {
92 polygon.inners().resize(total_coordinates.size() - 1);
93 }
94
95 bool is_outer = true;
96 int inner_index = 0;
97
98 for (auto& shape : total_coordinates) {
99 if (is_outer) {
100
101 for(auto &coords : shape) {
102 polygon.outer().push_back(
103 coordinate_t(coords[0], coords[1])
104 );
105 }
106
107 is_outer = false;
108 }
109 else {
110
111 for (auto &coords : shape) {
112
113 polygon.inners()[inner_index].push_back(
114 coordinate_t(coords[0], coords[1])
115 );
116
117 }
118 inner_index++;
119 }
120 }
121
122 return polygon;
123 }

164 four_dimensional_coordinates coordinates_groups;
165 for (auto &polygon : tree.get_child("coordinates")) {
166 three_dimensional_coordinates polygon_coordinates;
167 for (auto &group : polygon.second) {
168 two_dimensional_coordinates coordinate_group;

170 std::vector<double> latLon;
171 for(auto &coordinate : point.second) {
172 latLon.push_back(std::stod(coordinate.second.data()));
173 }
174 coordinate_group.push_back(std::move(latLon));
175 }
176 polygon_coordinates.push_back(std::move(coordinate_group));
177 }
178
179 coordinates_groups.push_back(std::move(polygon_coordinates));
180 }
181
182 bg::model::multi_polygon<polygon_t> multi_polygon;
183
184 for (auto& polygon_group : coordinates_groups) {
185 for (auto &polygon_points : polygon_group) {
186 polygon_t polygon;
187 for (auto &point : polygon_points) {
188 bg::append(polygon, coordinate_t(point[0], point[1]));
189 }
190 multi_polygon.push_back(std::move(polygon));
191 }
192 }

227 static Feature build_feature(boost::property_tree::ptree &tree) {
228 bool has_geometry_collection = false;
229 bool has_geometry = false;
230
231 geometry geometry_object;
232 std::vector<geometry> geometry_collection;
233 FeatureType type = FeatureType::None;
234 std::string id = "";
235 std::vector<double> bounding_box;
236 PropertyMap properties;
237 PropertyMap foreign_members;
238
239 for (auto& child : tree) {
240 if (child.first == "geometry") {
241 const std::string& geometry_type = child.second.get<std::string>("type");
242 geometry_object = build_geometry(child.second);
243 has_geometry = true;
244
245 if (geometry_type == "Point") {
246 type = FeatureType::Point;
247 }
248 else if (geometry_type == "LineString") {
249 type = FeatureType::LineString;
250 }
251 else if (geometry_type == "Polygon") {
252 type = FeatureType::Polygon;
253 }
254 else if (geometry_type == "MultiPoint") {
255 type = FeatureType::MultiPoint;
256 }
257 else if (geometry_type == "MultiLineString") {
258 type = FeatureType::MultiLineString;
259 }
260 else if (geometry_type == "MultiPolygon") {
261 type = FeatureType::MultiPolygon;
262 }
263 }
264 else if (child.first == "geometries") {
265 // Since the feature can have a number of different types of geometries and the
266 // type of the feature comes from the geometry, we simply set this as a collection
267 type = FeatureType::GeometryCollection;
268 has_geometry_collection = true;
269
270 // Loop through the underlying collection of geometric json definitions and use
271 // those to create geometric objects
272 for (auto &geom : child.second) {
273 geometry_collection.push_back(build_geometry(geom.second));
274 }
275 }
276 else if (child.first == "id") {
277 id = std::move(child.second.data());
278 }
279 else if (child.first == "bbox") {
280 for (auto &value : tree.get_child("bbox")) {
281 bounding_box.push_back(std::stod(value.second.data()));
282 }
283 }
284 else if (child.first == "properties") {
285 for (auto& property : child.second) {
286 properties.emplace(property.first, std::move(JSONProperty(property.first, property.second)));
287 }
288 }
289 else {
290 foreign_members.emplace(child.first, std::move(JSONProperty(child.first, child.second)));
291 }
292 }
293
294
295 switch (type) {
296 case FeatureType::Point:
297 return std::make_shared<PointFeature>(PointFeature(
298 boost::get<coordinate_t>(geometry_object),
299 id,
300 properties,
301 bounding_box,
302 std::vector<FeatureBase*>(),
303 std::vector<FeatureBase*>(),
304 foreign_members
305 ));
306 case FeatureType::LineString:
307 return std::make_shared<LineStringFeature>(LineStringFeature(
308 boost::get<linestring_t>(geometry_object),
309 id,
310 properties,
311 bounding_box,
312 std::vector<FeatureBase*>(),
313 std::vector<FeatureBase*>(),
314 foreign_members
315 ));
316 case FeatureType::Polygon:
317 return std::make_shared<PolygonFeature>(PolygonFeature(
318 boost::get<polygon_t>(geometry_object),
319 id,
320 properties,
321 bounding_box,
322 std::vector<FeatureBase*>(),
323 std::vector<FeatureBase*>(),
324 foreign_members
325 ));
326 case FeatureType::MultiPoint:
327 return std::make_shared<MultiPointFeature>(MultiPointFeature(
328 boost::get<multipoint_t>(geometry_object),
329 id,
330 properties,
331 bounding_box,
332 std::vector<FeatureBase*>(),
333 std::vector<FeatureBase*>(),
334 foreign_members
335 ));
336 case FeatureType::MultiLineString:
337 return std::make_shared<MultiLineStringFeature>(MultiLineStringFeature(
338 boost::get<multilinestring_t>(geometry_object),
339 id,
340 properties,
341 bounding_box,
342 std::vector<FeatureBase*>(),
343 std::vector<FeatureBase*>(),
344 foreign_members
345 ));
346 case FeatureType::MultiPolygon:
347 return std::make_shared<MultiPolygonFeature>(MultiPolygonFeature(
348 boost::get<multipolygon_t>(geometry_object),
349 id,
350 properties,
351 bounding_box,
352 std::vector<FeatureBase*>(),
353 std::vector<FeatureBase*>(),
354 foreign_members
355 ));
356 default:
357 return std::make_shared<CollectionFeature>(CollectionFeature(
358 geometry_collection,
359 id,
360 properties,
361 bounding_box,
362 std::vector<FeatureBase*>(),
363 std::vector<FeatureBase*>(),
364 foreign_members
365 ));
366 }
367 }

374 static GeoJSON build_collection(const boost::property_tree::ptree& tree, const std::vector<std::string>& ids={}) {
375 std::vector<double> bbox_values;
376 std::vector<Feature> features;
377 PropertyMap foreign_members;
378 std::string tmp_id; //a temporary string to hold feature identities
379
380 for (auto& child : tree) {
381 if (child.first == "bbox") {
382 std::vector<std::string> bounding_box;
383 for (auto &feature : tree.get_child("bbox")) {
384 bounding_box.push_back(std::move(feature.second.data()));
385 }
386
387 for (int point_index = 0; point_index < bounding_box.size(); point_index++) {
388 bbox_values.push_back(std::stod(bounding_box[point_index]));
389 }
390 }
391 else if (child.first == "features") {
392 boost::optional<const boost::property_tree::ptree&> e = tree.get_child_optional("features");
393
394 if (e) {
395 for(auto feature_tree : *e) {
396 Feature feature = build_feature(feature_tree.second);
397 tmp_id = feature->get_id();
398 //TODO feature identity isn't 100% spec compliant. GeoJSON allows for a feature to have an
399 //optional id, but the input files set id under the 'property' key, so when a feature is constructed
400 //feature->get_id() returns '' because the feature itself doesn't have an id, so we hae to read
401 //from the associated properties to find the id. If the inputs change, we will need to adjust
402 //this line to read feature->get_id()
403 if( tmp_id == "" ) {
404 try {
405 tmp_id = feature->get_property("id").as_string();
406 feature->set_id(tmp_id);
407 }
408 catch (const std::out_of_range& error) {
409 tmp_id = "";
410 }
411 }
412
413 if( ids.empty() || std::find(ids.begin(), ids.end(), tmp_id) != ids.end() ) {
414 //ids is empty, meaning we want all features,
415 //or feature id was found in the provided ids vector
416 //so hold the feature to add to collection later
417
418 features.push_back(std::move(feature));
419 }
420 feature_tree.second.erase("child"); //we are done with this feature, drop it from the ptree. Not convinced this releases any resources, though =(
421 }
422 }
423 else {
424 std::cout << "No features were found" << std::endl;
425 }
426 }
427 else {
428 foreign_members.emplace(child.first, std::move(JSONProperty(child.first, child.second)));
429
430 }
431
432 }
433 GeoJSON collection = std::make_shared<FeatureCollection>(FeatureCollection(std::move(features), std::move(bbox_values)));
434
435 for (Feature feature : features) {
436 if (feature->get_id() != "") {
437 collection->add_feature_id(feature->get_id(), feature);
438 }
439 }
440
441 return collection;
442 }
