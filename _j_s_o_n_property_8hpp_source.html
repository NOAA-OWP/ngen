92 JSONProperty(std::string value_key, const boost::property_tree::ptree& property_tree) {
93 key = value_key;
94
95 if (property_tree.empty() && !property_tree.data().empty()) {
96 // This is a terminal node and has a raw value
97 // TODO: Add handling for nested objects by determining if property.second is another ptree
98 std::string value = property_tree.data();
99
100 if (value == "true" || value == "false") {
101 type = PropertyType::Boolean;
102 boolean = value == "true";
103 data = boolean;
104 }
105 else {
106 //Try natural number first since double cant cast to int/long
107 try{
108 natural_number = boost::lexical_cast<long>(value);
109 type = PropertyType::Natural;
110 data = natural_number;
111 }
112 catch (boost::bad_lexical_cast &e)
113 {
114 try {
115 //Try to cast to double/real next
116 real_number = boost::lexical_cast<double>(value);
117 type = PropertyType::Real;
118 data = real_number;
119 }
120 catch (boost::bad_lexical_cast & e)
121 {
122 //At this point, we are left with string option
123 string = value;
124 type = PropertyType::String;
125 data = string;
126 }
127 }
128 }
129 }
130 else {
131 // This isn't a terminal node, therefore represents an object or array
132 List tmp_list;
133 Object tmp_obj;
134 //TODO unit test these construction paths...
135 for (auto &property : property_tree) {
136 if (property.first.empty()) {
137 type = PropertyType::List;
138 JSONProperty tmp = JSONProperty(value_key, property.second);
139 value_list.push_back(tmp);
140 tmp_list.values.push_back(tmp.data);
141 data = tmp_list;
142 }
143 else {
144 type = PropertyType::Object;
145 JSONProperty tmp = JSONProperty(property.first, property.second);
146 values.emplace(property.first, tmp);
147 tmp_obj.values.emplace(property.first, tmp.data);
148 data = tmp_obj;
149 }
150 }
151 }
152 }

238 JSONProperty(std::string value_key, std::string value) {
239 key = value_key;
240 string = value;
241
242 if (value == "true" || value == "false") {
243 type = PropertyType::Boolean;
244 boolean = value == "true";
245 data = boolean;
246 }
247 else {
248 bool is_numeric = true;
249 bool is_real = true;
250 bool decimal_already_hit = false;
251
252 for(int character_index = 0; character_index < value.length(); character_index++) {
253 char character = value[character_index];
254
255 // If the first character is a '0' or isn't a digit, the whole value cannot be a number
256 if (character_index == 0 && character == '0') {
257 is_numeric = false;
258 }
259 else if (character != '.' && !std::isdigit(character)) {
260 // If this character isn't a decimal point and isn't a digit, the whole value cannot be a number
261 is_numeric = false;
262 is_real = false;
263 break;
264 }
265 else if (character == '.' && decimal_already_hit) {
266 // If this character is a decimal point, but we've already seen one, the whole value cannot be a number
267 is_real = false;
268 break;
269 }
270 else if (character == '.') {
271 // If a decimal point is seen, the whole value cannot be an integer
272 is_numeric = false;
273 decimal_already_hit = true;
274 }
275 }
276
277 // If the value can be represented as a whole number, we want to go with that
278 if (is_numeric) {
279 type = PropertyType::Natural;
280 natural_number = std::stol(value);
281 data = natural_number;
282 }
283 else if (is_real) {
284 type = PropertyType::Real;
285 real_number = std::stod(value);
286 data = real_number;
287 }
288 else {
289 // Otherwise we'll store everything as a raw string
290 type = PropertyType::String;
291 string = value;
292 data = string;
293 }
294 }
295 }

305 JSONProperty(const JSONProperty &original) {
306 type = original.type;
307 key = original.key;
308 data = original.data;
309 switch (type) {
310 case PropertyType::Boolean:
311 boolean = original.boolean;

314 natural_number = original.natural_number;

316 case PropertyType::Real:

329 values.emplace(pair.first, JSONProperty(pair.second));

445 bool inline operator==(const JSONProperty& other) {
446 if (not (this->type == other.type)) {
447 return false;
448 }
449
450 if (this->type == PropertyType::Object) {
451 if (this->values.size() != other.values.size()) {
452 return false;

454
455 for (std::string their_key : other.keys()) {

457 return false;
458 }
459 }
460
461 }
