168 JSONProperty(std::string value_key, const boost::property_tree::ptree& property_tree):
169 key(std::move(value_key)) {
170
171 if(property_tree.empty() && property_tree.data().empty()) {
172 //Since property trees don't represent empty strings, list, or objects, they are all just `empty`
173 //so we can trap that case with `property_tree.data().empty()`, and the best we can do is make it an
174 //empty string
175 type = PropertyType::String;
176 //Also note that a boost::variant assigned to the static empty string "" like this
177 //`data = "";`
178 //will cause `data` to actually become a bool type, not the std::string.
179 //So use a default string to intialize the empty data
180 data = std::string();
181 }
182 else if (property_tree.empty() && !property_tree.data().empty()) {
183 // This is a terminal node and has a raw value
184
185 const std::string& value = property_tree.data();
186
187 if (value == "true" || value == "false") {
188 type = PropertyType::Boolean;
189 data = value == "true";
190 }
191 else {
192 //Try natural number first since double cant cast to int/long
193 long casted_long_data;
194 double casted_double_data;
195 if( boost::conversion::try_lexical_convert<long>(value, casted_long_data) ){
196 type = PropertyType::Natural;
197 data = casted_long_data;
198 }
199 else if( boost::conversion::try_lexical_convert<double>(value, casted_double_data) ){
200 //Try to cast to double/real next
201 type = PropertyType::Real;
202 data = casted_double_data;
203 }
204 else{
205 //At this point, we are left with string option
206 //string = value;
207 type = PropertyType::String;
208 data = std::move(value);
209 }
210 }
211 }
212 else {
213 // This isn't a terminal node, therefore represents an object or array
214 //TODO unit test these construction paths...
215 for (auto &property : property_tree) {
216 if (property.first.empty()) {
217 type = PropertyType::List;
218 value_list.push_back(std::move(JSONProperty(value_key, property.second)));
219 data = List(& value_list );
220 }
221 else {
222 type = PropertyType::Object;
223 values.emplace(property.first, std::move(JSONProperty(property.first, property.second)));
224 data = Object( & values );
225 }
226 }
227 }
228 }

308 JSONProperty(std::string value_key, std::string value):key(std::move(value_key)) {
309 if (value == "true" || value == "false") {
310 type = PropertyType::Boolean;
311 //boolean = value == "true";
312 data = value == "true";
313 }
314 else {
315 bool is_numeric = true;
316 bool is_real = true;
317 bool decimal_already_hit = false;
318
319 for(int character_index = 0; character_index < value.length(); character_index++) {
320 char character = value[character_index];
321
322 // If the first character is a '0' or isn't a digit, the whole value cannot be a number
323 if (character_index == 0 && character == '0') {
324 is_numeric = false;
325 }
326 else if (character != '.' && !std::isdigit(character)) {
327 // If this character isn't a decimal point and isn't a digit, the whole value cannot be a number
328 is_numeric = false;
329 is_real = false;
330 break;
331 }
332 else if (character == '.' && decimal_already_hit) {
333 // If this character is a decimal point, but we've already seen one, the whole value cannot be a number
334 is_real = false;
335 break;
336 }
337 else if (character == '.') {
338 // If a decimal point is seen, the whole value cannot be an integer
339 is_numeric = false;
340 decimal_already_hit = true;
341 }
342 }
343
344 // If the value can be represented as a whole number, we want to go with that
345 if (is_numeric) {
346 type = PropertyType::Natural;
347 //natural_number = std::stol(value);
348 data = std::stol(value);
349 }
350 else if (is_real) {
351 type = PropertyType::Real;
352 //real_number = std::stod(value);
353 data = std::stod(value);
354 }
355 else {
356 // Otherwise we'll store everything as a raw string
357 type = PropertyType::String;
358 //string = value;
359 data = value;
360 }
361 }
362 }
