56  JSONProperty(std::string value_key, const boost::property_tree::ptree& property_tree) {
57  key = value_key;
58 
59  if (property_tree.empty() && !property_tree.data().empty()) {
60  // This is a terminal node and has a raw value
61  // TODO: Add handling for nested objects by determining if property.second is another ptree
62  std::string value = property_tree.data();
63 
64  if (value == "true" || value == "false") {
65  type = PropertyType::Boolean;
66  boolean = value == "true";
67  }
68  else {
69  //Try natural number first since double cant cast to int/long
70  try{
71  natural_number = boost::lexical_cast<long>(value);
72  type = PropertyType::Natural;
73  }
74  catch (boost::bad_lexical_cast &e)
75  {
76  try {
77  //Try to cast to double/real next
78  real_number = boost::lexical_cast<double>(value);
79  type = PropertyType::Real;
80  }
81  catch (boost::bad_lexical_cast & e)
82  {
83  //At this point, we are left with string option
84  string = value;
85  type = PropertyType::String;
86  }
87  }
88  }
89  }
90  else {
91  // This isn't a terminal node, therefore represents an object or array
92  for (auto &property : property_tree) {
93  if (property.first.empty()) {
94  type = PropertyType::List;
95  value_list.push_back(JSONProperty(value_key, property.second));
96  }
97  else {
98  type = PropertyType::Object;
99  values.emplace(property.first, JSONProperty(property.first, property.second));
100  }
101  }
102  }

183  JSONProperty(std::string value_key, std::string value) {
184  key = value_key;
185  string = value;
186 
187  if (value == "true" || value == "false") {
188  type = PropertyType::Boolean;
189  boolean = value == "true";
190  }
191  else {
192  bool is_numeric = true;
193  bool is_real = true;
194  bool decimal_already_hit = false;
195 
196  for(int character_index = 0; character_index < value.length(); character_index++) {
197  char character = value[character_index];
198 
199  // If the first character is a '0' or isn't a digit, the whole value cannot be a number
200  if (character_index == 0 && character == '0') {
201  is_numeric = false;
202  }
203  else if (character != '.' && !std::isdigit(character)) {
204  // If this character isn't a decimal point and isn't a digit, the whole value cannot be a number
205  is_numeric = false;
206  is_real = false;
207  break;
208  }
209  else if (character == '.' && decimal_already_hit) {
210  // If this character is a decimal point, but we've already seen one, the whole value cannot be a number
211  is_real = false;
212  break;
213  }
214  else if (character == '.') {
215  // If a decimal point is seen, the whole value cannot be an integer
216  is_numeric = false;
217  decimal_already_hit = true;
218  }
219  }
220 
221  // If the value can be represented as a whole number, we want to go with that
222  if (is_numeric) {
223  type = PropertyType::Natural;
224  natural_number = std::stol(value);
225  }
226  else if (is_real) {
227  type = PropertyType::Real;
228  real_number = std::stod(value);
229  }
230  else {
231  // Otherwise we'll store everything as a raw string
232  type = PropertyType::String;
233  string = value;
234  }
235  }
236  }
