219 size_t get_ts_index_for_time(const time_t &epoch_time) override {
220 if (epoch_time < start_date_time_epoch) {
221 throw std::out_of_range("Forcing had bad pre-start time for index query: " + std::to_string(epoch_time));
222 }
223 size_t i = 0;
224 // 1 hour
225 time_t seconds_in_time_step = 3600;
226 time_t time = start_date_time_epoch;
227 while (epoch_time >= time + seconds_in_time_step && time < end_date_time_epoch) {
228 i++;
229 time += seconds_in_time_step;
230 }
231 // The end_date_time_epoch is the epoch value of the BEGINNING of the last time step, not its end.
232 // I.e., to make sure we cover it, we have to go another time step beyond.
233 if (time >= end_date_time_epoch + 3600) {
234 throw std::out_of_range("Forcing had bad beyond-end time for index query: " + std::to_string(epoch_time));
235 }
236 else {
237 return i;
238 }

253 //double get_value(const std::string &output_name, const time_t &init_time, const long &duration_s,
254 // const std::string &output_units) override
255 double get_value(const CatchmentAggrDataSelector& selector, data_access::ReSampleMethod m) override
256 {
257 size_t current_index;
258
259 time_t init_time = selector.get_init_time();
260 long duration_s = selector.get_duration_secs();
261 const std::string& output_name = selector.get_variable_name();
262
263 long time_remaining = duration_s;
264 try {
265 current_index = get_ts_index_for_time(init_time);
266 }
267 catch (const std::out_of_range &e) {
268 throw std::out_of_range("Forcing had bad init_time " + std::to_string(init_time) + " for value request");
269 }
270
271 std::vector<double> involved_time_step_values;
272 std::vector<long> involved_time_step_seconds;
273 long ts_involved_s;
274
275 time_t first_time_step_start_epoch = start_date_time_epoch + (current_index * 3600);
276 // Handle the first time step differently, since we need to do more to figure out how many seconds came from it
277 // Total time step size minus the offset of the beginning, before the init time
278 ts_involved_s = 3600 - (init_time - first_time_step_start_epoch);
279
280 involved_time_step_seconds.push_back(ts_involved_s);
281 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
282 time_remaining -= ts_involved_s;
283 current_index++;
284
285 while (time_remaining > 0) {
286 ts_involved_s = time_remaining > 3600 ? 3600 : time_remaining;
287 involved_time_step_seconds.push_back(ts_involved_s);
288 involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
289 time_remaining -= ts_involved_s;
290 current_index++;
291 }
292 double value = 0;
293 // TODO change this to use resample methode instead
294 for (size_t i = 0; i < involved_time_step_values.size(); ++i) {
295 if (is_param_sum_over_time_step(output_name))
296 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / 3600.0);
297 else
298 value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / (double)duration_s);
299 }
300 return value;
301 }

310 inline double get_value_for_param_name(const std::string& name, int index) {
311 if (index >= precipitation_rate_meters_per_second_vector.size() || index >= AORC_vector.size()) {
312 throw std::out_of_range("Forcing had bad index " + std::to_string(index) + " for value lookup of " + name);
313 }
314
315 if (name == AORC_FIELD_NAME_PRECIP_RATE) {
316 return precipitation_rate_meters_per_second_vector.at(index);
317 }
318 if (name == AORC_FIELD_NAME_SOLAR_SHORTWAVE || name == CSDMS_STD_NAME_SOLAR_SHORTWAVE) {
319 return AORC_vector.at(index).DSWRF_surface_W_per_meters_squared;
320 }
321 if (name == AORC_FIELD_NAME_SOLAR_LONGWAVE || name == CSDMS_STD_NAME_SOLAR_LONGWAVE) {
322 return AORC_vector.at(index).DLWRF_surface_W_per_meters_squared;
323 }
324 if (name == AORC_FIELD_NAME_PRESSURE_SURFACE || name == CSDMS_STD_NAME_SURFACE_AIR_PRESSURE) {
325 return AORC_vector.at(index).PRES_surface_Pa;
326 }
327 if (name == AORC_FIELD_NAME_TEMP_2M_AG || name == CSDMS_STD_NAME_SURFACE_TEMP) {
328 return AORC_vector.at(index).TMP_2maboveground_K;
329 }
330 if (name == AORC_FIELD_NAME_APCP_SURFACE || name == CSDMS_STD_NAME_LIQUID_EQ_PRECIP_RATE) {
331 return AORC_vector.at(index).APCP_surface_kg_per_meters_squared;
332 }
333 if (name == AORC_FIELD_NAME_WIND_U_10M_AG || name == CSDMS_STD_NAME_WIND_U_X) {
334 return AORC_vector.at(index).UGRD_10maboveground_meters_per_second;
335 }
336 if (name == AORC_FIELD_NAME_WIND_V_10M_AG || name == CSDMS_STD_NAME_WIND_V_Y) {
337 return AORC_vector.at(index).VGRD_10maboveground_meters_per_second;
338 }
339 if (name == AORC_FIELD_NAME_SPEC_HUMID_2M_AG || name == CSDMS_STD_NAME_HUMIDITY) {
340 return AORC_vector.at(index).SPFH_2maboveground_kg_per_kg;
341 }
342 else {
343 throw std::runtime_error("Cannot get forcing value for unrecognized parameter name '" + name + "'.");
344 }

351 inline void check_forcing_vector_index_bounds()
352 {
353 //Check if forcing index is less than zero and if so, set to zero.
354 if (forcing_vector_index < 0)
355 {
356 forcing_vector_index = 0;
358 cout << "WARNING: Forcing vector index is less than zero. Therefore, setting index to zero." << endl;
359 }
360
361 //Check if forcing index is greater than or equal to the size of the size of the precipiation vector and if so, set to zero.
362 else if (forcing_vector_index >= precipitation_rate_meters_per_second_vector.size())
363 {
364 forcing_vector_index = precipitation_rate_meters_per_second_vector.size() - 1;
366 cout << "WARNING: Reached beyond the size of the forcing vector. Therefore, setting index to last value of the vector." << endl;
367 }
368
369 return;
370 }

622 void read_forcing(string file_name)
623 {
624 //Call CSVReader constuctor
625 CSVReader reader(file_name);
626
627 //Get the data from CSV File
628 std::vector<std::vector<std::string> > data_list = reader.getData();
629
630 //Iterate through CSV starting on the third row
631 for (int i = 2; i < data_list.size(); i++)
632 {
633 //Row vector
634 std::vector<std::string>& vec = data_list[i];
635
636 //Declare pointer to struct for the current row date-time utc
637 struct tm *current_row_date_time_utc;
638
639 //Allocate memory to struct for the current row date-time utc
640 current_row_date_time_utc = new tm();
641
642 //Year
643 string year_str = vec[0];
644 int year = stoi(year_str);
645 current_row_date_time_utc->tm_year = year - 1900;
646
647 //Month
648 string month_str = vec[1];
649 int month = stoi(month_str);
650 current_row_date_time_utc->tm_mon = month - 1;
651
652 //Day
653 string day_str = vec[2];
654 int day = stoi(day_str);
655 current_row_date_time_utc->tm_mday = day;
656
657 //Hour
658 string hour_str = vec[3];
659 int hour = stoi(hour_str);
660 current_row_date_time_utc->tm_hour = hour;
661
662 //Convert current row date-time utc to epoch time
663 time_t current_row_date_time_epoch = timegm(current_row_date_time_utc);
664
665 //If the current row date-time is within the model date-time range, then add precipitation to vector
666 if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
667 {
668 //Precipitation
669 string precip_str = vec[5];
670
671 //Convert from string to double and from mm/hr to m/s
672 double precip = boost::lexical_cast<double>(precip_str) / (1000 * 3600);
673
674 //Add precip to vector
675 precipitation_rate_meters_per_second_vector.push_back(precip);
676 }
677
678 //Free memory from struct
679 delete current_row_date_time_utc;
680 }
681 }

688 void read_forcing_aorc(string file_name)
689 {
690 is_aorc_data = true;
691 //Call CSVReader constuctor
692 CSVReader reader(file_name);
693
694 //Get the data from CSV File
695 std::vector<std::vector<std::string> > data_list = reader.getData();
696 //Some quick validation of the data we read
697 if( data_list.size() > 0 ){
698 auto header = data_list[0];
699 if( header.size() != 10 ){
700 throw std::runtime_error("Error: Forcing data " + file_name +
701 " does not have the expected number of columns. Did you mean to use "+
702 "csvPerFeature forcing provider instead?" SOURCE_LOC);
703 }
704 }
705 else {
706 throw std::runtime_error("Error: Forcing data " + file_name +
707 " does not have readable rows." SOURCE_LOC);
708 }
709
710 //Iterate through CSV starting on the second row
711 for (int i = 1; i < data_list.size(); i++)
712 {
713 //Row vector
714 std::vector<std::string>& vec = data_list[i];
715
716 //Declare struct for the current row date-time
717 struct tm current_row_date_time_utc;
718
719 //Allocate memory to struct for the current row date-time
720 current_row_date_time_utc = tm();
721
722 //Grab time string from first column
723 string time_str = vec[0];
724
725 //Convert time string to time struct
726 strptime(time_str.c_str(), "%Y-%m-%d %H:%M:%S", &current_row_date_time_utc);
727
728 //Convert current row date-time UTC to epoch time
729 time_t current_row_date_time_epoch = timegm(&current_row_date_time_utc);
730
731 //Ensure that forcing data covers the entire model period. Otherwise, throw an error.
732 if (i == 1 && start_date_time_epoch < current_row_date_time_epoch)
733 {
735 //cout << "WARNING: Forcing data begins after the model start time." << endl;
736 struct tm start_date_tm;
737 gmtime_r(&start_date_time_epoch, &start_date_tm);
738
739 char tm_buff[128];
740 strftime(tm_buff, 128, "%Y-%m-%d %H:%M:%S", &start_date_tm);
741 throw std::runtime_error("Error: Forcing data " + file_name + " begins after the model start time:" + std::string(tm_buff) + " < " + time_str);
742 }
743
744
745 else if (i == data_list.size() - 1 && current_row_date_time_epoch < end_date_time_epoch)
747 cout << "WARNING: Forcing data ends before the model end time." << endl;
748 //throw std::runtime_error("Error: Forcing data ends before the model end time.");
749
750 //If the current row date-time is within the model date-time range, then add precipitation to vector
751 if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
752 {
753 //Grab data from columns
754 string APCP_surface_str = vec[1];
755 string DLWRF_surface_str = vec[2];
756 string DSWRF_surface_str = vec[3];
757 string PRES_surface_str = vec[4];
758 string SPFH_2maboveground_str = vec[5];
759 string TMP_2maboveground_str = vec[6];
760 string UGRD_10maboveground_str = vec[7];
761 string VGRD_10maboveground_str = vec[8];
762 string precip_rate_str = vec[9];
763
764 //Declare AORC struct
765 AORC_data AORC;
766
767 //Convert from strings to doubles and add to AORC struct
768 AORC.APCP_surface_kg_per_meters_squared = boost::lexical_cast<double>(APCP_surface_str);
769 AORC.DLWRF_surface_W_per_meters_squared = boost::lexical_cast<double>(DLWRF_surface_str);
770 AORC.DSWRF_surface_W_per_meters_squared = boost::lexical_cast<double>(DSWRF_surface_str);
771 AORC.PRES_surface_Pa = boost::lexical_cast<double>(PRES_surface_str);
772 AORC.SPFH_2maboveground_kg_per_kg = boost::lexical_cast<double>(SPFH_2maboveground_str);
773 AORC.TMP_2maboveground_K = boost::lexical_cast<double>(TMP_2maboveground_str);
774 AORC.UGRD_10maboveground_meters_per_second = boost::lexical_cast<double>(UGRD_10maboveground_str);
775 AORC.VGRD_10maboveground_meters_per_second = boost::lexical_cast<double>(VGRD_10maboveground_str);
776
777
778 //Add AORC struct to AORC vector
779 AORC_vector.push_back(AORC);
780
781 //Convert precip_rate from string to double
782 double precip_rate = boost::lexical_cast<double>(precip_rate_str);
783
784 //Add data to vectors
785 precipitation_rate_meters_per_second_vector.push_back(precip_rate);
