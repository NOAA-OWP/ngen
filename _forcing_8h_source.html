241  size_t get_ts_index_for_time(const time_t &epoch_time) override {
242  if (epoch_time < start_date_time_epoch) {
243  throw std::out_of_range("Forcing had bad pre-start time for index query: " + std::to_string(epoch_time));
244  }
245  size_t i = 0;
246  // 1 hour
247  time_t seconds_in_time_step = 3600;
248  time_t time = start_date_time_epoch;
249  while (epoch_time >= time + seconds_in_time_step && time < end_date_time_epoch) {
250  i++;
251  time += seconds_in_time_step;
252  }
253  // The end_date_time_epoch is the epoch value of the BEGINNING of the last time step, not its end.
254  // I.e., to make sure we cover it, we have to go another time step beyond.
255  if (time >= end_date_time_epoch + 3600) {
256  throw std::out_of_range("Forcing had bad beyond-end time for index query: " + std::to_string(epoch_time));
257  }
258  else {
259  return i;
260  }

275  double get_value(const std::string &output_name, const time_t &init_time, const long &duration_s,
276  const std::string &output_units) override
277  {
278  size_t current_index;
279  long time_remaining = duration_s;
280  try {
281  current_index = get_ts_index_for_time(init_time);
282  }
283  catch (const std::out_of_range &e) {
284  throw std::out_of_range("Forcing had bad init_time " + std::to_string(init_time) + " for value request");
285  }

287  std::vector<double> involved_time_step_values;
288  std::vector<long> involved_time_step_seconds;
289  long ts_involved_s;
290 
291  time_t first_time_step_start_epoch = start_date_time_epoch + (current_index * 3600);
292  // Handle the first time step differently, since we need to do more to figure out how many seconds came from it
293  // Total time step size minus the offset of the beginning, before the init time
294  ts_involved_s = 3600 - (init_time - first_time_step_start_epoch);
295 
296  involved_time_step_seconds.push_back(ts_involved_s);
297  involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
298  time_remaining -= ts_involved_s;
299  current_index++;
300 
301  while (time_remaining > 0) {
302  ts_involved_s = time_remaining > 3600 ? 3600 : time_remaining;
303  involved_time_step_seconds.push_back(ts_involved_s);
304  involved_time_step_values.push_back(get_value_for_param_name(output_name, current_index));
305  time_remaining -= ts_involved_s;
306  current_index++;
307  }
308  double value = 0;
309  for (size_t i = 0; i < involved_time_step_values.size(); ++i) {
310  if (is_param_sum_over_time_step(output_name))
311  value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / 3600.0);
312  else
313  value += involved_time_step_values[i] * ((double)involved_time_step_seconds[i] / (double)duration_s);
314  }
315  return value;
316  }

627  void read_forcing(string file_name)
628  {
629  //Call CSVReader constuctor
630  CSVReader reader(file_name);
631 
632  //Get the data from CSV File
633  std::vector<std::vector<std::string> > data_list = reader.getData();
634 
635  //Iterate through CSV starting on the third row
636  for (int i = 2; i < data_list.size(); i++)
637  {
638  //Row vector
639  std::vector<std::string>& vec = data_list[i];
640 
641  //Declare pointer to struct for the current row date-time utc
642  struct tm *current_row_date_time_utc;
643 
644  //Allocate memory to struct for the current row date-time utc
645  current_row_date_time_utc = new tm();
646 
647  //Year
648  string year_str = vec[0];
649  int year = stoi(year_str);
650  current_row_date_time_utc->tm_year = year - 1900;
651 
652  //Month
653  string month_str = vec[1];
654  int month = stoi(month_str);
655  current_row_date_time_utc->tm_mon = month - 1;
656 
657  //Day
658  string day_str = vec[2];
659  int day = stoi(day_str);
660  current_row_date_time_utc->tm_mday = day;
661 
662  //Hour
663  string hour_str = vec[3];
664  int hour = stoi(hour_str);
665  current_row_date_time_utc->tm_hour = hour;
666 
667  //Convert current row date-time utc to epoch time
668  time_t current_row_date_time_epoch = timegm(current_row_date_time_utc);
669 
670  //If the current row date-time is within the model date-time range, then add precipitation to vector
671  if (start_date_time_epoch <= current_row_date_time_epoch && current_row_date_time_epoch <= end_date_time_epoch)
672  {
673  //Precipitation
674  string precip_str = vec[5];
675 
676  //Convert from string to double and from mm/hr to m/s
677  double precip = boost::lexical_cast<double>(precip_str) / (1000 * 3600);
678 
679  //Add precip to vector
680  precipitation_rate_meters_per_second_vector.push_back(precip);
681  }
682 
683  //Free memory from struct
684  delete current_row_date_time_utc;
685  }
686  }

694  {
695  is_aorc_data = true;
696  //Call CSVReader constuctor
697  CSVReader reader(file_name);
698 
699  //Get the data from CSV File
700  std::vector<std::vector<std::string> > data_list = reader.getData();
701 
702  //Iterate through CSV starting on the second row
703  for (int i = 1; i < data_list.size(); i++)
704  {
705  //Row vector
706  std::vector<std::string>& vec = data_list[i];
707 
708  //Declare struct for the current row date-time
709  struct tm current_row_date_time_utc;
710 
711  //Allocate memory to struct for the current row date-time
712  current_row_date_time_utc = tm();
713 
714  //Grab time string from first column
715  string time_str = vec[0];
716 
717  //Convert time string to time struct
718  strptime(time_str.c_str(), "%Y-%m-%d %H:%M:%S", &current_row_date_time_utc);
719 
720  //Convert current row date-time UTC to epoch time
721  time_t current_row_date_time_epoch = timegm(&current_row_date_time_utc);
722 
723  //Ensure that forcing data covers the entire model period. Otherwise, throw an error.
724  if (i == 1 && start_date_time_epoch < current_row_date_time_epoch)
725  {
727  //cout << "WARNING: Forcing data begins after the model start time." << endl;
728  struct tm start_date_tm;
729  gmtime_r(&start_date_time_epoch, &start_date_tm);
