67 void link_external(geojson::Feature feature){
68
69 if(type == "bmi_multi"){
70 std::vector<geojson::JSONProperty> tmp;
71 for(auto& n : nested){
72 //Iterate and link any nested modules with this feature
73 if(n.parameters.count("model_params")){
74 n.link_external(feature);
75 }
76 //Need a temporary map to hold the updated formulation properties in
77 geojson::PropertyMap map = {};
78 map.emplace("name", geojson::JSONProperty("name", n.type));
79 map.emplace("params", geojson::JSONProperty("", n.parameters));
80 tmp.push_back(geojson::JSONProperty("", map));
81 }
82 //Reset the bmi_multi modules with the now linked module definitions
83 parameters.at("modules") = geojson::JSONProperty("modules", tmp);
84 return;
85 }
86 //Short circut
87 if(parameters.count("model_params") < 1 ) return;
88 //Have some model params, check to see if any should be linked to the hyrdofabric feature
89 geojson::PropertyMap attr = parameters.at("model_params").get_values();
90 for (decltype(auto) param : attr) {
91 // Check for type to short-circuit. If param.second is not an object, `.has_key()` will throw
92 if (param.second.get_type() != geojson::PropertyType::Object || !param.second.has_key("source")) {
93 attr.emplace(param.first, param.second);
94 continue;
95 }
96
97 decltype(auto) param_source = param.second.at("source");
98 decltype(auto) param_source_name = param_source.as_string();
99 if (param_source_name != "hydrofabric") {
100 // TODO: temporary until the logic for alternative sources is designed
101 throw std::logic_error("ERROR: 'model_params' source `" + param_source_name + "` not currently supported. Only `hydrofabric` is supported.");
102 }
103
104 // Property name in the feature properties is either
105 // the value of key "from", or has the same name as
106 // the expected model parameter key
107 decltype(auto) param_name = param.second.has_key("from")
108 ? param.second.at("from").as_string()
109 : param.first;
110
111 if (feature->has_property(param_name)) {
112 auto catchment_attribute = feature->get_property(param_name);
113
114 // Use param.first in the `.emplace` calls instead of param_name since
115 // the expected name is given by the key of the model_params values.
116 switch (catchment_attribute.get_type()) {
117 case geojson::PropertyType::List:
118 case geojson::PropertyType::Object:
119 // TODO: Should list/object values be passed to model parameters?
120 // Typically, feature properties *should* be scalars.
121 std::cerr << "WARNING: property type " << static_cast<int>(catchment_attribute.get_type()) << " not allowed as model parameter. "
122 << "Must be one of: Natural (int), Real (double), Boolean, or String" << '\n';
123 break;
124 default:
125 attr.at(param.first) = geojson::JSONProperty(param.first, catchment_attribute);
126 }
127 }
128 }
129 parameters.at("model_params") = geojson::JSONProperty("model_params", attr);
