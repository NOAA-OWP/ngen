15struct conceptual_reservoir
16{
17// this data structure describes a nonlinear reservoir having two outlets, one primary with an activation
18// threshold that may be zero, and a secondary outlet with a threshold that may be zero
19// this will also simulate a linear reservoir by setting the exponent parameter to 1.0 iff is_exponential==FALSE
20// iff is_exponential==TRUE, then it uses the exponential discharge function from the NWM V2.0 forumulation
21// as the primary discharge with a zero threshold, and does not calculate a secondary discharge.
22//--------------------------------------------------------------------------------------------------
23 // TODO: adjust this (and usages) to bool type once verfied and automated testing is in place.
24 int is_exponential; // set this true TRUE to use the exponential form of the discharge equation
25 double storage_max_m; // maximum storage in this reservoir
26 double storage_m; // state variable.
27 double coeff_primary; // the primary outlet
28 double exponent_primary;
29 double storage_threshold_primary_m;
30 double storage_threshold_secondary_m;
31 double coeff_secondary;
32 double exponent_secondary;
33};
