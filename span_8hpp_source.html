168template<class T, std::size_t E>
169class span {
170public:
171 typedef T element_type;
172 typedef typename std::remove_cv<T>::type value_type;
173 typedef std::size_t size_type;
174 typedef std::ptrdiff_t difference_type;
175 typedef T* pointer;
176 typedef const T* const_pointer;
177 typedef T& reference;
178 typedef const T& const_reference;
179 typedef T* iterator;
180 typedef const T* const_iterator;
181 typedef std::reverse_iterator<T*> reverse_iterator;
182 typedef std::reverse_iterator<const T*> const_reverse_iterator;
183
184 static constexpr std::size_t extent = E;
185
186 template<std::size_t N = E,
187 typename std::enable_if<N == dynamic_extent || N == 0, int>::type = 0>
188 constexpr span() noexcept
189 : s_(0, 0) { }
190
191 template<class I,
192 typename std::enable_if<E == dynamic_extent &&
193 detail::span_convertible<I, T>::value, int>::type = 0>
194 constexpr span(I* f, size_type c)
195 : s_(f, c) { }
196
197 template<class I,
198 typename std::enable_if<E != dynamic_extent &&
199 detail::span_convertible<I, T>::value, int>::type = 0>
200 explicit constexpr span(I* f, size_type c)
201 : s_(f, c) { }
202
203 template<class I, class L,
204 typename std::enable_if<E == dynamic_extent &&
205 detail::span_convertible<I, T>::value, int>::type = 0>
206 constexpr span(I* f, L* l)
207 : s_(f, l - f) { }
208
209 template<class I, class L,
210 typename std::enable_if<E != dynamic_extent &&
211 detail::span_convertible<I, T>::value, int>::type = 0>
212 explicit constexpr span(I* f, L* l)
213 : s_(f, l - f) { }
214
215 template<std::size_t N,
216 typename std::enable_if<detail::span_capacity<E, N>::value,
217 int>::type = 0>
218 constexpr span(typename std::enable_if<true, T>::type (&a)[N]) noexcept
219 : s_(a, N) { }
220
221 template<class U, std::size_t N,
222 typename std::enable_if<detail::span_compatible<T, E, U, N>::value,
223 int>::type = 0>
224 constexpr span(std::array<U, N>& a) noexcept
225 : s_(a.data(), N) { }
226
227 template<class U, std::size_t N,
228 typename std::enable_if<detail::span_compatible<T, E, const U,
229 N>::value, int>::type = 0>
230 constexpr span(const std::array<U, N>& a) noexcept
231 : s_(a.data(), N) { }
232
233 template<class R,
234 typename std::enable_if<E == dynamic_extent &&
235 detail::span_is_range<R, T>::value, int>::type = 0>
236 constexpr span(R&& r) noexcept(noexcept(boost::data(r)) &&
237 noexcept(r.size()))
238 : s_(boost::data(r), r.size()) { }
239
240 template<class R,
241 typename std::enable_if<E != dynamic_extent &&
242 detail::span_is_range<R, T>::value, int>::type = 0>
243 explicit constexpr span(R&& r) noexcept(noexcept(boost::data(r)) &&
244 noexcept(r.size()))
245 : s_(boost::data(r), r.size()) { }
246
247 template<class U, std::size_t N,
248 typename std::enable_if<detail::span_implicit<E, N>::value &&
249 detail::span_copyable<T, E, U, N>::value, int>::type = 0>
250 constexpr span(const span<U, N>& s) noexcept
251 : s_(s.data(), s.size()) { }

253 template<class U, std::size_t N,
254 typename std::enable_if<!detail::span_implicit<E, N>::value &&
255 detail::span_copyable<T, E, U, N>::value, int>::type = 0>
256 explicit constexpr span(const span<U, N>& s) noexcept
257 : s_(s.data(), s.size()) { }

259 template<std::size_t C>
260 constexpr span<T, C> first() const {
261 static_assert(C <= E, "Count <= Extent");
262 return span<T, C>(s_.p, C);
263 }
264
265 template<std::size_t C>
266 constexpr span<T, C> last() const {
267 static_assert(C <= E, "Count <= Extent");
268 return span<T, C>(s_.p + (s_.n - C), C);
269 }

271 template<std::size_t O, std::size_t C = dynamic_extent>
272 constexpr typename std::enable_if<C == dynamic_extent,
273 span<T, detail::span_sub<E, O>::value> >::type subspan() const {
274 static_assert(O <= E, "Offset <= Extent");
275 return span<T, detail::span_sub<E, O>::value>(s_.p + O, s_.n - O);
276 }
277
278 template<std::size_t O, std::size_t C = dynamic_extent>
279 constexpr typename std::enable_if<C != dynamic_extent,
280 span<T, C> >::type subspan() const {
281 static_assert(O <= E && C <= E - O,
282 "Offset <= Extent && Count <= Extent - Offset");
283 return span<T, C>(s_.p + O, C);
284 }

286 constexpr span<T, dynamic_extent> first(size_type c) const {
287 return span<T, dynamic_extent>(s_.p, c);
288 }
289
290 constexpr span<T, dynamic_extent> last(size_type c) const {
291 return span<T, dynamic_extent>(s_.p + (s_.n - c), c);
292 }
293
294 constexpr span<T, dynamic_extent> subspan(size_type o,
295 size_type c = dynamic_extent) const {
296 return span<T, dynamic_extent>(s_.p + o,
297 c == dynamic_extent ? s_.n - o : c);
298 }
299
300 constexpr size_type size() const noexcept {
301 return s_.n;
302 }
303
304 constexpr size_type size_bytes() const noexcept {
305 return s_.n * sizeof(T);
306 }
307
308 constexpr bool empty() const noexcept {
309 return s_.n == 0;
310 }
311
312 constexpr reference operator[](size_type i) const {
313 return s_.p[i];
314 }
315
316 constexpr reference front() const {
317 return *s_.p;
318 }
319
320 constexpr reference back() const {
321 return s_.p[s_.n - 1];
322 }
323
324 constexpr pointer data() const noexcept {
325 return s_.p;
326 }
327
328 constexpr iterator begin() const noexcept {
329 return s_.p;
330 }
331
332 constexpr iterator end() const noexcept {
333 return s_.p + s_.n;
334 }
335
336 constexpr reverse_iterator rbegin() const noexcept {
337 return reverse_iterator(s_.p + s_.n);
338 }
339
340 constexpr reverse_iterator rend() const noexcept {
341 return reverse_iterator(s_.p);
342 }
343
344 constexpr const_iterator cbegin() const noexcept {
345 return s_.p;
346 }
347
348 constexpr const_iterator cend() const noexcept {
349 return s_.p + s_.n;

352 constexpr const_reverse_iterator crbegin() const noexcept {
353 return const_reverse_iterator(s_.p + s_.n);
354 }
355
356 constexpr const_reverse_iterator crend() const noexcept {
357 return const_reverse_iterator(s_.p);
358 }

360private:
361 detail::span_store<T, E> s_;
362};
